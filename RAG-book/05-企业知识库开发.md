# 第五章：企业知识库开发

## 5.1 需求分析

企业知识库的构建始于对业务需求的深入理解。与通用问答系统不同，企业知识库需要精准匹配组织的信息架构和访问模式。

### 5.1.1 核心问题清单

在启动项目前，必须回答以下关键问题：

| 维度 | 关键问题 | 典型答案 |
|------|---------|---------|
| 文档类型 | 需要处理哪些格式的文档？ | PDF、Word、Confluence页面、代码仓库 |
| 数据规模 | 预期文档数量和总大小？ | 10万篇文档，约50GB |
| 查询模式 | 用户的主要查询类型？ | 事实查询、流程指引、技术排障 |
| 用户画像 | 谁在使用？技术背景如何？ | 内部员工、客服、合作伙伴 |
| 更新频率 | 知识更新的频率如何？ | 每日增量，季度全量 |
| 安全合规 | 是否有数据主权要求？ | 金融行业需本地部署 |

```python
from unstructured.partition.auto import partition
from unstructured.partition.pdf import partition_pdf
from langchain_core.documents import Document

def load_scanned_pdf(file_path: str) -> list[Document]:
    """通过OCR处理扫描PDF文档"""
    elements = partition_pdf(
        filename=file_path,
        strategy="hi_res",          # 使用高精度OCR模型
        infer_table_structure=True,  # 推断表格结构
        ocr_languages=["chi_sim", "eng"],  # 中英文OCR
    )

    documents = []
    for element in elements:
        doc = Document(
            page_content=element.text,
            metadata={
                "source": file_path,
                "element_type": element.category,  # Title, NarrativeText, Table, etc.
                "coordinates": getattr(element.metadata, "coordinates", None),
            }
        )
        documents.append(doc)
    return documents
```

## 5.3 企业级分块策略

分块质量直接决定检索精度。企业场景需要更精细的分块策略。

### 5.3.1 文档结构感知分块

```python
from langchain_text_splitters import MarkdownHeaderTextSplitter, RecursiveCharacterTextSplitter

def structure_aware_chunking(markdown_text: str) -> list[Document]:
    """按文档结构（标题层级）分块"""
    # 第一步：按Markdown标题切分
    headers_to_split_on = [
        ("#", "h1"),
        ("##", "h2"),
        ("###", "h3"),
    ]
    md_splitter = MarkdownHeaderTextSplitter(
        headers_to_split_on=headers_to_split_on,
        strip_headers=False,  # 保留标题文本，提供上下文
    )
    md_chunks = md_splitter.split_text(markdown_text)

    # 第二步：对过长的节继续切分
    text_splitter = RecursiveCharacterTextSplitter(
        chunk_size=800,
        chunk_overlap=100,
        separators=["\n\n", "\n", "。", "！", "？", ".", " ", ""],
    )
    final_chunks = text_splitter.split_documents(md_chunks)

    return final_chunks
```

### 5.3.2 父子分块（Parent-Child Chunking）

这是企业RAG中最重要的分块策略：用小块检索以保证精准度，用大块提供完整上下文。

```python
from langchain.storage import InMemoryStore
from langchain_text_splitters import RecursiveCharacterTextSplitter
from langchain_community.vectorstores import Chroma
from langchain_openai import OpenAIEmbeddings
import uuid

def create_parent_child_chunks(documents: list[Document]):
    """创建父子分块：小块用于检索，大块用于生成"""
    # 父分块器（大块，提供完整上下文）
    parent_splitter = RecursiveCharacterTextSplitter(
        chunk_size=2000,
        chunk_overlap=200,
    )
    # 子分块器（小块，精准检索）
    child_splitter = RecursiveCharacterTextSplitter(
        chunk_size=300,
        chunk_overlap=50,
    )

    parent_store = InMemoryStore()  # 存储父文档
    child_chunks = []

    for doc in documents:
        parent_id = str(uuid.uuid4())
        parent_chunks = parent_splitter.split_documents([doc])

        for parent_chunk in parent_chunks:
            # 存储父文档
            parent_store.mset([(parent_id, parent_chunk)])

            # 生成子文档
            children = child_splitter.split_documents([parent_chunk])
            for child in children:
                child.metadata["parent_id"] = parent_id
                child_chunks.append(child)

    return child_chunks, parent_store


def retrieve_with_parent_context(query: str, vectorstore, parent_store, k: int = 5):
    """检索子块，返回父块上下文"""
    # 检索子块
    child_results = vectorstore.similarity_search(query, k=k)

    # 获取唯一父块
    parent_ids = list(set(doc.metadata["parent_id"] for doc in child_results))
    parent_docs = parent_store.mget(parent_ids)

    return parent_docs
```

```python
def metadata_aware_chunking(doc: Document) -> list[Document]:
    """确保分块不切断关键结构，并传播元数据"""
    splitter = RecursiveCharacterTextSplitter(
        chunk_size=600,
        chunk_overlap=80,
        separators=["\n\n## ", "\n\n### ", "\n\n", "\n", "。", ""],
    )
    chunks = splitter.split_documents([doc])

    # 为每个块添加上下文前缀
    for i, chunk in enumerate(chunks):
        chunk.metadata["chunk_index"] = i
        chunk.metadata["total_chunks"] = len(chunks)

        chunk.page_content = f"[文档: {doc.metadata.get('title', '未知')}]{chunk.page_content}"
```

## 5.4 权限控制

企业知识库必须实现严格的权限控制，确保用户只能检索到有权限访问的文档。

### 5.4.1 文档级访问控制

```python
from langchain_core.documents import Document
from typing import Optional

class AccessControlledDocument(Document):
    """带权限标签的文档"""
    class Config:
        arbitrary_types_allowed = True

    def __init__(self, **data):
        super().__init__(**data)
        # 在metadata中注入权限信息
        self.metadata.setdefault("allowed_groups", [])
        self.metadata.setdefault("access_level", "internal")

def filter_documents_by_user(documents: list[Document], user_groups: list[str]) -> list[Document]:
    """根据用户组过滤文档"""
    filtered = []
    for doc in documents:
        allowed_groups = doc.metadata.get("allowed_groups", [])
        # 文档允许的组与用户所属组有交集
        if set(allowed_groups) & set(user_groups):
            filtered.append(doc)
    return filtered
```

### 5.4.2 向量搜索中的元数据过滤

```python
from langchain_milvus import Milvus

def search_with_access_control(
    vectorstore: Milvus,
    query: str,
    user_groups: list[str],
    user_department: str,
    k: int = 5,
):
    """基于用户权限的向量搜索"""
    # 构建Milvus过滤表达式
    # 允许条件：文档access_level <= 用户等级 OR 用户组在allowed_groups中
    access_level_map = {"public": 0, "internal": 1, "confidential": 2, "restricted": 3}
    user_level = max(
        access_level_map.get(g, 0) for g in user_groups
    )

    filter_expr = (
        f'access_level_int <= {user_level} || '
        f'array_contains_any(allowed_groups, {user_groups}) || '
        f'department == "{user_department}"'
    )

    results = vectorstore.similarity_search(
        query=query,
        k=k,
        expr=filter_expr,
    )
    return results
```

### 5.4.3 分层权限模型

```python
from dataclasses import dataclass
from enum import IntEnum

class ClassificationLevel(IntEnum):
    """文档密级"""
    PUBLIC = 0
    INTERNAL = 1
    CONFIDENTIAL = 2
    RESTRICTED = 3

@dataclass
class UserContext:
    """用户安全上下文"""
    user_id: str
    groups: list[str]
    department: str
    clearance_level: ClassificationLevel
    allowed_projects: list[str]

    def can_access(self, doc: Document) -> bool:
        """判断用户是否能访问该文档"""
        doc_level = ClassificationLevel[
            doc.metadata.get("classification", "INTERNAL").upper()
        ]
        if doc_level > self.clearance_level:
            return False

        # 检查部门限制
        doc_dept = doc.metadata.get("department")
        if doc_dept and doc_dept != self.department:
            # 跨部门访问需在allowed_groups中
            allowed_groups = doc.metadata.get("allowed_groups", [])
            if not set(allowed_groups) & set(self.groups):
                return False

        # 检查项目级限制
        doc_project = doc.metadata.get("project")
        if doc_project and doc_project not in self.allowed_projects:
            return False

        return True
```

## 5.6 监控与运维

### 5.6.1 检索质量指标

```python
from dataclasses import dataclass
from typing import Optional

@dataclass
class RetrievalMetrics:
    """检索质量评估指标"""
    precision_at_k: float       # Top-K中相关文档占比
    recall_at_k: float          # 相关文档被召回的比例
    mrr: float                  # Mean Reciprocal Rank
    ndcg_at_k: float            # 归一化折损累积增益
    latency_p50_ms: float       # 中位延迟
    latency_p99_ms: float       # P99延迟

def evaluate_retrieval(
    queries: list[str],
    relevant_doc_ids: list[set[str]],
    vectorstore,
    k: int = 5,
) -> RetrievalMetrics:
    """评估检索质量"""
    all_precisions = []
    all_recalls = []
    all_rrs = []
    all_latencies = []

    for query, relevant_ids in zip(queries, relevant_doc_ids):
        import time
        start = time.time()
        results = vectorstore.similarity_search(query, k=k)
        latency_ms = (time.time() - start) * 1000
        all_latencies.append(latency_ms)

        retrieved_ids = {doc.metadata["doc_id"] for doc in results}
        relevant_retrieved = retrieved_ids & relevant_ids

        # Precision@K
        all_precisions.append(len(relevant_retrieved) / k if k > 0 else 0)
        # Recall@K
        all_recalls.append(len(relevant_retrieved) / len(relevant_ids) if relevant_ids else 0)
        # MRR
        for i, doc in enumerate(results):
            if doc.metadata["doc_id"] in relevant_ids:
                all_rrs.append(1.0 / (i + 1))
                break
        else:
            all_rrs.append(0.0)

    import numpy as np
    return RetrievalMetrics(
        precision_at_k=np.mean(all_precisions),
        recall_at_k=np.mean(all_recalls),
        mrr=np.mean(all_rrs),
        ndcg_at_k=0.0,  # 需要完整排序分数计算
        latency_p50_ms=np.percentile(all_latencies, 50),
        latency_p99_ms=np.percentile(all_latencies, 99),
    )
```

### 5.6.2 索引新鲜度管理

```python
from datetime import datetime, timedelta

class IndexManager:
    """索引新鲜度管理"""

    def __init__(self, vectorstore, full_rebuild_interval_days: int = 90):
        self.vectorstore = vectorstore
        self.full_rebuild_interval = timedelta(days=full_rebuild_interval_days)
        self.last_full_rebuild = None
        self.incremental_tracker = {}  # source -> last_sync_time

    def sync_incremental(self, source: str, new_docs: list[Document]):
        """增量更新：只处理新增和修改的文档"""
        last_sync = self.incremental_tracker.get(source)
        if last_sync:
            # 只保留上次同步后更新的文档
            new_docs = [
                doc for doc in new_docs
                if doc.metadata.get("updated_at", "") > last_sync
            ]

        if new_docs:
            # 删除旧版本
            doc_ids = [doc.metadata["doc_id"] for doc in new_docs]
            self.vectorstore.delete(ids=doc_ids)
            # 添加新版本
            self.vectorstore.add_documents(new_docs)

        self.incremental_tracker[source] = datetime.now().isoformat()

    def check_full_rebuild_needed(self) -> bool:
        """判断是否需要全量重建"""
        if self.last_full_rebuild is None:
            return True
        return datetime.now() - self.last_full_rebuild > self.full_rebuild_interval

    def full_rebuild(self, all_docs: list[Document]):
        """全量重建索引"""
        self.vectorstore.delete_collection()
        self.vectorstore.add_documents(all_docs)
        self.last_full_rebuild = datetime.now()
```

### 5.6.3 成本优化

```python
def estimate_monthly_cost(
    doc_count: int,
    avg_doc_tokens: int,
    queries_per_day: int,
    embedding_model: str = "text-embedding-3-small",
    llm_model: str = "gpt-4o-mini",
) -> dict:
    """月度成本估算"""
    costs = {}

    # 嵌入成本（初始索引 + 每日增量）
    if "openai" in embedding_model:
        embed_cost_per_1k = 0.00002  # text-embedding-3-small
        total_embed_tokens = doc_count * avg_doc_tokens
        costs["embedding_index"] = total_embed_tokens / 1000 * embed_cost_per_1k
        costs["embedding_query_monthly"] = (
            queries_per_day * 30 * 500 / 1000 * embed_cost_per_1k  # 假设每查询500 tokens
        )

    # LLM 成本
    if llm_model == "gpt-4o-mini":
        input_cost = 0.00015 / 1000
        output_cost = 0.0006 / 1000
        # 假设每查询：2000 input tokens, 500 output tokens
        monthly_queries = queries_per_day * 30
        costs["llm_monthly"] = monthly_queries * (2000 * input_cost + 500 * output_cost)

    # 向量库存储成本
    costs["vector_storage_monthly"] = doc_count * 1536 * 4 / (1024**3) * 0.25  # 粗略估算

    costs["total_monthly"] = sum(v for v in costs.values())
    return costs
```

```python
"""
企业知识库系统 - LangChain + Milvus
功能：多源文档摄取、权限控制检索、增量同步
"""
import os
from datetime import datetime
from typing import Optional

from langchain_core.documents import Document
from langchain_core.output_parsers import StrOutputParser
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.runnables import RunnablePassthrough
from langchain_milvus import Milvus
from langchain_openai import OpenAIEmbeddings, ChatOpenAI
from langchain_text_splitters import RecursiveCharacterTextSplitter

# ─── 初始化 ───
embeddings = OpenAIEmbeddings(model="text-embedding-3-small")
llm = ChatOpenAI(model="gpt-4o-mini", temperature=0.1)

vectorstore = Milvus(
    embedding_function=embeddings,
    connection_args={"uri": os.environ.get("MILVUS_URI", "http://localhost:19530")},
    collection_name="enterprise_kb",
    auto_create_collection=True,
    metadata_fields=[
        {"name": "doc_id", "type": "VARCHAR", "max_length": 128},
        {"name": "department", "type": "VARCHAR", "max_length": 64},
        {"name": "access_level", "type": "VARCHAR", "max_length": 32},
        {"name": "source_system", "type": "VARCHAR", "max_length": 64},
        {"name": "updated_at", "type": "VARCHAR", "max_length": 32},
    ],
)

# ─── 数据摄取管道 ───
class DataIngestionPipeline:
    """企业文档摄取管道"""

    def __init__(self, vectorstore, chunk_size: int = 600, chunk_overlap: int = 80):
        self.vectorstore = vectorstore
        self.splitter = RecursiveCharacterTextSplitter(
            chunk_size=chunk_size,
            chunk_overlap=chunk_overlap,
            separators=["\n\n## ", "\n\n### ", "\n\n", "\n", "。", ""],
        )

    def ingest_documents(
        self,
        documents: list[Document],
        source_system: str,
        department: str,
        access_level: str = "internal",
    ) -> int:
        """摄取一批文档到知识库"""
        processed_docs = []

        for doc in documents:
            # 富化元数据
            doc.metadata.update({
                "source_system": source_system,
                "department": department,
                "access_level": access_level,
                "ingested_at": datetime.now().isoformat(),
            })

            # 分块
            chunks = self.splitter.split_documents([doc])
            for i, chunk in enumerate(chunks):
                chunk.metadata["chunk_index"] = i
                chunk.metadata["total_chunks"] = len(chunks)
                processed_docs.append(chunk)

        # 写入向量库
        self.vectorstore.add_documents(processed_docs)
        return len(processed_docs)

# ─── 权限控制检索 ───
class AccessControlledRetriever:
    """带权限控制的检索器"""

    def __init__(self, vectorstore: Milvus):
        self.vectorstore = vectorstore

    def retrieve(
        self,
        query: str,
        user_department: str,
        access_levels: list[str],
        k: int = 5,
    ) -> list[Document]:
        """基于权限的检索"""
        level_map = {"public": 0, "internal": 1, "confidential": 2}
        max_level = max(level_map.get(l, 0) for l in access_levels)

        filter_expr = (
            f'access_level_int <= {max_level} || department == "{user_department}"'
        )

        results = self.vectorstore.similarity_search(
            query=query, k=k, expr=filter_expr,
        )
        return results

# ─── 问答链 ───
def build_qa_chain(retriever: AccessControlledRetriever):
    """构建RAG问答链"""

    template = """你是一个企业知识库助手。基于以下检索到的文档片段回答用户问题。
如果文档中没有相关信息，请明确说明"知识库中未找到相关信息"，不要编造内容。

检索到的文档片段：
{context}

用户问题：{question}

请给出准确、有依据的回答，并注明信息来源。"""

    prompt = ChatPromptTemplate.from_template(template)

    def format_docs(docs: list[Document]) -> str:
        return "\n\n---\n\n".join(
            f"[来源: {d.metadata.get('source', '未知')} | "
            f"部门: {d.metadata.get('department', '未知')}]\n{d.page_content}"
            for d in docs
        )

    chain = (
        {
            "context": lambda x: format_docs(
                retriever.retrieve(
                    x["question"],
                    user_department=x["user_department"],
                    access_levels=x["access_levels"],
                )
            ),
            "question": lambda x: x["question"],
        }
        | prompt
        | llm
        | StrOutputParser()
    )
    return chain

# ─── 使用示例 ───
if __name__ == "__main__":
    # 1. 摄取文档
    pipeline = DataIngestionPipeline(vectorstore)
    sample_docs = [
        Document(
            page_content="## 休假政策\n\n正式员工每年享有15天年假...",
            metadata={"doc_id": "HR-POL-001", "title": "休假政策"},
        ),
    ]
    pipeline.ingest_documents(sample_docs, source_system="confluence", department="hr")

    # 2. 问答
    retriever = AccessControlledRetriever(vectorstore)
    qa_chain = build_qa_chain(retriever)

    answer = qa_chain.invoke({
        "question": "员工每年有多少天年假？",
        "user_department": "engineering",
        "access_levels": ["internal"],
    })
    print(answer)
```

## 5.8 小结

企业知识库开发是一个系统工程，需要在数据管道、分块策略、权限控制、部署架构和运维监控多个维度进行精心设计。本章要点：

- **需求先行**：明确定义文档类型、查询模式和用户画像，避免后期返工
- **元数据是核心**：权限控制、精准检索、新鲜度管理都依赖丰富的元数据
- **分块决定上限**：父子分块策略在企业场景中尤为重要，兼顾检索精度和上下文完整性
- **安全不可妥协**：权限过滤必须在检索层执行，而非仅在展示层
- **架构匹配规模**：从单节点到分布式，选择与业务规模匹配的架构
- **持续运维**：检索质量监控和索引新鲜度管理是长期保障
