## 46.2 共享库

### 46.2.1 事件系统

```python
# shared/events.py
import json
import redis.asyncio as aioredis
from dataclasses import dataclass, asdict
from typing import Any
from datetime import datetime

@dataclass
class Event:
    """基础事件类。"""
    event_type: str
    aggregate_id: str
    data: dict
    timestamp: str = None

    def __post_init__(self):
        if not self.timestamp:
            self.timestamp = datetime.utcnow().isoformat()

    def to_json(self) -> str:
        return json.dumps(asdict(self))

    @classmethod
    def from_json(cls, data: str) -> "Event":
        return cls(**json.loads(data))

# 事件类型
class EventType:
    USER_CREATED = "user.created"
    ORDER_CREATED = "order.created"
    ORDER_PAID = "order.paid"
    ORDER_CANCELLED = "order.cancelled"
    PAYMENT_COMPLETED = "payment.completed"
    PAYMENT_FAILED = "payment.failed"
    STOCK_DEDUCTED = "stock.deducted"
    STOCK_INSUFFICIENT = "stock.insufficient"

class EventBus:
    """基于 Redis Pub/Sub 的事件总线。"""
    def __init__(self, redis_url: str):
        self.redis = aioredis.from_url(redis_url)
        self.pubsub = self.redis.pubsub()

    async def publish(self, event: Event):
        await self.redis.publish(event.event_type, event.to_json())

    async def subscribe(self, event_types: list[str]):
        for event_type in event_types:
            await self.pubsub.subscribe(event_type)

    async def listen(self):
        async for message in self.pubsub.listen():
            if message["type"] == "message":
                yield Event.from_json(message["data"])
```

```python
# shared/auth.py
from fastapi import Depends, HTTPException, status
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials
import jwt
from jwt import PyJWTError

security = HTTPBearer()

async def verify_token(credentials: HTTPAuthorizationCredentials = Depends(security)):
    """验证 JWT Token。"""
    try:
        payload = jwt.decode(
            credentials.credentials,
            SECRET_KEY,
            algorithms=["HS256"],
        )
        user_id = payload.get("sub")
        if user_id is None:
            raise HTTPException(status_code=401, detail="Invalid token")
        return {"user_id": int(user_id), "role": payload.get("role", "user")}
    except PyJWTError:
        raise HTTPException(status_code=401, detail="Invalid token")

async def require_admin(user=Depends(verify_token)):
    if user["role"] != "admin":
        raise HTTPException(status_code=403, detail="Admin access required")
    return user
```

```python
# order-service/app/models/order.py
from sqlalchemy import Column, Integer, String, Float, DateTime, ForeignKey, Enum
from sqlalchemy.sql import func
from app.database import Base
import enum

class OrderStatus(str, enum.Enum):
    pending = "pending"
    confirmed = "confirmed"
    paid = "paid"
    shipped = "shipped"
    delivered = "delivered"
    cancelled = "cancelled"

class Order(Base):
    __tablename__ = "orders"

    id = Column(Integer, primary_key=True, index=True)
    user_id = Column(Integer, nullable=False, index=True)
    total_amount = Column(Float, nullable=False)
    status = Column(Enum(OrderStatus), default=OrderStatus.pending, nullable=False)
    shipping_address = Column(String(500), nullable=False)
    created_at = Column(DateTime(timezone=True), server_default=func.now())
    updated_at = Column(DateTime(timezone=True), server_default=func.now(), onupdate=func.now())

    items = relationship("OrderItem", back_populates="order", lazy="selectin")

class OrderItem(Base):
    __tablename__ = "order_items"

    id = Column(Integer, primary_key=True)
    order_id = Column(Integer, ForeignKey("orders.id"), nullable=False)
    product_id = Column(Integer, nullable=False)
    quantity = Column(Integer, nullable=False)
    unit_price = Column(Float, nullable=False)
    subtotal = Column(Float, nullable=False)

    order = relationship("Order", back_populates="items")
```

```python
# order-service/app/routers/orders.py
from fastapi import APIRouter, Depends, HTTPException, status
from app.schemas.order import OrderCreate, OrderResponse, OrderListResponse
from app.services.order import create_order, get_order, list_orders, cancel_order
from shared.auth import verify_token

router = APIRouter(prefix="/orders", tags=["orders"])

@router.post("/", response_model=OrderResponse, status_code=status.HTTP_201_CREATED)
async def create_new_order(
    order_in: OrderCreate,
    user=Depends(verify_token),
    db: AsyncSession = Depends(get_db),
):
    """创建订单：
    1. 验证商品库存（调用商品服务）
    2. 扣减库存（调用商品服务）
    3. 创建订单
    4. 发布 OrderCreated 事件
    """
    order = await create_order(db, order_in, user["user_id"])
    return order

@router.get("/{order_id}", response_model=OrderResponse)
async def get_order_by_id(
    order_id: int,
    user=Depends(verify_token),
    db: AsyncSession = Depends(get_db),
):
    order = await get_order(db, order_id)
    if not order:
        raise HTTPException(status_code=404, detail="Order not found")
    if order.user_id != user["user_id"] and user["role"] != "admin":
        raise HTTPException(status_code=403)
    return order

@router.get("/", response_model=OrderListResponse)
async def list_user_orders(
    page: int = Query(1, ge=1),
    page_size: int = Query(20, ge=1, le=100),
    user=Depends(verify_token),
    db: AsyncSession = Depends(get_db),
):
    return await list_orders(db, user["user_id"], page, page_size)

@router.post("/{order_id}/cancel", response_model=OrderResponse)
async def cancel_existing_order(
    order_id: int,
    user=Depends(verify_token),
    db: AsyncSession = Depends(get_db),
):
    order = await get_order(db, order_id)
    if not order:
        raise HTTPException(status_code=404)
    if order.user_id != user["user_id"]:
        raise HTTPException(status_code=403)
    if order.status not in (OrderStatus.pending, OrderStatus.confirmed):
        raise HTTPException(status_code=400, detail="Cannot cancel order in current status")
    return await cancel_order(db, order)
```

### 46.3.3 订单服务（业务逻辑）

```python
# order-service/app/services/order.py
from shared.events import EventBus, Event, EventType

async def create_order(db: AsyncSession, order_in: OrderCreate, user_id: int) -> Order:
    """创建订单（编排多个服务）。"""
    # 1. 验证并扣减库存（同步调用商品服务）
    async with httpx.AsyncClient() as client:
        stock_response = await client.post(
            f"{PRODUCT_SERVICE_URL}/products/deduct-stock",
            json=[{"product_id": item.product_id, "quantity": item.quantity} for item in order_in.items],
            headers={"Authorization": f"Bearer {get_service_token()}"},
        )
        if stock_response.status_code != 200:
            raise HTTPException(status_code=400, detail="Insufficient stock")

    # 2. 计算总金额
    total = sum(item.quantity * item.unit_price for item in order_in.items)

    # 3. 创建订单
    order = Order(user_id=user_id, total_amount=total, status=OrderStatus.pending)
    db.add(order)
    await db.flush()

    for item in order_in.items:
        order_item = OrderItem(
            order_id=order.id,
            product_id=item.product_id,
            quantity=item.quantity,
            unit_price=item.unit_price,
            subtotal=item.quantity * item.unit_price,
        )
        db.add(order_item)

    await db.commit()
    await db.refresh(order)

    # 4. 发布 OrderCreated 事件
    event_bus = EventBus(settings.REDIS_URL)
    await event_bus.publish(Event(
        event_type=EventType.ORDER_CREATED,
        aggregate_id=str(order.id),
        data={"order_id": order.id, "user_id": user_id, "total": total},
    ))

    return order
```

## 46.4 API 网关

### 46.4.1 网关配置

```python
# api-gateway/app/main.py
from fastapi import FastAPI, Request, HTTPException
from fastapi.middleware.cors import CORSMiddleware
import httpx

app = FastAPI(title="API Gateway", version="1.0.0")

app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_methods=["*"],
    allow_headers=["*"],
)

# 服务注册表
SERVICE_URLS = {
    "users": "http://user-service:8001",
    "products": "http://product-service:8002",
    "orders": "http://order-service:8003",
    "payments": "http://payment-service:8004",
}

# 路由规则
ROUTES = {
    "/api/users": "users",
    "/api/products": "products",
    "/api/orders": "orders",
    "/api/payments": "payments",
}

@app.api_route("/api/{path:path}", methods=["GET", "POST", "PUT", "PATCH", "DELETE"])
async def gateway(request: Request, path: str):
    """API 网关 — 路由转发。"""
    # 1. 确定目标服务
    service_name = None
    for prefix, service in ROUTES.items():
        if f"/api/{path}".startswith(prefix):
            service_name = service
            break

    if not service_name:
        raise HTTPException(status_code=404, detail="Service not found")

    # 2. 转发请求
    service_url = SERVICE_URLS[service_name]
    target_url = f"{service_url}/api/{path}"

    async with httpx.AsyncClient(timeout=30.0) as client:
        response = await client.request(
            method=request.method,
            url=target_url,
            headers=dict(request.headers),
            content=await request.body(),
            params=dict(request.query_params),
        )

    # 3. 返回响应
    return Response(
        content=response.content,
        status_code=response.status_code,
        headers=dict(response.headers),
    )
```

```python
async def handle_payment_events():
    event_bus = EventBus(settings.REDIS_URL)
    await event_bus.subscribe([EventType.PAYMENT_COMPLETED, EventType.PAYMENT_FAILED])

    async for event in event_bus.listen():
        if event.event_type == EventType.PAYMENT_COMPLETED:
            # 支付成功 → 更新订单状态
            order_id = int(event.aggregate_id)
            order = await get_order(db, order_id)

            await db.commit()

                event_type=EventType.ORDER_PAID,
                aggregate_id=str(order_id),
                data={"order_id": order_id},
```

### 46.5.2 链路追踪

```python
# shared/tracing.py
import uuid
from fastapi import Request

async def tracing_middleware(request: Request, call_next):
    """分布式链路追踪中间件。"""
    # 从上游服务获取或生成 trace_id
    trace_id = request.headers.get("X-Trace-ID") or str(uuid.uuid4())
    request.state.trace_id = trace_id

    # 传递给下游服务
    response = await call_next(request)
    response.headers["X-Trace-ID"] = trace_id
    return response

# 服务间调用时传递 trace_id
async def call_service(method: str, url: str, trace_id: str, **kwargs):
    async with httpx.AsyncClient() as client:
        headers = kwargs.pop("headers", {})
        headers["X-Trace-ID"] = trace_id
        return await client.request(method, url, headers=headers, **kwargs)
```

## 46.7 常见陷阱与最佳实践

| 陷阱 | 最佳实践 |
|------|---------|
| 服务间同步调用链过长 | 使用事件驱动减少同步调用 |
| 不处理分布式事务 | 使用 Saga 模式管理跨服务事务 |
| 服务间硬编码 URL | 使用服务发现或环境变量配置 |
| 不做链路追踪 | 每个请求传递 X-Trace-ID |
| 事件丢失 | 使用 Redis Streams 或消息队列持久化 |
| 数据不一致 | 实现补偿事务和幂等性 |
| 不做服务健康检查 | 每个服务提供 /health 端点 |
| 单一数据库共享 | 每个服务独立数据库 |
