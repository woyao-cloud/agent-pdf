# 第03章 请求处理：查询参数、路径参数与请求体

请求处理是 FastAPI 的核心功能。本章从参数解析的源码流程到 Pydantic 验证的调用时机，深入剖析 FastAPI 如何将 HTTP 请求转化为类型安全的 Python 对象。

---

## 3.1 概念与语法

### 3.1.1 路径参数

```python
from fastapi import FastAPI, Path
from typing import Annotated

app = FastAPI()

# 基本路径参数
@app.get("/items/{item_id}")
async def read_item(item_id: int):
    return {"item_id": item_id}

# 带验证的路径参数
@app.get("/items/{item_id}")
async def read_item(
    item_id: Annotated[int, Path(
        title="Item ID",
        description="The ID of the item to retrieve",
        ge=1,
        le=1000,
    )],
):
    return {"item_id": item_id}

# 路径参数验证约束：
# gt  — 大于
# ge  — 大于等于
# lt  — 小于
# le  — 小于等于

# 字符串路径参数
@app.get("/files/{file_path:path}")
async def read_file(file_path: str):
    return {"file_path": file_path}
# /files/home/user/documents/report.txt → file_path = "home/user/documents/report.txt"

# 枚举路径参数
from enum import Enum

class ModelName(str, Enum):
    alexnet = "alexnet"
    resnet = "resnet"
    lenet = "lenet"

@app.get("/models/{model_name}")
async def get_model(model_name: ModelName):
    return {"model_name": model_name, "message": "Deep Learning"}
```

### 3.1.2 查询参数

```python
from fastapi import FastAPI, Query
from typing import Annotated, Optional

app = FastAPI()

# 基本查询参数
@app.get("/items/")
async def read_items(skip: int = 0, limit: int = 10):
    return {"skip": skip, "limit": limit}
# GET /items/?skip=0&limit=10

# 可选查询参数
@app.get("/items/{item_id}")
async def read_item(item_id: str, q: Optional[str] = None):
    if q:
        return {"item_id": item_id, "q": q}
    return {"item_id": item_id}

# 带验证的查询参数
@app.get("/items/")
async def read_items(
    q: Annotated[Optional[str], Query(
        title="Query string",
        description="Search items by name",
        min_length=3,
        max_length=50,
        pattern="^[a-zA-Z0-9]+$",
    )] = None,
    skip: Annotated[int, Query(ge=0)] = 0,
    limit: Annotated[int, Query(ge=1, le=100)] = 10,
):
    return {"q": q, "skip": skip, "limit": limit}

# 查询参数列表（多个值）
@app.get("/items/")
async def read_items(q: Annotated[Optional[list[str]], Query()] = None):
    return {"q": q}
# GET /items/?q=foo&q=bar → q = ["foo", "bar"]

# 别名查询参数
@app.get("/items/")
async def read_items(
    item_query: Annotated[Optional[str], Query(alias="item-query")] = None,
):
    return {"item_query": item_query}
# GET /items/?item-query=foo
```

### 3.1.3 请求体

```python
from fastapi import FastAPI
from pydantic import BaseModel, Field
from typing import Optional
from datetime import datetime

app = FastAPI()

# 基本请求体模型
class ItemCreate(BaseModel):
    name: str
    description: Optional[str] = None
    price: float = Field(..., gt=0, description="Price must be positive")
    tax: Optional[float] = None

@app.post("/items/")
async def create_item(item: ItemCreate):
    return item

# 嵌套模型
class Address(BaseModel):
    street: str
    city: str
    zip_code: str

class UserCreate(BaseModel):
    username: str
    email: str
    address: Address  # 嵌套模型

@app.post("/users/")
async def create_user(user: UserCreate):
    return user

# 请求体 + 路径参数 + 查询参数
@app.put("/items/{item_id}")
async def update_item(
    item_id: int,
    q: Optional[str] = None,
    item: ItemCreate = None,
):
    result = {"item_id": item_id, **item.model_dump()}
    if q:
        result.update({"q": q})
    return result

# 多个请求体（Body 嵌入）
from fastapi import Body

class Item(BaseModel):
    name: str
    description: Optional[str] = None

class User(BaseModel):
    username: str

@app.put("/items/{item_id}")
async def update_item(
    item_id: int,
    item: Item,
    user: User,
    importance: Annotated[int, Body(gt=0)],
):
    return {"item_id": item_id, "item": item, "user": user, "importance": importance}
# 请求体格式：{"item": {...}, "user": {...}, "importance": 5}

# 单值请求体嵌入
@app.put("/items/{item_id}")
async def update_item(
    item_id: int,
    importance: Annotated[int, Body(embed=True)],
):
    return {"item_id": item_id, "importance": importance}
# 请求体格式：{"importance": 5}（而非裸值 5）
```

### 3.1.4 Field 验证

```python
from pydantic import BaseModel, Field
from typing import Annotated
from datetime import datetime

class ItemCreate(BaseModel):
    name: Annotated[str, Field(
        min_length=1,
        max_length=100,
        title="Item Name",
        description="The name of the item",
        examples=["Widget"],
    )]
    description: Annotated[Optional[str], Field(
        max_length=1000,
        default=None,
    )] = None
    price: Annotated[float, Field(
        gt=0,
        description="Price must be positive",
    )]
    tax: Annotated[Optional[float], Field(
        ge=0,
        le=1.0,
        description="Tax rate (0 to 1)",
    )] = None
    tags: Annotated[list[str], Field(
        default_factory=list,
        max_length=10,
        description="Item tags",
    )]

# Field 验证约束：
# 字符串：min_length, max_length, pattern
# 数字：gt, ge, lt, le, multiple_of
# 列表：min_length, max_length
# 通用：default, default_factory, title, description, examples, alias
```

## 3.2 原理与机制

### 3.2.1 参数解析的源码流程

```python
"""
FastAPI 参数解析的完整流程：

1. 请求到达 → ASGI 调用 app(scope, receive, send)
2. Starlette 路由匹配 → 找到对应的 APIRoute
3. 调用 get_request_handler() 创建的处理函数
4. 处理函数调用 solve_dependencies() 解析依赖
5. solve_dependencies() 内部：
   a. 从函数签名提取参数（inspect.signature）
   b. 分析每个参数的注解和元数据
   c. 根据参数类型确定数据来源：
      - Path → scope['path_params']
      - Query → scope['query_string']
      - Header → scope['headers']
      - Cookie → scope['headers'] 中的 Cookie 头
      - Body → 读取 receive() 中的请求体
   d. 对路径/查询参数执行类型转换
   e. 对请求体执行 Pydantic 验证
   f. 收集所有依赖的返回值
6. 调用路由函数，传入解析后的参数

关键源码位置：
- fastapi/dependencies/utils.py: solve_dependencies()
- fastapi/routing.py: get_request_handler()
- fastapi/params.py: Path(), Query(), Body() 参数定义
"""
```

### 3.2.2 请求体验证流程

```python
"""
请求体验证的详细流程：

1. 从 ASGI receive 读取请求体（bytes）
2. 根据 content-type 解析：
   - application/json → json.loads()
   - application/x-www-form-urlencoded → 解析表单数据
   - multipart/form-data → 解析多部分数据
3. 使用 Pydantic 模型验证：
   a. 调用 model.__init__(**data) 或 model.model_validate(data)
   b. Pydantic 执行类型转换和验证
   c. 验证失败 → 生成 422 Unprocessable Entity 响应
4. 验证成功 → 将模型实例注入路由函数

422 错误响应的格式：
{
    "detail": [
        {
            "loc": ["body", "price"],
            "msg": "ensure this value is greater than 0",
            "type": "value_error.number.gt",
            "ctx": {"limit_value": 0}
        }
    ]
}

loc 字段的含义：
- ["body"] → 整个请求体
- ["body", "field_name"] → 请求体中的字段
- ["body", "nested", "field_name"] → 嵌套字段
- ["query", "param_name"] → 查询参数
- ["path", "param_name"] → 路径参数
"""
```

### 3.2.3 参数默认值注入原理

```python
"""
FastAPI 的参数解析依赖 Python 的类型注解和默认值：

1. 必需参数（无默认值）：
   async def get_item(item_id: int)  → 必须提供

2. 可选参数（默认值为 None）：
   async def get_item(item_id: int, q: str | None = None)  → 可选

3. 查询参数（有默认值）：
   async def get_items(skip: int = 0, limit: int = 10)  → 查询参数

4. 请求体（Pydantic 模型类型）：
   async def create_item(item: ItemCreate)  → 请求体

FastAPI 通过 inspect.signature() 分析函数签名：
- 从类型注解确定参数的数据来源
- 从默认值确定参数是否必需
- 从 Annotated[] 元数据提取验证约束

参数优先级规则：
1. 路径参数：与路径模板中的 {param} 匹配
2. 单个请求体：如果只有一个 Pydantic 模型参数，作为请求体
3. 多个请求体：使用 Body(embed=True) 或字段名作为键
4. 其他：根据类型和默认值推断（查询参数）
"""
```

## 3.3 使用场景

### 3.3.1 完整 CRUD API

```python
"""完整的用户 CRUD API，展示所有参数类型"""
from fastapi import FastAPI, HTTPException, Path, Query, Body, status
from pydantic import BaseModel, EmailStr, Field
from typing import Annotated, Optional
from datetime import datetime

app = FastAPI(title="User CRUD API")

# 模型定义
class UserCreate(BaseModel):
    username: Annotated[str, Field(min_length=3, max_length=50)]
    email: EmailStr
    full_name: Annotated[Optional[str], Field(max_length=100)] = None
    password: Annotated[str, Field(min_length=8)]

class UserUpdate(BaseModel):
    email: Optional[EmailStr] = None
    full_name: Optional[str] = None

class UserResponse(BaseModel):
    id: int
    username: str
    email: str
    full_name: Optional[str] = None
    is_active: bool
    created_at: datetime

# 模拟数据库
db: dict[int, dict] = {}
next_id = 1

@app.post(
    "/users/",
    response_model=UserResponse,
    status_code=status.HTTP_201_CREATED,
    summary="创建用户",
    description="创建一个新用户账号",
)
async def create_user(user_in: UserCreate):
    global next_id
    user = {
        "id": next_id,
        **user_in.model_dump(),
        "is_active": True,
        "created_at": datetime.now(),
    }
    db[next_id] = user
    next_id += 1
    return user

@app.get(
    "/users/",
    response_model=list[UserResponse],
    summary="获取用户列表",
)
async def list_users(
    skip: Annotated[int, Query(ge=0, description="跳过记录数")] = 0,
    limit: Annotated[int, Query(ge=1, le=100, description="返回记录数")] = 10,
    is_active: Annotated[Optional[bool], Query(description="按活跃状态过滤")] = None,
):
    users = list(db.values())
    if is_active is not None:
        users = [u for u in users if u["is_active"] == is_active]
    return users[skip:skip + limit]

@app.get(
    "/users/{user_id}",
    response_model=UserResponse,
    summary="获取用户详情",
)
async def get_user(
    user_id: Annotated[int, Path(ge=1, description="用户ID")],
):
    if user_id not in db:
        raise HTTPException(status_code=404, detail="User not found")
    return db[user_id]

@app.patch(
    "/users/{user_id}",
    response_model=UserResponse,
    summary="更新用户信息",
)
async def update_user(
    user_id: Annotated[int, Path(ge=1)],
    user_update: UserUpdate,
):
    if user_id not in db:
        raise HTTPException(status_code=404, detail="User not found")
    update_data = user_update.model_dump(exclude_unset=True)
    db[user_id].update(update_data)
    return db[user_id]

@app.delete(
    "/users/{user_id}",
    status_code=status.HTTP_204_NO_CONTENT,
    summary="删除用户",
)
async def delete_user(
    user_id: Annotated[int, Path(ge=1)],
):
    if user_id not in db:
        raise HTTPException(status_code=404, detail="User not found")
    del db[user_id]
```

## 3.5 常见陷阱与最佳实践

### 3.5.1 可选参数 vs 默认值

```python
# 陷阱：混淆 Optional 和默认值
@app.get("/items/")
async def read_items(
    q: Optional[str] = None,     # 可选查询参数，默认为 None
    skip: int = 0,                 # 必需查询参数，有默认值
    limit: int = 10,               # 必需查询参数，有默认值
):
    ...

# Optional[str] = None → 参数可以不传，值为 None
# int = 0 → 参数可以不传，值为 0
# int（无默认值）→ 参数必须传递，否则 422 错误

# 最佳实践：
# 1. 必需参数不设默认值
# 2. 可选参数用 Optional[type] = None
# 3. 有默认值的参数放在后面
```

### 3.5.2 请求体中 Pydantic 模型的 mutable 问题

```python
from pydantic import BaseModel

# 陷阱：mutable 默认值
class Item(BaseModel):
    tags: list[str] = []  # 每次创建新实例都共享同一个列表？

# Pydantic V2 处理了这个问题，每个实例有独立的列表
# 但在 FastAPI 中，请求体模型不应该有可变默认值

# 最佳实践：使用 default_factory
from pydantic import Field

class Item(BaseModel):
    tags: list[str] = Field(default_factory=list)
```

### 3.5.3 422 错误的定制

```python
from fastapi import FastAPI, Request
from fastapi.exceptions import RequestValidationError
from fastapi.responses import JSONResponse

app = FastAPI()

@app.exception_handler(RequestValidationError)
async def validation_exception_handler(request: Request, exc: RequestValidationError):
    """自定义 422 验证错误格式。"""
    errors = []
    for error in exc.errors():
        errors.append({
            "field": ".".join(str(loc) for loc in error["loc"]),
            "message": error["msg"],
            "type": error["type"],
        })
    return JSONResponse(
        status_code=422,
        content={"detail": errors, "body": exc.body},
    )
```

### 3.5.4 总结

| 陷阱 | 最佳实践 |
|------|---------|
| 可选参数混淆 | Optional[type] = None 表示可选 |
| mutable 默认值 | 使用 Field(default_factory=list) |
| 422 错误信息不友好 | 自定义 exception_handler |
| 路径参数类型转换失败 | 使用 Path() 添加 ge/gt/lt/le 约束 |
| 请求体字段名与查询参数冲突 | 使用 Body(embed=True) 或别名 |
| 大请求体验证慢 | 限制请求体大小，使用 response_model_exclude |
