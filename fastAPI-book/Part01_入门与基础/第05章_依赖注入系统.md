# 第05章 依赖注入系统

依赖注入（Dependency Injection, DI）是 FastAPI 最强大和独特的特性。本章从 Depends() 的使用到 solve_dependencies 的源码分析，深入剖析 FastAPI 依赖注入的底层机制。

---

## 5.1 概念与语法

### 5.1.1 Depends() 基础

```python
from fastapi import FastAPI, Depends

app = FastAPI()

# 函数作为依赖
async def common_parameters(q: str | None = None, skip: int = 0, limit: int = 100):
    return {"q": q, "skip": skip, "limit": limit}

@app.get("/items/")
async def read_items(commons: dict = Depends(common_parameters)):
    return commons

@app.get("/users/")
async def read_users(commons: dict = Depends(common_parameters)):
    return commons

# 每个请求都会调用 common_parameters()
# 返回值注入到路由函数的 commons 参数
```

### 5.1.2 类作为依赖

```python
from fastapi import FastAPI, Depends

class CommonQueryParams:
    def __init__(self, q: str | None = None, skip: int = 0, limit: int = 100):
        self.q = q
        self.skip = skip
        self.limit = limit

# 方式1：显式使用 Depends
@app.get("/items/")
async def read_items(commons: CommonQueryParams = Depends(CommonQueryParams)):
    return {"q": commons.q, "skip": commons.skip, "limit": commons.limit}

# 方式2：简写（类型即依赖）
@app.get("/items/")
async def read_items(commons: CommonQueryParams = Depends()):
    return {"q": commons.q}
```

### 5.1.3 嵌套依赖

```python
from fastapi import FastAPI, Depends, Header, HTTPException

app = FastAPI()

# 依赖可以依赖其他依赖
async def verify_token(x_token: str = Header()):
    if x_token != "fake-super-secret-token":
        raise HTTPException(status_code=400, detail="X-Token header invalid")
    return x_token

async def verify_key(x_key: str = Header()):
    if x_key != "fake-super-secret-key":
        raise HTTPException(status_code=400, detail="X-Key header invalid")
    return x_key

# 嵌套依赖：verify_token_and_key 依赖 verify_token 和 verify_key
async def verify_token_and_key(
    token: str = Depends(verify_token),
    key: str = Depends(verify_key),
):
    return {"token": token, "key": key}

@app.get("/protected/")
async def protected(verified: dict = Depends(verify_token_and_key)):
    return {"verified": verified}
```

### 5.1.4 yield 依赖（资源清理）

```python
from fastapi import FastAPI, Depends

app = FastAPI()

# yield 依赖：在请求结束后执行清理
async def get_db():
    db = Database()
    try:
        yield db  # 请求处理期间提供 db
    finally:
        db.close()  # 请求结束后清理

@app.get("/users/")
async def read_users(db: Database = Depends(get_db)):
    return db.query("SELECT * FROM users")

# yield 依赖的异常处理
async def get_db_safe():
    db = Database()
    try:
        yield db
    except Exception:
        await db.rollback()  # 异常时回滚
    finally:
        await db.close()

# 多个 yield 依赖的清理顺序：与创建顺序相反（类似上下文管理器栈）
```

### 5.1.5 全局依赖与路由组依赖

```python
from fastapi import FastAPI, Depends, APIRouter

async def verify_api_key(api_key: str = Header()):
    if api_key != "secret":
        raise HTTPException(status_code=403)

# 全局依赖：应用于所有路由
app = FastAPI(dependencies=[Depends(verify_api_key)])

# 路由组依赖：应用于该路由器的所有路由
router = APIRouter(
    prefix="/admin",
    dependencies=[Depends(verify_api_key)],
)

@router.get("/users/")  # 自动应用 verify_api_key
async def admin_users():
    return {"users": []}

# 单个路由依赖
@app.get("/items/", dependencies=[Depends(verify_api_key)])
async def read_items():
    return {"items": []}
```

## 5.2 原理与机制

### 5.2.1 solve_dependencies 源码分析

```python
"""
solve_dependencies 是 FastAPI 依赖注入的核心函数
源码位置：fastapi/dependencies/utils.py

简化流程：

async def solve_dependencies(
    request: Request,
    dependant: Dependant,
    dependency_overrides_cache: dict | None = None,
):
    # 1. 初始化结果字典
    values: dict[str, Any] = {}
    errors: list[ValidationError] = []
    
    # 2. 递归解析依赖图
    for dep in dependant.dependencies:
        # 2a. 检查缓存（同一个请求内，use_cache=True 时复用）
        call = dep.call
        if use_cache and call in cache:
            values[dep.name] = cache[call]
            continue
        
        # 2b. 递归解析子依赖
        sub_values, sub_errors = await solve_dependencies(
            request, dep, cache
        )
        if sub_errors:
            errors.extend(sub_errors)
            continue
        
        # 2c. 执行依赖函数
        try:
            if inspect.isasyncgenfunction(call):
                # yield 依赖
                stack = AsyncExitStack()
                gen = call(**sub_values)
                value = await stack.enter_async_context(
                    contextlib.asynccontextmanager(gen)
                )
            elif inspect.isgeneratorfunction(call):
                # 同步 yield 依赖
                stack = ExitStack()
                gen = call(**sub_values)
                value = stack.enter_context(
                    contextlib.contextmanager(gen)
                )
            else:
                # 普通依赖
                value = await call(**sub_values) if \
                    inspect.iscoroutinefunction(call) else \
                    call(**sub_values)
            
            # 2d. 缓存结果
            if use_cache:
                cache[call] = value
            values[dep.name] = value
            
        except Exception as e:
            errors.append(e)
    
    return values, errors
"""
```

### 5.2.2 依赖图构建

```python
"""
FastAPI 在路由注册时构建依赖图：

1. get_dependant() 解析函数签名：
   - 识别 Depends() 参数 → 创建子 Dependant
   - 识别路径/查询/头部参数 → 创建 Param
   - 识别请求体 → 创建 BodyField
   - 构建依赖树

2. 依赖图的结构：
   Dependant (路由函数)
   ├── dependencies: [Dependant, ...]  # 子依赖列表
   ├── path_params: [Param, ...]         # 路径参数
   ├── query_params: [Param, ...]        # 查询参数
   ├── header_params: [Param, ...]      # 头部参数
   ├── cookie_params: [Param, ...]      # Cookie 参数
   └── body: BodyField                   # 请求体

3. 依赖解析顺序：
   - 先解析子依赖（深度优先）
   - 再解析当前依赖的参数
   - 最后执行当前依赖函数

4. 缓存机制：
   - 同一请求中，use_cache=True 的依赖只执行一次
   - 不同请求之间，缓存不共享
"""
```

## 5.3 使用场景

### 5.3.1 数据库会话管理

```python
from fastapi import FastAPI, Depends
from sqlalchemy.ext.asyncio import AsyncSession, create_async_engine
from sqlalchemy.orm import sessionmaker

engine = create_async_engine("sqlite+aiosqlite:///./test.db")
AsyncSessionLocal = sessionmaker(engine, class_=AsyncSession)

async def get_db():
    async with AsyncSessionLocal() as session:
        try:
            yield session
            await session.commit()
        except Exception:
            await session.rollback()
            raise

app = FastAPI()

@app.get("/users/")
async def read_users(db: AsyncSession = Depends(get_db)):
    result = await db.execute("SELECT * FROM users")
    return result.fetchall()
```

## 5.5 常见陷阱与最佳实践

### 5.5.1 yield 依赖的异常处理

```python
# 陷阱：yield 依赖的 finally 不捕获路由函数的异常
async def get_db():
    db = Database()
    try:
        yield db  # 如果路由抛异常，finally 仍执行
    finally:
        await db.close()  # 总是执行

# 如果需要在异常时回滚：
async def get_db():
    db = Database()
    try:
        yield db
    except Exception:
        await db.rollback()  # 路由异常时回滚
        raise  # 重新抛出异常
    finally:
        await db.close()
```

### 5.5.2 依赖缓存陷阱

```python
# 陷阱：use_cache=True（默认）导致同一请求内依赖结果被复用
async def get_query(q: str | None = None):
    return q  # 这个依赖在同一个请求中只执行一次

@app.get("/items/")
async def read_items(
    query1: str = Depends(get_query),  # 第一次调用
    query2: str = Depends(get_query),  # 复用第一次的结果！
):
    return {"query1": query1, "query2": query2}

# 如果需要每次都执行，使用 use_cache=False
@app.get("/items/")
async def read_items(
    query: str = Depends(get_query, use_cache=False),
):
    return {"query": query}
```

### 5.5.3 总结

| 陷阱 | 最佳实践 |
|------|---------|
| 忘记 yield 的 finally | 使用 try/yield/finally 确保资源清理 |
| 依赖缓存导致意外 | 理解 use_cache 语义，必要时设为 False |
| 同步依赖阻塞事件循环 | 确保异步路由中使用 async def 依赖 |
| 依赖循环引用 | 重构为非循环依赖或延迟导入 |
| 依赖中抛异常不被捕获 | 在 yield 后添加 except 块处理异常 |
