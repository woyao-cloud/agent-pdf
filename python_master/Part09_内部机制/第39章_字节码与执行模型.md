# 第39章 字节码与执行模型

## 39.1 概念与语法

### 39.1.1 什么是字节码

Python 源代码在执行前会被编译为一种中间表示——**字节码(bytecode)**。字节码是 CPython 虚拟机的指令集，每条指令由一个操作码(opcode)和可选的操作数(oparg)组成。这一设计使得 Python 可以在运行时动态编译并执行代码，而无需生成平台相关的机器码。

```python
def greet(name):
    return f"Hello, {name}!"

# 编译为字节码
import dis
dis.dis(greet)
```

输出类似：
```
  2           0 LOAD_CONST               1 ('Hello, ')
              2 LOAD_FAST                0 (name)
              4 FORMAT_SIMPLE
              6 LOAD_CONST               2 ('!')
              8 BUILD_STRING             3
             10 RETURN_VALUE
```

### 39.1.2 dis 模块概览

`dis` 模块是 Python 标准库中用于反汇编字节码的核心工具，它将二进制形式的字节码还原为人类可读的助记符。

```python
import dis

# dis.dis — 反汇编函数、方法或代码对象
dis.dis(greet)

# dis.disco — 反汇编指定偏移量范围的代码对象
dis.disco(greet.__code__.co_code, 0, 10)

# dis.Bytecode — 面向对象的字节码分析接口
bytecode = dis.Bytecode(greet)
for instr in bytecode:
    print(f"{instr.offset:4d} {instr.opname:<20} {instr.argrepr}")

# dis.Instruction 对象属性
instr = list(dis.Bytecode(greet))[0]
print(instr.opname)     # 'LOAD_CONST'
print(instr.opcode)     # 100 (数值操作码)
print(instr.arg)        # 操作数
print(instr.argval)     # 操作数解析后的值
print(instr.argrepr)    # 操作数的字符串表示
print(instr.offset)     # 字节偏移量
print(instr.starts_line) # 对应源码行号(可能为None)
print(instr.is_jump_target) # 是否为跳转目标
```

### 39.1.4 栈式虚拟机模型

CPython 采用**栈式虚拟机(stack-based VM)**架构，与寄存式虚拟机不同，操作数通过栈传递而非寄存器编号：

```python
# 理解栈式执行：加法运算 a + b
# 源代码: result = x + y
# 字节码:
#   LOAD_FAST    0 (x)    -- 将x压入栈
#   LOAD_FAST    1 (y)    -- 将y压入栈
#   BINARY_ADD            -- 弹出两个值，相加，结果压入栈
#   STORE_FAST    2 (result) -- 弹出栈顶，存入result

# 对比寄存式虚拟机(如Lua VM)的伪代码：
#   ADD R2, R0, R1  -- R2 = R0 + R1

# 栈式虚拟机的特点：
# 1. 指令更短(无需指定寄存器编号)
# 2. 编译器更简单
# 3. 但指令数量更多(需要频繁的栈操作)
```

## 39.2 原理与机制

### 39.2.1 编译流水线：从源码到字节码

CPython 的编译过程在 C 源码中经历了以下阶段：

```
源码(.py) → 词法分析(tokenize) → 语法分析(parse) → AST → 符号表分析 → 字节码生成 → Code Object
```

在 C 源码层面，核心数据结构为：

```c
// Python/compile.c — 编译器状态结构体
typedef struct compiler {
    PyObject *c_filename;      // 文件名
    PyObject *c_const_cache;   // 常量缓存
    struct symtable *c_st;     // 符号表
    int c_flags;               // 编译标志
    PyFutureFeatures c_future; // __future__ 特性

    struct compiler_unit *c_u; // 当前编译单元
    PyObject *c_stack;         // 编译单元栈
    int c_nblocks;             // 代码块计数
    // ...
} compiler;

// Python/symtable.c — 符号表条目
typedef struct _symtable_entry {
    PyObject *ste_id;        // 标识符
    PyObject *ste_symbols;   // 符号字典{name: flags}
    PyObject *ste_name;      // 作用域名
    int ste_type;            // Module, Class, Function等
    int ste_varargs;         // *args标志
    int ste_varkeywords;     // **kwargs标志
    int ste_nested;          // 嵌套标志
    int ste_free;            // 自由变量标志
    int ste_child_offset;    // 子作用域偏移
    // ...
} PyST_EntryObject;
```

### 39.2.2 字节码编码格式

每条字节码指令的编码方式如下：

```
# Python 3.6+ 使用 wordcode 格式：
# 每条指令占2字节 = 1字节opcode + 1字节oparg
# 当oparg > 255时，使用 EXTENDED_ARG 前缀指令

# 例如: LOAD_CONST 260
# 编码为:
#   offset 0: EXTENDED_ARG 1    (0x90, 0x01)  -- 1 << 8 = 256
#   offset 2: LOAD_CONST  4    (0x64, 0x04)  -- 256 + 4 = 260
```

```c
// Python/wordcode_helpers — 字节码编码辅助
// 每条指令 = 2 bytes: opcode(1 byte) + oparg(1 byte)
#define BYTECODE_SIZE 2

// Python/compile.c — 发出一条指令
static int
compiler_addop_i(struct compiler *c, int opcode, Py_ssize_t oparg)
{
    // oparg > 0xFFFFFF 时报错
    if (oparg > 0xFFFFFF) {
        PyErr_SetString(PyExc_OverflowError, "oparg too large");
        return -1;
    }
    // 处理扩展参数：每8位一个EXTENDED_ARG
    int extended = 0;
    Py_ssize_t val = oparg;
    while (val >> 8) {
        extended++;
        val >>= 8;
    }
    // 逆序插入EXTENDED_ARG指令
    for (int i = extended; i > 0; i--) {
        Py_ssize_t shift = (i - 1) * 8;
        int ext_arg = (oparg >> shift) & 0xFF;
        ADDOP_I(c, EXTENDED_ARG, ext_arg);
    }
    ADDOP_I(c, opcode, oparg & 0xFF);
    return 1;
}
```

### 39.2.3 帧对象与执行帧栈

帧(frame)是 CPython 执行模型的核心数据结构，每个函数调用都会创建一个新的帧对象：

```c
// Include/internal/pycore_frame.h — 帧结构定义(Python 3.11+)
typedef struct _frame {
    PyObject_VAR_HEAD
    // ---- 以下是可变部分 ----
    _Py_CODEUNIT *prev_instr;   // 上一条执行的指令位置
    PyObject *func_obj;          // 函数对象(可能为NULL)
    PyObject *localsplus[1];     // 局部变量+单元格+自由变量(变长数组)
} _PyInterpreterFrame;

// Python 3.11之前使用 PyFrameObject(Python/frameobject.c)
typedef struct {
    PyObject_HEAD
    PyCodeObject *f_code;       // 代码对象
    PyObject *f_globals;        // 全局命名空间
    PyObject *f_locals;         // 局部命名空间
    PyObject *f_builtins;       // 内置命名空间
    PyObject **f_localsplus;    // locals+cells+free 合并数组
    int f_lasti;                // 上次执行的字节码偏移
    int f_lineno;               // 当前行号
    char f_trace_lines;         // 是否追踪行事件
    char f_trace_opcodes;       // 是否追踪操作码事件
    // ...
} PyFrameObject;
```

帧栈管理机制：

```c

// Python/pythonrun.c — 顶层帧创建
PyStatus _PyRun_SimpleFileObject(FILE *fp, PyObject *filename, int close)

    // 执行模块字节码

```

Python 层面访问帧对象：

```python
import sys

def demonstrate_frames():
    """演示帧栈的层级关系"""
    current_frame = sys._getframe(0)  # 当前帧
    caller_frame = sys._getframe(1)    # 调用者帧

    # 帧对象的核心属性
    print(f"当前函数: {current_frame.f_code.co_name}")
    print(f"代码对象: {current_frame.f_code}")
    print(f"全局命名空间: {len(current_frame.f_globals)} 个条目")
    print(f"局部变量: {current_frame.f_locals}")
    print(f"上次执行的偏移: {current_frame.f_lasti}")
    print(f"当前行号: {current_frame.f_lineno}")

    # 帧栈遍历
    frame = current_frame
    depth = 0
    while frame is not None:
        print(f"  [{depth}] {frame.f_code.co_name} "
              f"({frame.f_code.co_filename}:{frame.f_lineno})")
        frame = frame.f_back
        depth += 1

demonstrate_frames()
```

### 39.2.4 Python 3.11+ 的 Specializing Adaptive Interpreter

PEP 659 引入了 **Specializing Adaptive Interpreter**，这是 CPython 性能优化最重要的改进之一。其核心思想是：**在运行时将通用的字节码指令替换为针对特定类型优化的特化指令**。

```c
// Include/internal/pycore_opcode.h — 特化指令示例
// 通用指令在运行时可能被替换为：
#define LOAD_ATTR              106  // 通用属性加载
#define LOAD_ATTR_INSTANCE_VALUE  1 // 优化: 实例属性(通过偏移量直接访问)
#define LOAD_ATTR_MODULE          2 // 优化: 模块属性(缓存字典查找)
#define LOAD_ATTR_WITH_HINT       3 // 优化: 带字典偏移提示的属性

#define BINARY_OP              107  // 通用二元运算
#define BINARY_OP_ADD_INT         1 // 优化: 整数加法
#define BINARY_OP_ADD_FLOAT       2 // 优化: 浮点数加法
#define BINARY_OP_ADD_UNICODE     3 // 优化: 字符串拼接
#define BINARY_OP_MULTIPLY_INT    4 // 优化: 整数乘法
#define BINARY_OP_MULTIPLY_FLOAT  5 // 优化: 浮点数乘法
```

特化机制的工作流程：

```c
// Python/ceval.c — 内联缓存与特化
// 1. 初始执行: 使用通用指令(adaptive opcode)
// 2. 执行若干次后(默认8次): 尝试特化
// 3. 特化后: 使用针对特定类型优化的快速路径
// 4. 如果类型不匹配: 去特化(deoptimize)回通用指令

// Include/internal/pycore_inline_cache.h
typedef struct {
    uint16_t counter;    // 执行计数器
    uint16_t index;      // 特化索引/缓存索引
    uint16_t module_keys_version;  // 模块字典版本
    uint16_t builtin_keys_version; // 内置字典版本
} _PyAdaptiveEntry;

// 特化阈值
#define ADAPTIVE_CACHE_BACKOFF 64  // 退避阈值
// 当特化失败时，counter 被重置为这个值，避免频繁重试
```

```python
# 观察特化过程
import dis
import opcode

def add_numbers(a, b):
    return a + b

# 初始字节码
print("=== 特化前 ===")
dis.dis(add_numbers)

# 执行多次触发特化
for i in range(100):
    add_numbers(1, 2)  # 始终传入int参数

print("\n=== 特化后 ===")
dis.dis(add_numbers)

# 查看指令是否发生特化
code = add_numbers.__code__
print(f"\nco_code 长度: {len(code.co_code)}")
print(f"co_consts: {code.co_consts}")
print(f"co_varnames: {code.co_varnames}")
```

## 39.3 使用场景

### 39.3.1 性能分析与调优

```python
import dis
import time
import opcode

def slow_loop(items):
    """使用 + 拼接字符串 — 低效"""
    result = ""
    for item in items:
        result += str(item)
    return result

def fast_loop(items):
    """使用列表拼接 — 高效"""
    parts = []
    append = parts.append  # 缓存方法查找
    for item in items:
        append(str(item))
    return "".join(parts)

# 对比字节码
print("=== slow_loop 字节码 ===")
dis.dis(slow_loop)
print("\n=== fast_loop 字节码 ===")
dis.dis(fast_loop)

# 性能对比
data = list(range(10000))
start = time.perf_counter()
slow_loop(data)
t1 = time.perf_counter()
fast_loop(data)
t2 = time.perf_counter()
print(f"\nslow: {t1-start:.4f}s, fast: {t2-t1:.4f}s")
```

### 39.3.2 代码混淆与保护

```python
import dis
import types

def original_function(x, y):
    """原始函数逻辑"""
    return x * x + y * y

# 获取字节码
code_obj = original_function.__code__
print("原始字节码:")
dis.dis(code_obj)

# 创建修改后的代码对象(修改常量表)
new_consts = tuple(
    -c if isinstance(c, int) else c  # 翻转整数常量
    for c in code_obj.co_consts
)

# 注意: 这仅作演示，实际混淆需要同步修改操作码
# 以下仅展示 Code Object 的构造方式
new_code = types.CodeType(
    code_obj.co_argcount,       # 参数数量
    code_obj.co_posonlyargcount, # 仅位置参数数量
    code_obj.co_kwonlyargcount,  # 仅关键字参数数量
    code_obj.co_nlocals,        # 局部变量数
    code_obj.co_stacksize,       # 栈大小
    code_obj.co_flags,           # 标志
    code_obj.co_code,            # 字节码(未修改)
    new_consts,                  # 修改后的常量表
    code_obj.co_names,           # 名称表
    code_obj.co_varnames,        # 变量名表
    code_obj.co_filename,        # 文件名
    code_obj.co_name,            # 函数名
    code_obj.co_firstlineno,     # 首行号
```

```python
"""
comprehensive_bytecode_analyzer.py — 完整的字节码分析工具

可独立运行，深度分析Python字节码的各个层面

import dis
import sys
import types
import opcode
import struct
from collections import defaultdict, Counter


class BytecodeAnalyzer:
    """全面的字节码分析器"""

    def __init__(self, func_or_code):
        if isinstance(func_or_code, types.FunctionType):
            self.func = func_or_code
            self.code = func_or_code.__code__
        elif isinstance(func_or_code, types.CodeType):
            self.func = None
            self.code = func_or_code
        else:
            raise TypeError(f"不支持类型: {type(func_or_code)}")

    def summary(self):
        """输出代码对象概要"""
        c = self.code
        print(f"=== 代码对象概要 ===")
        print(f"名称: {c.co_name}")
        print(f"文件: {c.co_filename}")
        print(f"首行号: {c.co_firstlineno}")
        print(f"参数数量: {c.co_argcount}")
        print(f"仅位置参数: {c.co_posonlyargcount}")
        print(f"仅关键字参数: {c.co_kwonlyargcount}")
        print(f"局部变量数: {c.co_nlocals}")
        print(f"栈大小: {c.co_stacksize}")
        print(f"标志: {self._decode_flags(c.co_flags)}")
        print(f"常量: {c.co_consts}")
        print(f"名称: {c.co_names}")
        print(f"变量名: {c.co_varnames}")
        print(f"自由变量: {c.co_freevars}")
        print(f"单元格变量: {c.co_cellvars}")
        print(f"字节码长度: {len(c.co_code)} 字节")

    def _decode_flags(self, flags):
        """解码代码对象标志"""
        flag_names = {
            0x01: "OPTIMIZED",
            0x02: "NEWLOCALS",
            0x04: "VARARGS",
            0x08: "VARKEYWORDS",
            0x10: "NESTED",
            0x20: "GENERATOR",
            0x40: "NOFREE",
            0x100: "COROUTINE",
            0x200: "ITERABLE_COROUTINE",
            0x400: "ASYNC_GENERATOR",
        }
        result = []
        for bit, name in flag_names.items():
            if flags & bit:
                result.append(name)
        return " | ".join(result) if result else f"0x{flags:x}"

    def instruction_stats(self):
        """统计指令使用频率"""
        instructions = list(dis.Bytecode(self.code))
        counter = Counter(instr.opname for instr in instructions)

        print(f"\n=== 指令统计 (共{len(instructions)}条) ===")
        for opname, count in counter.most_common():
            print(f"  {opname:<30} {count:3d}次  "
                  f"({count/len(instructions)*100:.1f}%)")

        # 按类别分组
        categories = defaultdict(int)
        for instr in instructions:
            cat = self._categorize_opcode(instr.opname)
            categories[cat] += 1

        print(f"\n=== 指令类别统计 ===")
        for cat, count in sorted(categories.items(),
                                  key=lambda x: -x[1]):
            print(f"  {cat:<20} {count:3d}次  "
                  f"({count/len(instructions)*100:.1f}%)")

    def _categorize_opcode(self, opname):
        """将操作码归类"""
        if opname.startswith('LOAD_') or opname.startswith('STORE_') or \
           opname.startswith('DELETE_'):
            return "加载/存储"
        elif opname.startswith('BINARY_') or opname.startswith('UNARY_'):
            return "算术运算"
        elif 'JUMP' in opname or opname in ('FOR_ITER', 'SETUP_FINALLY',
                                              'POP_BLOCK'):
            return "控制流"
        elif opname.startswith('CALL') or opname == 'RETURN_VALUE' or \
             opname == 'RETURN_CONST':
            return "函数调用"
        elif opname.startswith('BUILD_'):
            return "数据结构"
        elif opname in ('GET_ITER', 'YIELD_VALUE', 'RESUME', 'SEND'):
            return "迭代/生成器"
        elif 'EXC' in opname or opname in ('RAISE_VARARGS', 'RERAISE',
                                             'POP_EXCEPT'):
            return "异常处理"
        else:
            return "其他"

    def stack_simulation(self):
        """模拟栈操作，展示执行过程中的栈状态变化"""
        print(f"\n=== 栈模拟 ===")
        instructions = list(dis.Bytecode(self.code))
        stack = []
        max_depth = 0

        for instr in instructions:
            # 模拟栈效果
            delta = self._stack_effect(instr)
            before = len(stack)
            stack_change = ""

            if delta > 0:
                for _ in range(delta):
                    stack.append(f"<{instr.opname}>")
                stack_change = f"栈+{delta}"
            elif delta < 0:
                popped = []
                for _ in range(-delta):
                    if stack:
                        popped.append(stack.pop())
                stack_change = f"栈{delta}"
            else:
                stack_change = "栈不变"

            if instr.opname == 'RETURN_VALUE':
                if stack:
                    stack.pop()
                stack_change = "返回"

            max_depth = max(max_depth, len(stack))
            print(f"  {instr.offset:3d} {instr.opname:<25} "
                  f"{instr.argrepr:<15} "
                  f"栈深={len(stack):2d} {stack_change}")

        print(f"  最大栈深度: {max_depth} "
              f"(声明的co_stacksize={self.code.co_stacksize})")

    def _stack_effect(self, instr):
        """计算指令的栈效果"""
        try:
            # 使用 dis.stack_effect 获取精确的栈效果
            return dis.stack_effect(instr.opcode, instr.arg)
        except Exception:
            # 回退到估算
            name = instr.opname
            if name.startswith('LOAD_') or name == 'DUP_TOP':
                return 1
            elif name.startswith('STORE_') or name.startswith('DELETE_'):
                return -1
            elif name == 'POP_TOP':
                return -1
            elif name == 'RETURN_VALUE':
                return -1
            elif name.startswith('BUILD_'):
                return -instr.arg + 1 if instr.arg else 1
            elif name.startswith('BINARY_'):
                return -1
            elif name.startswith('UNARY_'):
                return 0
            elif name.startswith('CALL'):
                return -instr.arg if instr.arg else 0
            return 0

    def control_flow_graph(self):
        """简单的控制流图分析"""
        print(f"\n=== 控制流分析 ===")
        instructions = list(dis.Bytecode(self.code))
        jump_targets = set()
        jumps = []

        for instr in instructions:
            if instr.is_jump_target:
                jump_targets.add(instr.offset)
            if 'JUMP' in instr.opname or instr.opname in (
                'FOR_ITER', 'SETUP_FINALLY', 'POP_JUMP_IF_TRUE',
                'POP_JUMP_IF_FALSE', 'POP_JUMP_IF_NOT_NONE',
                'POP_JUMP_IF_NONE'
            ):
                target = instr.argval if instr.argval is not None else instr.arg
                jumps.append((instr.offset, instr.opname, target))

        print(f"  跳转目标: {sorted(jump_targets)}")
        print(f"  跳转指令:")
        for offset, opname, target in jumps:
            print(f"    {offset:3d} {opname} → {target}")

    def full_analysis(self):
        """执行完整分析"""
        print(f"\n{'='*60}")
        print(f"  字节码分析: {self.code.co_name}")
        print(f"{'='*60}")
        self.summary()
        self.instruction_stats()
        self.stack_simulation()
        self.control_flow_graph()
        print(f"\n=== 原始字节码 ===")
        dis.dis(self.code)
```

```python
class FrameInspector:
    """帧对象深度检查器"""

    @staticmethod
    def inspect_frame(frame: FrameType, indent: int = 0) -> str:
        """递归检查帧对象链"""
        prefix = "  " * indent
        lines = []

        c = frame.f_code
        lines.append(f"{prefix}┌─ Frame: {c.co_name}")
        lines.append(f"{prefix}│ 文件: {c.co_filename}:{frame.f_lineno}")
        lines.append(f"{prefix}│ 参数: {c.co_varnames[:c.co_argcount]}")
        lines.append(f"{prefix}│ 局部变量数: {c.co_nlocals}")
        lines.append(f"{prefix}│ 栈大小: {c.co_stacksize}")
        lines.append(f"{prefix}│ 标志: 0x{c.co_flags:x}")

        # 局部变量值
        for name in c.co_varnames[:c.co_nlocals]:
            value = frame.f_locals.get(name, "<未绑定>")
            # 防止递归打印
            value_str = repr(value)[:80]
            lines.append(f"{prefix}│ {name} = {value_str}")

        lines.append(f"{prefix}└─")

        if frame.f_back:
            lines.append(FrameInspector.inspect_frame(frame.f_back, indent + 1))

        return "\n".join(lines)

    @staticmethod
    def trace_function(frame: FrameType, event: str, arg):
        """跟踪函数调用事件"""
        code = frame.f_code

        if event == 'call':
            print(f"\n>>> CALL {code.co_name} "
                  f"(args={code.co_varnames[:code.co_argcount]})")
            return FrameInspector.trace_function

        elif event == 'return':
            print(f"<<< RETURN from {code.co_name} = {arg!r:.80}")

        elif event == 'line':
            pass  # 行事件太频繁，跳过

        elif event == 'exception':
            exc_type, exc_value, exc_tb = arg
            print(f"!!! EXCEPTION in {code.co_name}: "
                  f"{exc_type.__name__}: {exc_value}")

        return None
```

```python
import sys

# 陷阱1: 版本差异
```

### 39.5.3 闭包与变量捕获陷阱

```python
"""

字节码层面理解：LOAD_DEREF 加载的是 cell 对象的引用，

# 陷阱：循环闭包共享i

    funcs = []
    for i in range(5):
        def func():
            return i
```

### 39.5.5 字节码优化最佳实践

```python
"""
最佳实践：利用字节码知识优化Python代码
"""

import dis
import time

# 最佳实践1: 缓存属性查找
class DataProcessor:
    def __init__(self, data):
        self.data = data

    def process_slow(self):
        """每次迭代都查找属性"""
        result = []
        for item in self.data:
            result.append(item * 2)  # 每次查找self.data
        return result

    def process_fast(self):
        """缓存属性查找"""
        result = []
        data = self.data       # 缓存self.data
        append = result.append  # 缓存list.append
        for item in data:
            append(item * 2)
        return result

# 字节码对比
processor = DataProcessor(list(range(1000)))
print("=== process_slow 字节码 ===")
dis.dis(processor.process_slow)
print("\n=== process_fast 字节码 ===")
dis.dis(processor.process_fast)

# 最佳实践2: 局部变量优于全局变量
g_x = 10

def global_access(n):
    """访问全局变量 — 每次LOAD_GLOBAL"""
    total = 0
    for i in range(n):
        total += i + g_x
    return total

def local_access(n):
    """访问局部变量 — LOAD_FAST更快"""
    total = 0
    x = g_x  # 缓存到局部变量
    for i in range(n):
        total += i + x
    return total

# 字节码对比
print("\n=== global_access 字节码 ===")
dis.dis(global_access)
print("\n=== local_access 字节码 ===")
dis.dis(local_access)

# 性能对比
n = 1000000
start = time.perf_counter()
global_access(n)
t1 = time.perf_counter()
local_access(n)
t2 = time.perf_counter()

print(f"\n全局变量访问: {t1-start:.4f}s")
print(f"局部变量访问: {t2-t1:.4f}s")
print(f"加速比: {(t1-start)/(t2-t1):.2f}x")

# 最佳实践3: 避免不必要的属性查找
import math

def calc_slow(angle):
    """每次都查找math.sin"""
    return math.sin(angle) + math.cos(angle)

def calc_fast(angle):
    """缓存math模块函数"""
    sin = math.sin
    cos = math.cos
    return sin(angle) + cos(angle)

print("\n=== calc_slow 字节码 ===")
dis.dis(calc_slow)
print("\n=== calc_fast 字节码 ===")
dis.dis(calc_fast)

# 最佳实践4: 使用列表推导式而非循环+append
def loop_append(data):
    result = []
    for item in data:
        result.append(item ** 2)
    return result

def list_comp(data):
    return [item ** 2 for item in data]

print("\n=== loop_append 字节码 ===")
dis.dis(loop_append)
print("\n=== list_comp 字节码 ===")
dis.dis(list_comp)
# 列表推导式使用 LIST_APPEND 指令，
# 比方法调用(result.append)快得多
```

### 39.5.6 特化与去特化陷阱

```python
"""
陷阱：类型多态导致去特化，反而比不特化更慢

Python 3.11+的adaptive interpreter在类型不稳定时会反复
特化与去特化，造成性能下降
"""

import time
import dis

def polymorphic_add(a, b):
    """多态加法 — 类型不稳定"""
    return a + b

def stable_int_add(a, b):
    """稳定整数加法 — 类型一致"""
    return a + b

# 陷阱: 交替传入不同类型
def benchmark_polymorphic():
    # 类型不稳定 → 反复特化/去特化
    total = 0
    for i in range(100000):
        if i % 2 == 0:
            total += polymorphic_add(1, 2)
        else:
            total += polymorphic_add(1.0, 2.0)
    return total

def benchmark_stable():
    # 类型稳定 → 一次特化
    total = 0
    for i in range(100000):
        total += stable_int_add(1, 2)
    return total

# 最佳实践: 保持类型稳定
# 如果必须处理多种类型，使用类型分发
def typed_add_int(a: int, b: int) -> int:
    return a + b

def typed_add_float(a: float, b: float) -> float:
    return a + b

def smart_add(a, b):
    """根据类型分发到特化函数"""
    if isinstance(a, int) and isinstance(b, int):
        return typed_add_int(a, b)
    elif isinstance(a, float) and isinstance(b, float):
        return typed_add_float(a, b)
    return a + b

# 性能测试
start = time.perf_counter()
benchmark_polymorphic()
t1 = time.perf_counter()
benchmark_stable()
t2 = time.perf_counter()

print(f"多态类型: {t1-start:.4f}s")
print(f"稳定类型: {t2-t1:.4f}s")
if t1 > t2:
    print(f"稳定类型更快 {(t1-start)/(t2-t1):.2f}x")

# 观察特化指令
if hasattr(dis, 'specialize'):
    print("\n=== 多态函数特化状态 ===")
    # 反复混合类型调用后查看
    for i in range(100):
        polymorphic_add(1, 2)
        polymorphic_add(1.0, 2.0)
    dis.dis(polymorphic_add)
```

### 39.5.7 关键最佳实践总结

```python
"""
字节码与执行模型 — 最佳实践清单
"""

# 1. 使用dis模块理解代码行为，而非猜测
#    dis.dis(your_function) 是最好的性能分析起点

# 2. 保持类型稳定以利用特化
#    在热循环中避免混用不同类型的操作数

# 3. 缓存属性和方法查找
#    obj.method 比 method = obj.method 更慢
#    self.attr 比 local_var = self.attr 更慢

# 4. 优先使用局部变量而非全局变量
#    LOAD_FAST 比 LOAD_GLOBAL 快得多

# 5. 使用列表/字典推导式代替循环+append
#    LIST_APPEND 比 方法调用快

# 6. 不要依赖字节码的稳定性
#    字节码是CPython实现细节，版本间会变更

# 7. 避免在热路径上使用sys._getframe()
#    帧访问有显著性能开销

# 8. 使用code.replace()修改Code Object
#    不要直接修改不可变字段

# 9. 理解闭包变量捕获机制
#    使用默认参数或工厂函数避免共享闭包变量

# 10. 使用timeit而非手动time.perf_counter()
#     避免微观基准测试中的常见错误
import timeit

result = timeit.timeit('sum(range(100))', number=10000)
print(f"sum(range(100)) x 10000: {result:.4f}s")
```
