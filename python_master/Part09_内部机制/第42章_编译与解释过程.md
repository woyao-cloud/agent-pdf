# 第42章 编译与解释过程

## 42.1 概念与语法

### 42.1.1 Python 编译与解释全景

Python 的执行模型是**先编译后解释**：源代码首先被编译为字节码，再由虚拟机逐条解释执行。这个编译过程是自动的、运行时的、对用户透明的。

```
源代码(.py) → 词法分析 → Token流
                ↓
           语法分析 → CST(具体语法树)
                ↓
           AST构建 → 抽象语法树
                ↓
        符号表分析 → 作用域与绑定信息
                ↓
           字节码生成 → Code Object
                ↓
           虚拟机执行 → 运行结果
```

```python
# 编译与解释的核心API

# 1. compile() — 将源码编译为Code Object
source = "x = 1 + 2"
code = compile(source, '<string>', 'exec')
print(f"Code Object: {code}")
print(f"字节码: {code.co_code.hex()}")

# 2. eval() — 执行表达式字符串
result = eval("1 + 2")
print(f"eval结果: {result}")

# 3. exec() — 执行语句字符串
exec("y = 1 + 2")
print(f"exec结果: y = {y}")

# 4. ast模块 — 抽象语法树操作
import ast
tree = ast.parse("x = 1 + 2")
print(f"AST: {ast.dump(tree, indent=2)}")

# 5. dis模块 — 反汇编字节码
import dis
dis.dis(code)
```

## 42.2 原理与机制

### 42.2.1 词法分析(Tokenization)

词法分析是编译的第一步，将源代码字符流分解为 Token 流。CPython 使用 `tokenize` 模块和 C 层面的 `tokenizer` 实现：

```c
// Parser/tokenizer.c — 词法分析器核心结构
typedef struct {
    // 输入状态
    const char *input;          // 输入字符串
    FILE *fp;                   // 或文件指针
    char *buf;                  // 当前缓冲区
    char *cur;                  // 当前位置指针
    char *end;                  // 缓冲区结束

    // Token状态
    int tok_state;              // 状态机当前状态
    int indent;                 // 当前缩进级别
    int atbol;                  // 是否在行首

    // 缩进栈(用于处理缩进/反缩进)
    PyObject *indent_stack;     // 缩进级别栈
    int indent_level;           // 当前缩进级别

    // 行追踪
    int lineno;                 // 当前行号
    int col_offset;             // 当前列偏移

    // 模式
    int flags;                  // 编译标志
    // ... 其他字段
} tok_state;

// 词法分析状态机
// 识别: 标识符、关键字、数字、字符串、运算符、分隔符、缩进等
```

### 42.2.2 语法分析(Parsing)

语法分析将 Token 流构建为**具体语法树(CST)**，再转换为**抽象语法树(AST)**：

```c
// Parser/parser.c — PEG解析器(Python 3.9+)
// Python 3.9之前使用LL(1)解析器
// Python 3.9+使用PEG(Packrat)解析器

// PEG解析器的核心优势:
// 1. 无回溯限制(与LL(1)不同)
// 2. 支持左递归(通过特殊处理)
// 3. 语法更直观自然
// 4. 错误信息更友好

// Python/peg_parser.c — PEG解析器入口
// 解析函数声明在 Parser/parser.h 中自动生成
// 语法规则定义在 Grammar/python.gram 中

// 示例: 表达式语法规则(简化)
// expr:
//     | disjunction
// disjunction:
//     | conjunction ('or' conjunction)*
// conjunction:
//     | inversion ('and' inversion)*
// inversion:
//     | 'not' inversion
//     | comparison
// comparison:
//     | bitwise_or (comp_op bitwise_or)*
```

```python
# 观察解析过程

source = "x = 1 + 2 * 3"
```


### 42.2.4 符号表分析

符号表分析是连接AST和字节码生成的关键步骤，它确定每个名称的**作用域**和**绑定类型**：

```c
// Python/symtable.c — 符号表核心数据结构
typedef struct _symtable_entry {
    PyObject *ste_id;          // 标识符(哈希值)
    PyObject *ste_symbols;     // 符号字典 {name: flags}
    PyObject *ste_name;        // 作用域名(函数名/类名)
    int ste_type;              // 作用域类型
    int ste_varargs;           // 是否有*args
    int ste_varkeywords;       // 是否有**kwargs
    int ste_nested;            // 是否是嵌套作用域
    int ste_free;              // 是否有自由变量
    int ste_child_offset;      // 子作用域偏移
    int ste_needs_class_closure; // 类作用域闭包标志
    PyObject *ste_directives;  // 作用域指令(global/nonlocal)
    // ...
} PyST_EntryObject;

// 作用域类型
#define SCOPE_MODULE    0     // 模块级作用域
#define SCOPE_CLASS     1     // 类级作用域
#define SCOPE_FUNCTION  2     // 函数级作用域
#define SCOPE_COMP      3     // 推导式作用域

// 符号标志位
#define DEF_GLOBAL      0x01  // global声明
#define DEF_NONLOCAL    0x02  // nonlocal声明
#define DEF_LOCAL       0x04  // 本地定义
#define USE             0x08  // 使用
#define DEF_FREE        0x10  // 自由变量
#define DEF_FREE_CLASS  0x20  // 类作用域中的自由变量
#define DEF_IMPORT      0x40  // import导入
#define DEF_BOUND       (DEF_LOCAL | DEF_GLOBAL | DEF_NONLOCAL)
```

```python
# 使用symtable模块查看符号表
import symtable

source = """
def outer(x, y):
    z = x + y
    def inner(a):
        return a + z
    return inner
"""

table = symtable.symtable(source, '<string>', 'exec')

def print_symtable(st, indent=0):
    """递归打印符号表"""
    prefix = "  " * indent
    print(f"{prefix}作用域: {st.get_name()} (类型: {st.get_type()})")

    symbols = st.get_symbols()
    for sym in symbols:
        flags = []
        if sym.is_global(): flags.append("global")
        if sym.is_local(): flags.append("local")
        if sym.is_free(): flags.append("free")
        if sym.is_parameter(): flags.append("param")
        if sym.is_imported(): flags.append("import")
        if sym.is_assigned(): flags.append("assigned")
        if sym.is_referenced(): flags.append("referenced")

        print(f"{prefix}  {sym.get_name()}: {', '.join(flags)}")

    for child in st.get_children():
        print_symtable(child, indent + 1)

print_symtable(table)
```

输出示例：
```
作用域: top (类型: module)
  outer: local, assigned, referenced
  作用域: outer (类型: function)
    x: local, param, referenced
    y: local, param, referenced
    z: local, assigned, referenced
    inner: local, assigned
    作用域: inner (类型: function)
      a: local, param, referenced
      z: free, referenced
```

### 42.2.5 字节码生成

字节码生成是编译的最后阶段，将AST和符号表信息转换为 Code Object：

```c
// Python/compile.c — 编译器核心结构
typedef struct compiler {
    PyObject *c_filename;        // 文件名
    PyObject *c_const_cache;     // 常量缓存(去重)
    struct symtable *c_st;        // 符号表
    int c_flags;                  // 编译标志

    // 编译单元栈(用于嵌套函数/类)
    struct compiler_unit *c_u;    // 当前编译单元
    PyObject *c_stack;           // 编译单元栈

    int c_nblocks;               // 代码块计数
    // ...
} compiler;

// 编译单元 — 每个函数/类对应一个
typedef struct compiler_unit {
    PyCodeObject *u_code;        // 生成的代码对象
    PyObject *u_consts;          // 常量表
    PyObject *u_names;            // 名称表
    PyObject *u_varnames;        // 变量名表
    PyObject *u_cellvars;        // 单元格变量表
    PyObject *u_freevars;        // 自由变量表

    int u_argcount;              // 参数数量
    int u_nlocals;               // 局部变量数量
    int u_stacksize;              // 栈大小

    // 基本块列表(字节码生成单位)
    basicblock *u_blocks;        // 基本块链表
    basicblock *u_curblock;      // 当前基本块

    // 异常处理器栈
    struct excepthandler *u_excepthandlers;
    // ...
} compiler_unit;

// 字节码生成的主要流程:
// 1. compiler_mod() — 编译模块
//    → compiler_body() — 编译模块体
//      → compiler_stmt() — 编译每条语句
//        → compiler_expr() — 编译表达式
//          → compiler_addop_*() — 发出指令

// 2. 关键编译函数:
static int compiler_function(struct compiler *c, stmt_ty s);
static int compiler_class(struct compiler *c, stmt_ty s);
static int compiler_lambda(struct compiler *c, expr_ty s);
static int compiler_if(struct compiler *c, stmt_ty s);
static int compiler_for(struct compiler *c, stmt_ty s);
static int compiler_while(struct compiler *c, stmt_ty s);
static int compiler_try(struct compiler *c, stmt_ty s);
static int compiler_match(struct compiler *c, stmt_ty s);

// 3. 指令发射函数:
static int compiler_addop_i(struct compiler *c, int opcode, Py_ssize_t oparg);
static int compiler_addop_j(struct compiler *c, int opcode, basicblock *target);
static int compiler_addop_o(struct compiler *c, int opcode, PyObject *obj);
```

### 42.3.1 代码静态分析

```python
"""
static_analyzer.py — 基于AST的静态分析工具
"""

import ast
import sys
from collections import defaultdict
from typing import List, Dict, Set, Tuple


class CodeAnalyzer(ast.NodeVisitor):
    """代码静态分析器

    分析:
    - 函数复杂度(圈复杂度)
    - 未使用的导入
    - 变量使用情况
    - 代码风格问题
    """

    def __init__(self, filename='<string>'):
        self.filename = filename
        self.functions = []
        self.classes = []
        self.imports = []
        self.unused_imports = set()
        self.used_names = set()
        self.global_vars = set()
        self.complexity = {}
        self.issues = []

    def analyze(self, source_code):
        """分析源代码"""
        tree = ast.parse(source_code, self.filename)
        self.visit(tree)
        self._check_unused_imports()
        return self._generate_report()

    def visit_FunctionDef(self, node):
        """访问函数定义"""
        # 计算圈复杂度
        complexity = self._calculate_complexity(node)
        self.complexity[node.name] = complexity

        # 检查函数参数数量
        if len(node.args.args) > 5:
            self.issues.append({
                'type': 'style',
                'message': f"函数 '{node.name}' 参数过多({len(node.args.args)}个)",
                'line': node.lineno,
            })

        # 检查函数体是否为空
        if not node.body or (len(node.body) == 1 and
                isinstance(node.body[0], ast.Pass)):
            self.issues.append({
                'type': 'warning',
                'message': f"函数 '{node.name}' 体为空",
                'line': node.lineno,
            })

        self.functions.append({
            'name': node.name,
            'line': node.lineno,
            'args': [arg.arg for arg in node.args.args],
            'decorators': [self._get_decorator_name(d)
                          for d in node.decorator_list],
            'returns': ast.unparse(node.returns) if node.returns else None,
            'complexity': complexity,
        })
        self.generic_visit(node)

    visit_AsyncFunctionDef = visit_FunctionDef

    def visit_ClassDef(self, node):
        """访问类定义"""
        bases = [ast.unparse(b) for b in node.bases]
        self.classes.append({
            'name': node.name,
            'line': node.lineno,
            'bases': bases,
            'methods': [n.name for n in node.body
                       if isinstance(n, (ast.FunctionDef,
                                        ast.AsyncFunctionDef))],
        })
        self.generic_visit(node)

    def visit_Import(self, node):
        """访问import语句"""
        for alias in node.names:
            name = alias.asname or alias.name
            self.imports.append({
                'module': alias.name,
                'alias': alias.asname,
                'line': node.lineno,
            })
            self.unused_imports.add(name)
        self.generic_visit(node)

    def visit_ImportFrom(self, node):
        """访问from ... import语句"""
        for alias in node.names:
            name = alias.asname or alias.name
            self.imports.append({
                'module': f"{node.module}.{alias.name}",
                'alias': alias.asname,
                'line': node.lineno,
            })
            if name != '*':
                self.unused_imports.add(name)
        self.generic_visit(node)

    def visit_Name(self, node):
        """访问名称引用"""
        if isinstance(node.ctx, ast.Load):
            self.used_names.add(node.id)
            self.unused_imports.discard(node.id)
        self.generic_visit(node)

    def visit_Attribute(self, node):
        """访问属性引用"""
        if isinstance(node.value, ast.Name):
            self.used_names.add(node.value.id)
            self.unused_imports.discard(node.value.id)
        self.generic_visit(node)

    def visit_Global(self, node):
        """访问global声明"""
        for name in node.names:
            self.global_vars.add(name)
        self.generic_visit(node)

    def _calculate_complexity(self, node):
        """计算圈复杂度(McCabe)"""
        complexity = 1  # 基础复杂度

        for child in ast.walk(node):
            # 分支语句增加复杂度
            if isinstance(child, (ast.If, ast.While, ast.For,
                                  ast.ExceptHandler)):
                complexity += 1
            elif isinstance(child, ast.BoolOp):
                complexity += len(child.values) - 1
            elif isinstance(child, (ast.ListComp, ast.SetComp,
                                    ast.DictComp, ast.GeneratorExp)):
                for gen in child.generators:
                    complexity += len(gen.ifs)
            elif isinstance(child, ast.Assert):
                complexity += 1
            elif isinstance(child, ast.Match):
                complexity += 1  # match语句本身

        return complexity

    def _get_decorator_name(self, node):
        """获取装饰器名称"""
        if isinstance(node, ast.Name):
            return node.id
        elif isinstance(node, ast.Attribute):
            return f"{self._get_decorator_name(node.value)}.{node.attr}"
        elif isinstance(node, ast.Call):
            return self._get_decorator_name(node.func)
        return str(ast.unparse(node))

    def _check_unused_imports(self):
        """检查未使用的导入"""
        for imp in self.imports:
            name = imp['alias'] or imp['module'].split('.')[0]
            if name in self.unused_imports and name != '*':
                self.issues.append({
                    'type': 'unused_import',
                    'message': f"未使用的导入: {imp['module']}",
                    'line': imp['line'],
                })

    def _generate_report(self):
        """生成分析报告"""
        return {
            'filename': self.filename,
            'functions': self.functions,
            'classes': self.classes,
            'imports': self.imports,
            'complexity': self.complexity,
            'issues': self.issues,
            'stats': {
                'total_functions': len(self.functions),
                'total_classes': len(self.classes),
                'total_imports': len(self.imports),
                'avg_complexity': sum(self.complexity.values()) /
                                 max(len(self.complexity), 1),
                'max_complexity': max(self.complexity.values())
                                  if self.complexity else 0,
            }
        }
```

```python
class CodeTransformer(ast.NodeTransformer):
    """代码转换器基类"""

    def transform(self, source):
        """转换源代码"""
        tree = ast.parse(source)
        new_tree = self.visit(tree)
        ast.fix_missing_locations(new_tree)
        if hasattr(ast, 'unparse'):
            return ast.unparse(new_tree)
        return None


class ConstantFolder(ast.NodeTransformer):
    """常量折叠转换器

    在编译时计算常量表达式
    """

    def visit_BinOp(self, node):
        """折叠二元运算"""
        self.generic_visit(node)

        # 两个操作数都是常量
        if isinstance(node.left, ast.Constant) and \
           isinstance(node.right, ast.Constant):
            left = node.left.value
            right = node.right.value

            result = self._compute(node.op, left, right)
            if result is not None:
                return ast.Constant(value=result)

        return node

    def visit_UnaryOp(self, node):
        """折叠一元运算"""
        self.generic_visit(node)

        if isinstance(node.operand, ast.Constant):
            operand = node.operand.value
            if isinstance(node.op, ast.USub):
                return ast.Constant(value=-operand)
            elif isinstance(node.op, ast.UAdd):
                return ast.Constant(value=+operand)
            elif isinstance(node.op, ast.Not):
                return ast.Constant(value=not operand)
            elif isinstance(node.op, ast.Invert):
                return ast.Constant(value=~operand)

        return node

    def _compute(self, op, left, right):
        """计算常量运算"""
        ops = {
            ast.Add: lambda a, b: a + b,
            ast.Sub: lambda a, b: a - b,
            ast.Mult: lambda a, b: a * b,
            ast.Div: lambda a, b: a / b if b != 0 else None,
            ast.FloorDiv: lambda a, b: a // b if b != 0 else None,
            ast.Mod: lambda a, b: a % b if b != 0 else None,
            ast.Pow: lambda a, b: a ** b,
        }
        compute = ops.get(type(op))
        if compute:
            try:
                result = compute(left, right)
                if isinstance(result, (int, float, str, bool)):
                    return result
            except Exception:
                pass
        return None


class TypeAnnotator(ast.NodeTransformer):
    """自动添加类型注解"""

    TYPE_MAP = {
        'True': 'bool',
        'False': 'bool',
        'None': 'None',
    }

    def __init__(self):
        self.inferred_types = {}

    def visit_FunctionDef(self, node):
        """为函数添加返回类型注解"""
        # 简单推断: 如果有return语句且返回常量
        return_type = self._infer_return_type(node)
        if return_type and node.returns is None:
            node.returns = ast.parse(return_type, mode='eval').body

        self.generic_visit(node)
        return node

    def _infer_return_type(self, node):
        """推断返回类型"""
        for child in ast.walk(node):
            if isinstance(child, ast.Return) and child.value:
                if isinstance(child.value, ast.Constant):
                    vtype = type(child.value.value).__name__
                    return vtype
                elif isinstance(child.value, ast.Name):
                    return self.inferred_types.get(child.value.id, 'Any')
        return None


class DebugInserter(ast.NodeTransformer):
    """在函数入口插入调试打印"""

    def __init__(self, functions_to_debug=None):
        self.functions_to_debug = functions_to_debug or set()

    def visit_FunctionDef(self, node):
        """在函数体开头插入调试信息"""
        if self.functions_to_debug and \
           node.name not in self.functions_to_debug:
            return node

        # 创建调试打印语句
        args_str = ", ".join(
            f"{arg.arg}={{{arg.arg}!r}}"
            for arg in node.args.args
        )

        debug_code = f'print(f"调用 {node.name}({args_str})")'
        debug_stmt = ast.parse(debug_code).body[0]
        ast.fix_missing_locations(debug_stmt)

        # 插入到函数体开头
        node.body.insert(0, debug_stmt)
        return node


class PrintToLogging(ast.NodeTransformer):
    """将print语句转换为logging"""

    def visit_Call(self, node):
        """转换print调用为logging"""
        self.generic_visit(node)

        if isinstance(node.func, ast.Name) and node.func.id == 'print':
            # 将print(...)转换为logging.info(...)
            node.func.id = 'info'
            node.func = ast.Attribute(
                value=ast.Name(id='logging', ctx=ast.Load()),
                attr='info',
                ctx=ast.Load(),
            )
            ast.fix_missing_locations(node)

        return node
```

```python
"""
safe_executor.py — 安全的Python代码执行环境

基于AST分析和受限命名空间
"""

import ast
import sys
import builtins
from types import CodeType


class DangerousNodeChecker(ast.NodeVisitor):
    """检查AST中是否存在危险操作"""

    DANGEROUS_IMPORTS = {
        'os', 'sys', 'subprocess', 'socket', 'ctypes',
        'shutil', 'signal', 'multiprocessing', 'threading',
    }

    DANGEROUS_ATTRS = {
        '__import__', '__builtins__', '__class__',
        '__subclasses__', 'exec', 'eval', 'compile',
        'open', 'input',
    }

    DANGEROUS_CALLS = {
        'exec', 'eval', 'compile', '__import__',
        'open', 'exit', 'quit',
    }

    def __init__(self):
        self.issues = []

    def check(self, source):
        """检查源代码是否安全"""
        try:
            tree = ast.parse(source)
        except SyntaxError as e:
            return [{'type': 'syntax', 'message': str(e)}]

        self.issues = []
        self.visit(tree)
        return self.issues

    def visit_Import(self, node):
        """检查import语句"""
        for alias in node.names:
            root_module = alias.name.split('.')[0]
            if root_module in self.DANGEROUS_IMPORTS:
                self.issues.append({
                    'type': 'dangerous_import',
                    'message': f"禁止导入模块: {alias.name}",
                    'line': node.lineno,
                })
        self.generic_visit(node)

    def visit_ImportFrom(self, node):
        """检查from ... import语句"""
        if node.module:
            root_module = node.module.split('.')[0]
            if root_module in self.DANGEROUS_IMPORTS:
                self.issues.append({
                    'type': 'dangerous_import',
                    'message': f"禁止从模块导入: {node.module}",
                    'line': node.lineno,
                })
        self.generic_visit(node)

    def visit_Call(self, node):
        """检查函数调用"""
        if isinstance(node.func, ast.Name):
            if node.func.id in self.DANGEROUS_CALLS:
                self.issues.append({
                    'type': 'dangerous_call',
                    'message': f"禁止调用: {node.func.id}",
                    'line': node.lineno,
                })
        self.generic_visit(node)

    def visit_Attribute(self, node):
        """检查属性访问"""
        if isinstance(node.attr, str) and \
           node.attr.startswith('__') and node.attr.endswith('__'):
            self.issues.append({
                'type': 'dunder_access',
                'message': f"禁止访问特殊属性: {node.attr}",
                'line': node.lineno,
            })
        self.generic_visit(node)

    def visit_Global(self, node):
        """检查global声明"""
        self.issues.append({
            'type': 'global_decl',
            'message': f"禁止global声明: {node.names}",
            'line': node.lineno,
        })

    def visit_Assign(self, node):
        """检查赋值目标"""
        for target in node.targets:
            if isinstance(target, ast.Name):
                if target.id.startswith('__') and target.id.endswith('__'):
                    self.issues.append({
                        'type': 'dunder_assign',
                        'message': f"禁止修改特殊属性: {target.id}",
                        'line': node.lineno,
                    })
        self.generic_visit(node)


class SafeExecutor:
    """安全的Python代码执行器"""

    # 允许的内置函数
    SAFE_BUILTINS = {
        'abs': abs,
        'all': all,
        'any': any,
        'bin': bin,
        'bool': bool,
        'chr': chr,
        'dict': dict,
        'divmod': divmod,
        'enumerate': enumerate,
        'filter': filter,
        'float': float,
        'format': format,
        'frozenset': frozenset,
        'hash': hash,
        'hex': hex,
        'int': int,
        'isinstance': isinstance,
        'len': len,
        'list': list,
        'map': map,
        'max': max,
        'min': min,
        'oct': oct,
        'ord': ord,
        'pow': pow,
        'print': print,
        'range': range,
        'repr': repr,
        'reversed': reversed,
        'round': round,
        'set': set,
        'sorted': sorted,
        'str': str,
        'sum': sum,
        'tuple': tuple,
        'zip': zip,
        'True': True,
        'False': False,
        'None': None,
    }

    def __init__(self, allowed_modules=None):
        self.checker = DangerousNodeChecker()
        self.allowed_modules = allowed_modules or {
            'math', 'json', 're', 'string', 'itertools',
            'functools', 'collections', 'datetime', 'decimal',
        }

    def execute(self, source, timeout=5):
        """安全执行Python代码"""
        # 1. AST级别安全检查
        issues = self.checker.check(source)
        if issues:
            dangerous = [i for i in issues
                        if i['type'] in ('dangerous_import', 'dangerous_call')]
            if dangerous:
                raise SecurityError(
                    f"安全检查失败: {[i['message'] for i in dangerous]}")

        # 2. 编译代码
        try:
            code = compile(source, '<safe_exec>', 'exec')
        except SyntaxError as e:
            raise ExecutionError(f"语法错误: {e}")

        # 3. 构建受限命名空间
        safe_globals = {'__builtins__': self.SAFE_BUILTINS.copy()}

        # 添加允许的模块
        for mod_name in self.allowed_modules:
            try:
                mod = __import__(mod_name)
                safe_globals[mod_name] = mod
            except ImportError:
                pass

        safe_locals = {}

        # 4. 执行
        try:
            exec(code, safe_globals, safe_locals)
        except Exception as e:
            raise ExecutionError(f"执行错误: {type(e).__name__}: {e}")

        return safe_locals


class SecurityError(Exception):
    """安全检查错误"""
    pass


class ExecutionError(Exception):
    """执行错误"""
    pass
```

```python
"""
ast_visualizer.py — AST可视化工具

将AST转换为可读的树形结构或Graphviz图
"""

import ast
import sys
from collections import defaultdict


class ASTVisualizer:
    """AST可视化器"""

    def __init__(self, max_depth=None):
        self.max_depth = max_depth
        self.indent = "  "
        self.current_depth = 0

    def visualize(self, source):
        """可视化AST"""
        tree = ast.parse(source)
        return self._visit(tree, 0)

    def _visit(self, node, depth):
        """递归访问AST节点"""
        if self.max_depth and depth > self.max_depth:
            return "..."

        lines = []

        if isinstance(node, ast.AST):
            # 获取节点类型名
            node_name = type(node).__name__

            # 获取节点属性
            attrs = self._get_node_attrs(node)

            # 格式化节点信息
            if attrs:
                attr_str = ", ".join(f"{k}={v}" for k, v in attrs.items())
                lines.append(f"{self.indent * depth}{node_name}({attr_str})")
            else:
                lines.append(f"{self.indent * depth}{node_name}")

            # 递归处理子节点
            for field, value in ast.iter_fields(node):
                if isinstance(value, list):
                    if value:
                        lines.append(f"{self.indent * (depth+1)}{field}:")
                        for item in value:
                            lines.append(self._visit(item, depth + 2))
                elif isinstance(value, ast.AST):
                    lines.append(f"{self.indent * (depth+1)}{field}:")
                    lines.append(self._visit(value, depth + 2))

        elif isinstance(node, (int, float, str, bool, type(None))):
            lines.append(f"{self.indent * depth}{repr(node)}")

        return "\n".join(lines)

    def _get_node_attrs(self, node):
        """获取节点的显示属性"""
        attrs = {}

        # 常见节点的关键属性
        if isinstance(node, ast.Name):
            attrs['id'] = repr(node.id)
            attrs['ctx'] = type(node.ctx).__name__
        elif isinstance(node, ast.Constant):
            attrs['value'] = repr(node.value)
        elif isinstance(node, ast.FunctionDef):
            attrs['name'] = repr(node.name)
        elif isinstance(node, ast.ClassDef):
            attrs['name'] = repr(node.name)
        elif isinstance(node, ast.arg):
            attrs['arg'] = repr(node.arg)
        elif isinstance(node, ast.BinOp):
            attrs['op'] = type(node.op).__name__
        elif isinstance(node, ast.Compare):
            attrs['ops'] = [type(op).__name__ for op in node.ops]
        elif isinstance(node, ast.Call):
            if isinstance(node.func, ast.Name):
                attrs['func'] = repr(node.func.id)
        elif isinstance(node, ast.Attribute):
            attrs['attr'] = repr(node.attr)
        elif isinstance(node, ast.Import):
            attrs['names'] = [alias.name for alias in node.names]
        elif isinstance(node, ast.ImportFrom):
            attrs['module'] = repr(node.module)

        return attrs

    def to_graphviz(self, source):
        """生成Graphviz DOT格式"""
        tree = ast.parse(source)
        lines = ['digraph AST {']
        lines.append('  node [shape=box, fontname="monospace"];')
        self._graphviz_visit(tree, lines, [0])
        lines.append('}')
        return '\n'.join(lines)

    def _graphviz_visit(self, node, lines, counter):
        """递归生成Graphviz节点"""
        node_id = counter[0]
        counter[0] += 1

        node_name = type(node).__name__
        attrs = self._get_node_attrs(node)
        label = node_name
        if attrs:
            label += '\\n' + '\\n'.join(f'{k}={v}' for k, v in attrs.items())

        lines.append(f'  node{node_id} [label="{label}"];')

        for field, value in ast.iter_fields(node):
            if isinstance(value, list):
                for item in value:
                    child_id = counter[0]
                    self._graphviz_visit(item, lines, counter)
                    lines.append(f'  node{node_id} -> node{child_id} '
                                f'[label="{field}"];')
            elif isinstance(value, ast.AST):
                child_id = counter[0]
                self._graphviz_visit(value, lines, counter)
                lines.append(f'  node{node_id} -> node{child_id} '
                            f'[label="{field}"];')

        return node_id


class ASTComparator:
    """AST比较器 — 比较两段代码的AST差异"""

    def compare(self, source1, source2):
        """比较两段代码的AST"""
        tree1 = ast.parse(source1)
        tree2 = ast.parse(source2)

        diff = self._diff(tree1, tree2)
        return diff

    def _diff(self, node1, node2, path=""):
        """递归比较AST节点"""
        differences = []

        # 类型不同
        if type(node1) != type(node2):
            differences.append({
                'path': path,
                'type': 'node_type_mismatch',
                'left': type(node1).__name__,
                'right': type(node2).__name__,
            })
            return differences

        # 比较字段
        for field1, field2 in zip(ast.iter_fields(node1),
                                   ast.iter_fields(node2)):
            fname1, fval1 = field1
            fname2, fval2 = field2
            current_path = f"{path}.{fname1}"

            if type(fval1) != type(fval2):
                differences.append({
                    'path': current_path,
                    'type': 'field_type_mismatch',
                    'left': type(fval1).__name__,
                    'right': type(fval2).__name__,
                })
            elif isinstance(fval1, list):
                if len(fval1) != len(fval2):
                    differences.append({
                        'path': current_path,
                        'type': 'list_length_mismatch',
                        'left': len(fval1),
                        'right': len(fval2),
                    })
                for i, (item1, item2) in enumerate(zip(fval1, fval2)):
                    if isinstance(item1, ast.AST) and isinstance(item2, ast.AST):
                        differences.extend(
                            self._diff(item1, item2,
                                       f"{current_path}[{i}]")
                        )
                    elif item1 != item2:
                        differences.append({
                            'path': f"{current_path}[{i}]",
                            'type': 'value_mismatch',
                            'left': repr(item1),
                            'right': repr(item2),
                        })
            elif isinstance(fval1, ast.AST) and isinstance(fval2, ast.AST):
                differences.extend(
                    self._diff(fval1, fval2, current_path)
                )
            elif fval1 != fval2:
                differences.append({
                    'path': current_path,
                    'type': 'value_mismatch',
                    'left': repr(fval1),
                    'right': repr(fval2),
                })

        return differences
```

```python
"""
compile_pipeline.py — 完整模拟Python编译流水线

从源代码到字节码的完整过程:
源码 → Token流 → AST → 符号表 → Code Object
"""

import ast
import dis
import sys
import symbol
import tokenize
import io
from collections import defaultdict


class CompilePipeline:
    """编译流水线模拟器"""

    def __init__(self):
        self.stages = {}

    def compile(self, source, filename='<string>', mode='exec'):
        """完整编译流程"""

        print(f"\n{'='*70}")
        print(f"  编译流水线: {filename}")
        print(f"{'='*70}")

        # 阶段1: 词法分析
        print(f"\n--- 阶段1: 词法分析 (Tokenization) ---")
        tokens = self._tokenize(source)
        self.stages['tokens'] = tokens

        # 阶段2: 语法分析
        print(f"\n--- 阶段2: 语法分析 (Parsing → AST) ---")
        tree = self._parse(source, filename, mode)
        self.stages['ast'] = tree

        # 阶段3: AST优化
        print(f"\n--- 阶段3: AST优化 ---")
        optimized = self._optimize_ast(tree)
        self.stages['optimized_ast'] = optimized

        # 阶段4: 符号表分析
        print(f"\n--- 阶段4: 符号表分析 ---")
        symtab = self._analyze_symbols(source)
        self.stages['symtab'] = symtab

        # 阶段5: 字节码生成
        print(f"\n--- 阶段5: 字节码生成 ---")
        code = self._generate_bytecode(source, filename, mode)
        self.stages['code'] = code

        # 阶段6: 执行
        print(f"\n--- 阶段6: 执行 ---")
        result = self._execute(code, mode)
        self.stages['result'] = result

        return self.stages

    def _tokenize(self, source):
        """词法分析"""
        tokens = []
        try:
            for tok in tokenize.generate_tokens(
                io.StringIO(source).readline
            ):
                token_info = {
                    'type': tokenize.tok_name.get(tok.type, str(tok.type)),
                    'string': tok.string,
                    'start': tok.start,
                    'end': tok.end,
                    'line': tok.line.strip() if tok.line else '',
                }
                tokens.append(token_info)
                print(f"  {token_info['type']:<12} "
                      f"{token_info['string']:<20} "
                      f"line={token_info['start'][0]}")
        except tokenize.TokenError as e:
            print(f"  词法错误: {e}")

        print(f"\n  总Token数: {len(tokens)}")
        return tokens

    def _parse(self, source, filename, mode):
        """语法分析"""
        tree = ast.parse(source, filename, mode)
        print(f"  AST根节点: {type(tree).__name__}")
        print(f"  语句数: {len(tree.body)}")

        for i, stmt in enumerate(tree.body):
            print(f"   [{i}] {type(stmt).__name__}")

        # 显示AST结构
        print(f"\n  AST结构:")
        for line in ast.dump(tree, indent=2).split('\n')[:30]:
            print(f"    {line}")
        if len(ast.dump(tree, indent=2).split('\n')) > 30:
            print(f"    ... (更多节点)")

        return tree

    def _optimize_ast(self, tree):
        """AST优化 — 常量折叠"""
        print("  执行常量折叠...")

        class ConstantFolder(ast.NodeTransformer):
            def visit_BinOp(self, node):
                self.generic_visit(node)
                if isinstance(node.left, ast.Constant) and \
                   isinstance(node.right, ast.Constant):
                    ops = {
                        ast.Add: lambda a, b: a + b,
                        ast.Sub: lambda a, b: a - b,
                        ast.Mult: lambda a, b: a * b,
                    }
                    compute = ops.get(type(node.op))
                    if compute:
                        try:
                            result = compute(
                                node.left.value, node.right.value)
                            return ast.Constant(value=result)
                        except Exception:
                            pass
                return node

        folder = ConstantFolder()
        optimized = folder.visit(tree)
        ast.fix_missing_locations(optimized)

        # 统计优化次数
        original_count = sum(1 for _ in ast.walk(tree)
                            if isinstance(_, ast.BinOp))
        optimized_count = sum(1 for _ in ast.walk(optimized)
                             if isinstance(_, ast.BinOp))
        folded = original_count - optimized_count
        print(f"  常量折叠: {folded}个表达式")

        return optimized

    def _analyze_symbols(self, source):
        """符号表分析"""
        import symtable

        try:
            symtab = symtable.symtable(source, '<string>', 'exec')
            self._print_symtable(symtab, 0)
            return symtab
        except Exception as e:
            print(f"  符号表分析失败: {e}")
            return None

    def _print_symtable(self, st, depth):
        """打印符号表"""
        prefix = "  " * (depth + 1)
        scope_type = st.get_type()
        scope_name = st.get_name()
        print(f"{prefix}作用域: {scope_name} ({scope_type})")

        symbols = st.get_symbols()
        for sym in symbols:
            flags = []
            if sym.is_local(): flags.append("local")
            if sym.is_global(): flags.append("global")
            if sym.is_free(): flags.append("free")
            if sym.is_parameter(): flags.append("param")
            if sym.is_assigned(): flags.append("assigned")
            if sym.is_referenced(): flags.append("referenced")
            print(f"{prefix}  {sym.get_name()}: "
                  f"{', '.join(flags) if flags else 'none'}")

        for child in st.get_children():
            self._print_symtable(child, depth + 1)

    def _generate_bytecode(self, source, filename, mode):
        """字节码生成"""
        code = compile(source, filename, mode)

        print(f"  Code Object: {code.co_name}")
        print(f"  参数数量: {code.co_argcount}")
        print(f"  局部变量: {code.co_varnames}")
        print(f"  栈大小: {code.co_stacksize}")
        print(f"  字节码长度: {len(code.co_code)} 字节")
        print(f"  常量表: {code.co_consts}")
        print(f"  名称表: {code.co_names}")

        print(f"\n  字节码反汇编:")
        dis.dis(code)

        # 递归显示嵌套Code Object
        for const in code.co_consts:
            if isinstance(const, type(code)):
                print(f"\n  --- 嵌套Code Object: {const.co_name} ---")
                dis.dis(const)

        return code

    def _execute(self, code, mode):
        """执行字节码"""
        namespace = {}
        try:
            if mode == 'exec':
                exec(code, namespace)
            elif mode == 'eval':
                result = eval(code, namespace)
                print(f"  结果: {result}")
                return result
            elif mode == 'single':
                exec(code, namespace)

            # 显示命名空间
            public_vars = {k: v for k, v in namespace.items()
                         if not k.startswith('_')}
            if public_vars:
                print(f"  命名空间:")
                for k, v in public_vars.items():
                    v_repr = repr(v)
                    if len(v_repr) > 80:
                        v_repr = v_repr[:77] + "..."
                    print(f"    {k} = {v_repr}")

        except Exception as e:
            print(f"  执行错误: {type(e).__name__}: {e}")


# 使用示例
if __name__ == "__main__":
    pipeline = CompilePipeline()

    # 示例1: 简单计算
    pipeline.compile("""
x = 1 + 2
y = x * 3
print(f"x = {x}, y = {y}")
""")

    # 示例2: 函数定义
    pipeline.compile("""
def greet(name):
    message = f"Hello, {name}!"
    return message

result = greet("Python")
print(result)
""")
```

```python
            0x200: "ITERABLE_COROUTINE",
            0x400: "ASYNC_GENERATOR",

                  if flags & bit]
```

### 42.5.2 exec/eval安全陷阱

```python
"""
陷阱：exec和eval的安全风险

永远不要直接执行不受信任的代码!
"""

import ast

# 陷阱1: eval可以执行任意Python代码
# 即使限制了命名空间也不安全!

unsafe_code = "__import__('os').system('rm -rf /')"

# 错误: 以为限制命名空间就安全
try:
    # 即使__builtins__为空，也有方法逃逸
    result = eval(unsafe_code, {"__builtins__": {}})
except Exception as e:
    print(f"安全限制生效: {e}")

# 陷阱2: 通过类型系统逃逸
escape_code = """
().__class__.__bases__[0].__subclasses__()
"""

# 更安全的替代方案:

# 方案1: ast.literal_eval — 只评估字面量
safe_result = ast.literal_eval("{'a': 1, 'b': [2, 3]}")
print(f"literal_eval结果: {safe_result}")

# 方案2: 使用AST白名单
class SafeEval(ast.NodeVisitor):
    """安全的表达式评估器"""

    ALLOWED_NODES = {
        ast.Expression, ast.Constant, ast.Num, ast.Str,
        ast.List, ast.Tuple, ast.Set, ast.Dict,
        ast.UnaryOp, ast.UAdd, ast.USub, ast.Not,
        ast.BinOp, ast.Add, ast.Sub, ast.Mult,
        ast.Div, ast.FloorDiv, ast.Mod, ast.Pow,
        ast.Compare, ast.Eq, ast.NotEq, ast.Lt,
        ast.LtE, ast.Gt, ast.GtE, ast.Is, ast.IsNot,
        ast.BoolOp, ast.And, ast.Or,
        ast.Name, ast.Load, ast.NameConstant,
    }

    def safe_eval(self, expr, variables=None):
        """安全评估表达式"""
        tree = ast.parse(expr, mode='eval')
        self.visit(tree)  # 检查节点安全性
        variables = variables or {}
        # 使用受限命名空间
        return eval(compile(tree, '<safe>', 'eval'),
                   {"__builtins__": {}}, variables)

    def visit(self, node):
        """检查节点类型"""
        if type(node) not in self.ALLOWED_NODES:
            raise ValueError(
                f"不允许的节点类型: {type(node).__name__}")
        super().visit(node)

evaluator = SafeEval()
result = evaluator.safe_eval("1 + 2 * 3", {"x": 10})
print(f"安全评估结果: {result}")

# 方案3: 使用子进程或沙箱(如Docker)
# 对于真正不受信任的代码，使用容器隔离
```

```python
"""
陷阱：修改源代码后.pyc缓存可能不更新

在开发中修改了.py文件但行为没变，
可能是__pycache__中的缓存问题
"""

import importlib
import os
import sys
import tempfile
import time

# 创建临时模块
temp_dir = tempfile.mkdtemp()
module_path = os.path.join(temp_dir, 'cached_module.py')

# 写入初始版本
with open(module_path, 'w') as f:
    f.write('VERSION = "1.0"\n\ndef get_version():\n    return VERSION\n')

sys.path.insert(0, temp_dir)

# 第一次导入 — 生成.pyc
import cached_module
print(f"初始版本: {cached_module.get_version()}")

# 修改源代码
time.sleep(0.1)  # 确保时间戳不同
with open(module_path, 'w') as f:
    f.write('VERSION = "2.0"\n\ndef get_version():\n    return VERSION\n')

# 陷阱: 直接import不会重新加载!
# Python缓存了模块对象，不会重新导入
print(f"修改后导入(未重载): {cached_module.get_version()}")  # 仍然是"1.0"

# 正确做法: 使用importlib.reload
importlib.reload(cached_module)
print(f"重载后: {cached_module.get_version()}")  # "2.0"

# 清理
del sys.modules['cached_module']
sys.path.remove(temp_dir)
os.remove(module_path)
# 删除__pycache__
pycache = os.path.join(temp_dir, '__pycache__')
if os.path.exists(pycache):
    import shutil
    shutil.rmtree(pycache)
os.rmdir(temp_dir)

# 最佳实践:
# 1. 开发时使用环境变量 PYTHONDONTWRITEBYTECODE=1 禁用.pyc
# 2. 或使用 python -B 运行脚本
# 3. 修改模块后使用 importlib.reload()
```

### 42.5.4 符号表分析陷阱

```python
"""
陷阱：误解变量作用域和绑定规则

Python的变量作用域规则容易混淆
"""

# 陷阱1: 循环变量在列表推导式中的作用域
# Python 3: 列表推导式有自己的作用域
result = [x for x in range(5)]
try:
    print(x)  # NameError! x不外泄
except NameError:
    print("列表推导式中的x不外泄(Python 3)")

# 但在Python 2中，x会外泄!
# 列表推导式在Python 2中不是闭包

# 陷阱2: 函数内的变量赋值使变量变为局部变量
counter = 0

def increment_wrong():
    counter += 1  # UnboundLocalError!
    # Python看到赋值，将counter视为局部变量
    # 但在赋值前就使用了

def increment_right():
    global counter
    counter += 1  # 正确: 显式声明global

def closure_wrong():
    counter = 0
    def inner():
        counter += 1  # UnboundLocalError!
        # Python看到赋值，将counter视为局部变量
    return inner

def closure_right():
    counter = 0
    def inner():
        nonlocal counter
        counter += 1  # 正确: 显式声明nonlocal
        return counter
    return inner

# 陷阱3: 类作用域与嵌套作用域
class MyClass:
    x = 10

    def method(self):
        # 这里不能直接引用x
        # 类作用域不参与嵌套作用域!
        try:
            print(x)  # NameError!
        except NameError:
            pass
        print(self.x)  # 正确: 通过self访问

    # 但列表推导式可以访问类作用域
    items = [x for x in range(3)]  # 正确: x来自类作用域

# 陷阱4: del和UnboundLocalError
def tricky_delete():
    x = 1
    def inner():
        try:
            print(x)  # 在del之前可以读取
        except UnboundLocalError:
            pass
    del x  # 删除后变量不再绑定
    # 但inner闭包已经捕获了x的值

# 最佳实践:
# 1. 理解Python的LEGB规则(Local, Enclosing, Global, Built-in)
# 2. 赋值(包括+=)使变量变为局部变量
# 3. 使用global/nonlocal显式声明非局部变量
# 4. 类作用域不参与嵌套作用域
# 5. 列表推导式有独立作用域(Python 3+)
```
