### 2.2 列表扩容机制源码分析

```c
// Objects/listobject.c — list 扩容算法
static int
list_resize(PyListObject *self, Py_ssize_t newsize)
{
    Py_ssize_t oldsize = self->ob_size;
    size_t new_allocated;

    /* 增长策略：
     * new_allocated = (newsize + (newsize >> 3) + 6) & ~3
     * 等价于: newsize * 1.125 + 6, 向下对齐到4的倍数
     *
     * 这个策略保证：
     * 1. 平摊 O(1) 的 append
     * 2. 内存增长约12.5%，避免过多浪费
     * 3. 小列表有最小增长量(+6)
     */
    if (newsize > oldsize) {
        new_allocated = ((size_t)newsize + (newsize >> 3) + 6) & ~(size_t)3;
    } else {
        // 缩容策略：只有空闲超过一半时才缩容
        new_allocated = newsize + (newsize >> 3) + 6;
        if (new_allocated > oldsize) {
            new_allocated = oldsize;
        }
    }

    // realloc 扩容
    items = PyMem_Realloc(self->ob_item, new_allocated * sizeof(PyObject*));
}
```

### 2.3 字典实现原理（Python 3.6+ 紧凑字典）

```
Python 3.6+ 字典内部结构（紧凑有序字典）:

entries 数组（存储实际键值对）:
index:  0        1        2        3
      ┌────────┬────────┬────────┬────────┐
      │ hash0  │ hash1  │ hash2  │ hash3  │
      │ key0   │ key1   │ key2   │ key3   │
      │ value0 │ value1 │ value2 │ value3 │
      └────────┴────────┴────────┴────────┘

indices 数组（哈希表，存储 entries 索引）:
      ┌───┬───┬───┬───┬───┬───┬───┬───┐
      │ - │ 1 │ - │ 0 │ - │ - │ 3 │ 2 │   (- 表示空位)
      └───┴───┴───┴───┴───┴───┴───┴───┘

查找过程:
1. hash(key) → 计算哈希值
2. hash % table_size → 确定起始位置
3. 在 indices 中探测 → 得到 entries 索引
4. 比较 entries[index] 的 key
```

**性能影响**：
- Python 3.6 之前：字典内存 = 8/3 * n * 3 * 8 bytes = 64n bytes（n为表大小）
- Python 3.6+：字典内存 ≈ n * 3 * 8 + n * 1 = 25n bytes（约节省60%）
- 查找仍是 O(1)，但缓存更友好

### 2.4 局部变量优化原理

```c
// Python 字节码中局部变量的访问方式:
// LOAD_FAST  (局部变量) — 数组索引访问, O(1), 1条指令
// LOAD_GLOBAL(全局变量) — 字典查找, O(1)但常数更大, 1条指令
// LOAD_NAME  (自由变量) — 多层查找, 最慢, 1条指令但查找路径长

// CPython/ceval.c — LOAD_FAST 实现（简化）
case TARGET(LOAD_FAST): {
    PyObject *value = GETLOCAL(oparg);  // 数组索引，直接访问
    if (value == NULL) {
        // 未初始化局部变量错误
    }
    Py_INCREF(value);
    PUSH(value);
    DISPATCH();  // 无条件跳转到下一条指令
}

// LOAD_GLOBAL 实现（简化）
case TARGET(LOAD_GLOBAL): {
    PyObject *name = GETITEM(names, oparg);  // 获取变量名
    PyObject *v = PyDict_GetItem(f->f_globals, name);  // 全局字典查找
    if (v == NULL) {
        v = PyDict_GetItem(f->f_builtins, name);  // 内置字典查找
    }
    Py_INCREF(v);
    PUSH(v);
    DISPATCH();
}
```

**关键差异**：局部变量通过数组索引访问（1次内存访问），全局变量需要字典查找（2次哈希+比较），速度差距约 2-5 倍。

### 2.5 lru_cache 实现原理

```python
# functools.lru_cache 内部结构（简化）
# 使用双向链表 + 字典实现 O(1) 的 LRU 操作

class _LRUCache:
    """
    内部结构:
    - cache: dict[key -> link]      # O(1) 查找
    - root: 双向链表头              # O(1) 淘汰
    - link = (key, result, prev, next)
    """

    def __getitem__(self, key):
        link = self.cache[key]           # O(1) 字典查找
        # 移动到链表尾部（最近使用）
        link_prev = link.prev
        link_next = link.next
        link_prev.next = link_next
        link_next.prev = link_prev
        root = self.root
        link.prev = root.prev
        link.next = root
        root.prev.next = link
        root.prev = link
        return link.result

    def __setitem__(self, key, result):
        if len(self.cache) >= self.maxsize:
            # 淘汰链表头部（最久未使用）
            root = self.root
            link = root.next       # O(1) 获取最久未使用的
            del self.cache[link.key]
            # 复用 link 节点（避免分配）
            link.key = key
            link.result = result
        else:
            # 创建新节点
            link = (key, result)
            self.cache[key] = link
```

"""
常见数据结构的性能对比实验
通过实际测量验证理论复杂度
"""
import time
import random
from collections import deque, defaultdict, Counter


def benchmark(func, *args, repeat=3, number=1):
    """通用基准测试框架"""
    times = []
    for _ in range(repeat):
        start = time.perf_counter()
        for _ in range(number):
            func(*args)
        elapsed = time.perf_counter() - start
        times.append(elapsed)
    return min(times)


def test_list_vs_set_lookup():
    """列表 vs 集合 查找性能"""
    print("=== 列表 vs 集合 查找性能 ===\n")

    for size in [100, 1000, 10000, 100000]:
        data = list(range(size))
        data_set = set(data)
        target = size - 1  # 查找最后一个元素（最坏情况for list）

        # list: O(n)
        list_time = benchmark(lambda: target in data, number=100)

        # set: O(1)
        set_time = benchmark(lambda: target in data_set, number=100)

        speedup = list_time / set_time if set_time > 0 else float('inf')
        print(f"  size={size:>6}: list={list_time*1000:.3f}ms, "
              f"set={set_time*1000:.3f}ms, "
              f"set快{speedup:.0f}倍")


def test_list_vs_deque():
    """列表 vs 双端队列 头部操作性能"""
    print("\n\n=== 列表 vs 双端队列 头部操作 ===\n")

    for size in [100, 1000, 10000]:
        lst = list(range(size))
        dq = deque(range(size))

        # 头部插入: list O(n), deque O(1)
        list_insert = benchmark(lambda: lst.insert(0, -1), number=100)
        # 恢复
        if lst[0] == -1:
            lst.pop(0)

        deque_append = benchmark(lambda: dq.appendleft(-1), number=100)
        if dq[0] == -1:
            dq.popleft()

        print(f"  size={size:>5}: "
              f"list.insert(0)={list_insert*1000:.3f}ms, "
              f"deque.appendleft={deque_append*1000:.3f}ms, "
              f"deque快{list_insert/deque_append:.0f}倍")


def test_dict_vs_defaultdict():
    """dict vs defaultdict 性能"""
    print("\n\n=== dict vs defaultdict 分组性能 ===\n")

    data = [random.randint(0, 99) for _ in range(100000)]

    # 方法1: 普通 dict + if判断
    def group_dict():
        groups = {}
        for item in data:
            if item not in groups:
                groups[item] = []
            groups[item].append(item)
        return groups

    # 方法2: defaultdict
    def group_defaultdict():
        groups = defaultdict(list)
        for item in data:
            groups[item].append(item)
        return groups

    # 方法3: setdefault
    def group_setdefault():
        groups = {}
        for item in data:
            groups.setdefault(item, []).append(item)
        return groups

    t1 = benchmark(group_dict, number=10)
    t2 = benchmark(group_defaultdict, number=10)
    t3 = benchmark(group_setdefault, number=10)

    print(f"  dict + if:      {t1*1000:.3f}ms")
    print(f"  defaultdict:    {t2*1000:.3f}ms  ({t1/t2:.1f}x faster)")
    print(f"  setdefault:     {t3*1000:.3f}ms  ({t1/t3:.1f}x faster)")


def test_string_concat():
    """字符串拼接方法对比"""
    print("\n\n=== 字符串拼接方法对比 ===\n")

    words = [str(i) for i in range(10000)]

    # 方法1: += 拼接（最慢）
    def concat_plus():
        result = ""
        for word in words:
            result += word
        return result

    # 方法2: join（推荐）
    def concat_join():
        return "".join(words)

    # 方法3: 列表推导 + join
    def concat_join_comp():
        return "".join(str(i) for i in range(10000))

    # 方法4: io.StringIO
    def concat_stringio():
        import io
        buf = io.StringIO()
        for word in words:
            buf.write(word)
        return buf.getvalue()

    # 方法5: f-string 拼接（不适用大量拼接场景，但对比）
    def concat_fstring():
        return f"{''.join(words)}"

    t1 = benchmark(concat_plus, number=10)
    t2 = benchmark(concat_join, number=10)
    t3 = benchmark(concat_join_comp, number=10)
    t4 = benchmark(concat_stringio, number=10)

    print(f"  += 拼接:        {t1*1000:.3f}ms  (baseline)")
    print(f"  join:           {t2*1000:.3f}ms  ({t1/t2:.0f}x faster)")
    print(f"  生成器+join:    {t3*1000:.3f}ms  ({t1/t3:.0f}x faster)")
    print(f"  StringIO:       {t4*1000:.3f}ms  ({t1/t4:.0f}x faster)")


if __name__ == '__main__':

# ===== 2. 手写 LRU 缓存（理解原理）=====

class LRUCache:
    """
    手写 LRU 缓存，展示底层原理
    使用 OrderedDict 实现 O(1) 操作
    """

    def __init__(self, maxsize=128):
        self.maxsize = maxsize
        self._cache = OrderedDict()

    def get(self, key):
        if key in self._cache:
            # 移动到末尾（最近使用）
            self._cache.move_to_end(key)
            return self._cache[key]
        return None

    def put(self, key, value):
        if key in self._cache:
            self._cache.move_to_end(key)
        self._cache[key] = value
        # 超出容量时淘汰最久未使用的（链表头部）
        if len(self._cache) > self.maxsize:
            self._cache.popitem(last=False)

    def __contains__(self, key):
        if key in self._cache:
            self._cache.move_to_end(key)
            return True
        return False

    def __repr__(self):
        return f"LRUCache({len(self._cache)}/{self.maxsize})"


# ===== 3. 带 TTL 的缓存 =====

class TTLCache:
    """带过期时间的缓存"""

    def __init__(self, maxsize=128, ttl=60.0):
        self.maxsize = maxsize
        self.ttl = ttl
        self._cache = OrderedDict()
        self._timestamps = {}

    def get(self, key):
        if key in self._cache:
            if time.time() - self._timestamps[key] > self.ttl:
                # 已过期
                del self._cache[key]
                del self._timestamps[key]
                return None
            self._cache.move_to_end(key)
            return self._cache[key]
        return None

    def put(self, key, value):
        if key in self._cache:
            self._cache.move_to_end(key)
        self._cache[key] = value
        self._timestamps[key] = time.time()
        if len(self._cache) > self.maxsize:
            oldest = next(iter(self._cache))
            self._cache.popitem(last=False)
            del self._timestamps[oldest]

    def cleanup(self):
        """清理所有过期条目"""
        now = time.time()
        expired = [k for k, t in self._timestamps.items()
                   if now - t > self.ttl]
        for k in expired:
            del self._cache[k]
            del self._timestamps[k]


def ttl_cache_decorator(maxsize=128, ttl=60.0):
    """带 TTL 的缓存装饰器"""
    cache = TTLCache(maxsize=maxsize, ttl=ttl)

    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            key = (args, tuple(sorted(kwargs.items())))
            result = cache.get(key)
            if result is not None:
                return result
            result = func(*args, **kwargs)
            cache.put(key, result)
            return result
        return wrapper
    return decorator

"""
局部变量优化与字节码分析
展示如何通过局部化提升性能
"""
import dis
import time


def benchmark(func, *args, number=100000):
    """简单基准测试"""
    start = time.perf_counter()
    for _ in range(number):
        func(*args)
    return time.perf_counter() - start


# ===== 1. 全局变量 vs 局部变量 =====

# 全局变量查找
GLOBAL_VALUE = 42

def use_global(n):
    """每次循环都查找全局变量"""
    result = 0
    for i in range(n):
        result += GLOBAL_VALUE
    return result

def use_local(n):
    """将全局变量缓存到局部变量"""
    local_value = GLOBAL_VALUE  # 一次全局查找
    result = 0
    for i in range(n):
        result += local_value  # 后续都是局部查找
    return result


# ===== 2. 方法查找局部化 =====

def use_method_global(data):
    """每次循环都查找 list.append 方法"""
    result = []
    for item in data:
        result.append(item * 2)
    return result

def use_method_local(data):
    """将方法缓存到局部变量"""
    result = []
    append = result.append  # 一次方法查找

"""
算法选择与数据结构优化的实战案例
通过具体场景展示性能差异

import time
import random
from collections import deque, defaultdict, Counter


def benchmark(func, *args, number=10):
    start = time.perf_counter()
    for _ in range(number):
        func(*args)
    return (time.perf_counter() - start) / number


# ===== 案例1: 查找问题 — list vs set vs dict =====

def case_lookup():
    """查找性能：选择正确的数据结构"""
    print("=== 案例1: 查找性能 ===\n")

    size = 100000
    data = list(range(size))
    data_set = set(data)
    data_dict = {k: k for k in data}

    # 查找存在的元素
    targets = random.sample(data, 1000)

    def list_lookup():
        for t in targets:
            _ = t in data  # O(n)

    def set_lookup():
        for t in targets:
            _ = t in data_set  # O(1)

    def dict_lookup():
        for t in targets:
            _ = t in data_dict  # O(1)

    t1 = benchmark(list_lookup, number=1)
    t2 = benchmark(set_lookup, number=100)
    t3 = benchmark(dict_lookup, number=100)

    print(f"  list:   {t1*1000:.2f}ms (O(n))")
    print(f"  set:    {t2*1000:.2f}ms (O(1), 快 {t1/t2:.0f}x)")
    print(f"  dict:   {t3*1000:.2f}ms (O(1), 快 {t1/t3:.0f}x)")

def case_frequency():
    """频率统计：手动dict vs Counter"""
    print("\n=== 案例2: 频率统计 ===\n")

    data = [random.randint(0, 99) for _ in range(1000000)]

    def manual_count():
        counts = {}
        for item in data:
            counts[item] = counts.get(item, 0) + 1
        return counts

    def defaultdict_count():
        counts = defaultdict(int)
        for item in data:
            counts[item] += 1
        return dict(counts)

    def counter_count():
        return Counter(data)

    t1 = benchmark(manual_count, number=10)
    t2 = benchmark(defaultdict_count, number=10)
    t3 = benchmark(counter_count, number=10)

    print(f"  dict.get:      {t1*1000:.2f}ms")
    print(f"  defaultdict:   {t2*1000:.2f}ms (快 {t1/t2:.1f}x)")
    print(f"  Counter:       {t3*1000:.2f}ms (快 {t1/t3:.1f}x)")

def case_dedupe():
    """去重：保持顺序 vs 不保持顺序"""
    print("\n=== 案例3: 去重 ===\n")

    data = [random.randint(0, 999) for _ in range(100000)]

    # 方法1: set去重（不保持顺序）
    def dedupe_set():
        return list(set(data))

    # 方法2: dict去重（Python 3.7+ 保持顺序）
    def dedupe_dict():
        return list(dict.fromkeys(data))

    # 方法3: 生成器去重（保持顺序，内存友好）
    def dedupe_generator():
        seen = set()
        for item in data:
            if item not in seen:
                seen.add(item)
                yield item
        return list(dedupe_generator())

    t1 = benchmark(dedupe_set, number=10)
    t2 = benchmark(dedupe_dict, number=10)
    t3 = benchmark(dedupe_generator, number=10)

    print(f"  set去重:        {t1*1000:.2f}ms (不保持顺序)")
    print(f"  dict去重:       {t2*1000:.2f}ms (保持顺序)")
    print(f"  generator去重:  {t3*1000:.2f}ms (保持顺序+内存友好)")

def case_sorting():

    print("\n=== 案例4: 排序优化 ===\n")

    data = [(random.randint(0, 1000), random.random()) for _ in range(100000)]

    # 方法1: lambda排序
    def sort_lambda():
        return sorted(data, key=lambda x: x[0])

    # 方法2: operator.itemgetter
    from operator import itemgetter
    getter = itemgetter(0)

    def sort_itemgetter():
        return sorted(data, key=getter)

    # 方法3: 列表原地排序（如果不需要保留原列表）
    def sort_inplace():
        lst = data[:]
        lst.sort(key=getter)
        return lst

    t1 = benchmark(sort_lambda, number=10)
    t2 = benchmark(sort_itemgetter, number=10)
    t3 = benchmark(sort_inplace, number=10)

    print(f"  lambda排序:       {t1*1000:.2f}ms")

def case_generator():

    print("\n=== 案例5: 生成器 vs 列表 ===\n")

    import sys

    def list_comprehension():
        return [x ** 2 for x in range(1000000)]

    # 方法2: 生成器表达式
    def generator_expression():

    def generator_function():

## 5. 常见陷阱与最佳实践

### 5.1 陷阱：过早优化

```python
# 错误：在没有测量的情况下优化
def process_items_slow(items):

    result = []
    for item in items:
        if item.id not in seen:
            seen.add(item.id)
            result.append(item)
    return result

# 正确：先测量，确认瓶颈
# 1. 使用 cProfile 确认 process_items 是否真的是瓶颈
# 2. 只优化确认的瓶颈
# 3. 对比优化前后的测量结果

# 真正的瓶颈可能完全不在你猜的地方
import cProfile
cProfile.run('process_items_slow(large_dataset)')
```

### 5.2 陷阱：lru_cache 的内存泄漏

```python
# 错误：无限缓存导致内存增长
from functools import lru_cache

@lru_cache(maxsize=None)  # 危险！缓存永不释放
def get_user_data(user_id):
    return fetch_from_database(user_id)

# 在长期运行的服务中，缓存会无限增长！

# 正确：设置合理的缓存大小
@lru_cache(maxsize=1024)  # 限制缓存大小
def get_user_data(user_id):
    return fetch_from_database(user_id)

# 或者使用带TTL的缓存
# 对于参数变化多的函数，缓存命中率低时不要用 lru_cache
```

### 5.3 陷阱：列表推导的副作用

```python
# 错误：用列表推导执行副作用
_ = [print(item) for item in range(10)]  # 创建了无用的列表

# 正确：用普通循环执行副作用
for item in range(10):
    print(item)

# 错误：列表推导中使用昂贵的操作
result = [expensive_func(x) for x in large_data if expensive_func(x) > 0]
# expensive_func 被调用了两次！

# 正确：使用生成器表达式 + walrus operator (Python 3.8+)
result = [y for x in large_data if (y := expensive_func(x)) > 0]
```

### 5.5 最佳实践总结

| 实践 | 说明 |
|------|------|
| **先测量后优化** | 使用 cProfile/line_profiler 定位瓶颈 |
| **选择正确的数据结构** | 查找用 set/dict，队列用 deque，排序用 list |
| **使用内置函数** | sum/max/min/sorted 比手写循环快 5-50 倍 |
| **局部变量优化** | 热路径中将全局/属性/方法缓存到局部变量 |
| **使用 join 拼接** | 大量字符串拼接永远用 `"".join(parts)` |
| **生成器节省内存** | 不需要随机访问时用生成器替代列表 |
| **lru_cache 缓存** | 纯函数、递归、重复计算用缓存 |
| **避免不必要的列表创建** | 用生成器表达式替代列表推导 |
| **`__slots__` 减少内存** | 大量实例时使用 `__slots__` |
| **算法优先** | O(n) 到 O(n log n) 的提升远大于常数优化 |
