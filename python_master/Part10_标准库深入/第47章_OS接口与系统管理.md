### 47.1.1 os 模块

        print(os.path.join(root, f))

print(os.path.exists('file.txt'))         # True/False
print(os.path.isfile('file.txt'))         # True/False
print(os.path.isdir('dir'))              # True/False

import subprocess

# 推荐方式：subprocess.run()
result = subprocess.run(
    ['ls', '-la'],
    capture_output=True,    # 捕获 stdout 和 stderr

    capture_output=True,
    text=True,
)

proc = subprocess.Popen(
    ['python', 'script.py'],
    stdout=subprocess.PIPE,
    stderr=subprocess.PIPE,
    text=True,

for line in proc.stdout:
    print(line, end='')

# 管道连接
p1 = subprocess.Popen(['cat', 'file.txt'], stdout=subprocess.PIPE)

## 47.2 原理与机制

### 47.2.1 os 模块的 C 封装

```python
# os 模块的核心函数是对 POSIX/Windows API 的直接封装
# 位于 Modules/posixmodule.c（Unix）和 Modules/ntmodule.c（Windows）

"""
os.mkdir(path, mode=0o777)
  ↓
Unix:  mkdir(path, mode)     — POSIX 系统调用
Windows: _mkdir(path)         — Win32 API

os.stat(path)
  ↓
Unix:  stat(path, &buf)      — POSIX 系统调用
Windows: _stat64(path, &buf)  — Win32 API

os.listdir(path)
  ↓
Unix:  opendir() + readdir() + closedir()
Windows: FindFirstFileW() + FindNextFileW() + FindClose()

os.fork()  (Unix only)
  ↓
fork() — POSIX 系统调用，创建子进程
  父进程返回子进程 PID
  子进程返回 0
"""
```

### 47.2.2 sys 模块的运行时状态

```python
import sys

# sys 模块提供对 CPython 运行时内部状态的访问

# sys._getframe() — 访问调用栈帧
def outer():
    def inner():
        frame = sys._getframe()
        print(frame.f_code.co_name)        # 'inner'
        print(frame.f_back.f_code.co_name)  # 'outer'
        print(frame.f_locals)               # 局部变量
        print(frame.f_globals)              # 全局变量
    inner()

# sys.settrace() — 设置跟踪函数（调试器的基础）
def trace_func(frame, event, arg):
    if event == 'call':
        print(f"Calling {frame.f_code.co_name}")
    return trace_func

sys.settrace(trace_func)
# 所有函数调用都会触发 trace_func

# sys.setprofile() — 设置性能分析函数
# 类似 settrace 但开销更小，只追踪 call/return/c_exception

# sys.intern() — 字符串驻留
s1 = sys.intern('hello_world_' + 'x' * 100)
s2 = sys.intern('hello_world_' + 'x' * 100)
print(s1 is s2)  # True — 同一对象，节省内存
# Python 自动驻留短字符串和标识符
# sys.intern() 手动驻留长字符串或动态生成的字符串
```

### 47.2.3 subprocess 的进程创建模型

```python
"""
Unix (fork/exec 模型):
  1. fork() — 创建当前进程的副本（子进程）
     - COW（Copy-on-Write）：子进程共享父进程的内存页
     - 子进程获得父进程文件描述符的副本
  2. exec() — 在子进程中替换为新程序
     - execvp() 搜索 PATH
     - exec 前设置 stdin/stdout 重定向

Windows (CreateProcess 模型):
  1. CreateProcess() — 创建新进程
     - 一步完成，无需 fork
     - 直接指定要运行的程序
     - 可设置启动信息（stdin/stdout 管道等）

subprocess.Popen 的实现:
  1. Unix: os.fork() → 子进程关闭不需要的 fd → os.execvp()
  2. Windows: CreateProcess() 使用 STARTUPINFO 设置管道
  3. 父进程通过管道与子进程通信
"""

# Popen 的管道实现
import subprocess

# subprocess.PIPE 的原理：
# 创建一对 Unix 管道（或 Windows 匿名管道）
# 管道是内核缓冲区，一端写入，一端读取
# 父进程持有读端，子进程持有写端（stdout）
# 或父进程持有写端，子进程持有读端（stdin）

# communicate() 的内部流程：
# 1. 如果有 stdin 数据，写入 stdin 管道
# 2. 使用线程同时读取 stdout 和 stderr（避免死锁）
# 3. 等待子进程退出
# 4. 返回 (stdout_data, stderr_data)
```

### 47.2.4 pathlib 的路径抽象

```python
from pathlib import PurePath, PurePosixPath, PureWindowsPath, Path

# PurePath vs Path 的继承体系
"""
PurePath (抽象基类，不访问文件系统)
  ├── PurePosixPath (POSIX 路径: /usr/bin)
  └── PureWindowsPath (Windows 路径: C:\\Users)

Path (具体路径，访问文件系统)
  ├── PosixPath (POSIX 系统上的 Path)
  └── WindowsPath (Windows 系统上的 Path)
"""

# PurePath 不访问文件系统，只做字符串操作
p = PurePosixPath('/usr/local/bin/python')
print(p.parts)   # ('/', 'usr', 'local', 'bin', 'python')
print(p.parent)  # /usr/local/bin
print(p.name)    # python

# 跨平台路径操作
p = PurePath('dir/subdir/file.txt')
# 在 POSIX 上自动创建 PurePosixPath
# 在 Windows 上自动创建 PureWindowsPath

# 路径拼接使用 / 运算符
# 实现原理：Path.__truediv__() 返回新 Path 对象
base = Path('/usr')
full = base / 'local' / 'bin'
# 等价于：Path('/usr').joinpath('local', 'bin')
```

## 47.3 使用场景

### 47.3.1 跨平台文件管理器

```python
"""跨平台文件管理器"""
import os
import shutil
from pathlib import Path


class FileManager:
    def __init__(self, base_dir=None):
        self.base_dir = Path(base_dir or os.getcwd())

    def list_files(self, pattern='*', recursive=False):
        """列出匹配模式的文件。"""
        if recursive:
            return sorted(self.base_dir.glob(f'**/{pattern}'))
        return sorted(self.base_dir.glob(pattern))

    def ensure_dir(self, path):
        """确保目录存在。"""
        full_path = self.base_dir / path
        full_path.mkdir(parents=True, exist_ok=True)
        return full_path

    def copy_file(self, src, dst):
        """复制文件。"""
        src_path = self.base_dir / src
        dst_path = self.base_dir / dst
        dst_path.parent.mkdir(parents=True, exist_ok=True)
        shutil.copy2(src_path, dst_path)
        return dst_path

    def move_file(self, src, dst):
        """移动文件。"""
        src_path = self.base_dir / src
        dst_path = self.base_dir / dst
        dst_path.parent.mkdir(parents=True, exist_ok=True)
        shutil.move(str(src_path), str(dst_path))
        return dst_path

    def get_size(self, path):
        """获取文件或目录大小。"""
        full_path = self.base_dir / path
        if full_path.is_file():
            return full_path.stat().st_size
        total = 0
        for f in full_path.rglob('*'):
            if f.is_file():
                total += f.stat().st_size
        return total
```

### 47.3.2 进程监控工具

```python
"""简单的进程监控工具"""
import subprocess
import os
import signal
import time


class ProcessMonitor:
    def __init__(self):
        self.processes = {}

    def start(self, name, command, cwd=None, env=None):
        """启动一个进程。"""
        proc = subprocess.Popen(
            command,
            shell=isinstance(command, str),
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE,
            cwd=cwd,
            env=env,
        )
        self.processes[name] = {
            'process': proc,
            'command': command,
            'started': time.time(),
        }
        return proc.pid

    def stop(self, name, timeout=10):
        """停止指定进程。"""
        if name not in self.processes:
            return False

        proc = self.processes[name]['process']
        proc.terminate()  # SIGTERM

        try:
            proc.wait(timeout=timeout)
        except subprocess.TimeoutExpired:
            proc.kill()  # SIGKILL
            proc.wait()

        del self.processes[name]
        return True

    def is_alive(self, name):
        """检查进程是否存活。"""
        if name not in self.processes:
            return False
        return self.processes[name]['process'].poll() is None

    def status(self):
        """获取所有进程状态。"""
        result = {}
        for name, info in self.processes.items():
            proc = info['process']
            result[name] = {
                'pid': proc.pid,
                'running': proc.poll() is None,
                'uptime': time.time() - info['started'],
            }
        return result
```

## 47.4 示例代码

### 47.4.1 日志轮转工具

```python
"""日志文件轮转工具"""
import os
import gzip
import shutil
from pathlib import Path
from datetime import datetime


class LogRotator:
    """按大小和时间轮转日志文件。"""

    def __init__(self, log_path, max_size=100 * 1024 * 1024,
                 max_backups=10, compress=True):
        self.log_path = Path(log_path)
        self.max_size = max_size
        self.max_backups = max_backups
        self.compress = compress

    def should_rotate(self):
        """检查是否需要轮转。"""
        if not self.log_path.exists():
            return False
        return self.log_path.stat().st_size >= self.max_size

    def rotate(self):
        """执行日志轮转。"""
        if not self.should_rotate():
            return

        # 轮转现有备份文件
        for i in range(self.max_backups - 1, 0, -1):
            src = self._backup_path(i)
            dst = self._backup_path(i + 1)
            if src.exists():
                if dst.exists():
                    dst.unlink()
                src.rename(dst)

        # 压缩并移动当前日志
        backup_1 = self._backup_path(1)
        if self.compress:
            with open(self.log_path, 'rb') as f_in:
                with gzip.open(backup_1, 'wb') as f_out:
                    shutil.copyfileobj(f_in, f_out)
            self.log_path.unlink()
        else:
            self.log_path.rename(backup_1)

        # 删除超出数量限制的备份
        for i in range(self.max_backups + 1, self.max_backups + 100):
            p = self._backup_path(i)
            if p.exists():
                p.unlink()
            else:
                break

    def _backup_path(self, index):
        suffix = '.gz' if self.compress else ''
        return self.log_path.parent / f"{self.log_path.stem}.{index}{self.log_path.suffix}{suffix}"


# 使用
rotator = LogRotator('/var/log/app.log', max_size=10 * 1024 * 1024)
if rotator.should_rotate():
    rotator.rotate()
```

## 47.5 常见陷阱与最佳实践

### 47.5.1 路径编码问题

```python
# 陷阱：Windows 上文件名可能使用 GBK 编码
# os.listdir() 在 Python 3 中返回 str（正确解码的 Unicode）

# 最佳实践：
# 1. 始终使用 Path 对象代替字符串拼接
# 2. 处理未知编码的文件名时使用 os.listdir(b'.') 返回 bytes
# 3. 文件操作始终指定 encoding='utf-8'
```

### 47.5.2 符号链接

```python
from pathlib import Path

# 陷阱：Path.resolve() 默认跟随符号链接
# 可能导致循环引用或意外路径

# 最佳实践：
# - 使用 Path.is_symlink() 检查是否为符号链接
# - 使用 Path.lstat() 获取符号链接本身的信息
# - os.walk() 默认不跟随符号链接（防止循环）
# - 需要跟随时设置 followlinks=True（注意循环风险）
```

### 47.5.3 子进程安全

```python
import subprocess

# 陷阱：shell=True 有命令注入风险
user_input = 'file.txt; rm -rf /'
# subprocess.run(f'cat {user_input}', shell=True)  # 危险！

# 最佳实践：避免 shell=True
subprocess.run(['cat', user_input], shell=False)  # 安全

# 必须用 shell 时，使用 shlex 转义
import shlex
safe_input = shlex.quote(user_input)
subprocess.run(f'cat {safe_input}', shell=True)
```

### 47.5.4 总结

| 陷阱 | 最佳实践 |
|------|---------|
| 路径字符串拼接 | 使用 pathlib 的 / 运算符 |
| 路径编码问题 | 使用 Path 对象，指定 encoding |
| shell=True 命令注入 | 使用参数列表或 shlex.quote |
| 符号链接循环 | 检查 is_symlink，注意 resolve() |
| subprocess 死锁 | 使用 communicate() 而非手动读写管道 |
| 忘记关闭文件描述符 | 使用 with 语句或 close_fds=True |
| os.walk 递归深度 | 限制深度或使用 Path.rglob |
