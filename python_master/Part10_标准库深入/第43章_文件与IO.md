## 1. 概念与语法

### 1.1 文件对象协议 (File Object Protocol)

Python 的文件 I/O 建立在一套精心设计的协议之上。`io` 模块定义了文件对象的抽象层次结构，每一层都对应不同的 I/O 语义。

**io 模块的类层次结构：**

```
IOBase                          # 所有 I/O 类的基类
├── RawIOBase                   # 无缓冲原始 I/O
│   ├── FileIO                  # 操作系统级文件描述符封装
│   ├── SocketIO               # socket 封装 (内部使用)
│   └── BytesIO                 # 内存中的原始字节缓冲
├── BufferedIOBase              # 缓冲 I/O 基类
│   ├── BufferedReader          # 缓冲读
│   ├── BufferedWriter          # 缓冲写
│   ├── BufferedRandom          # 缓冲随机访问
│   └── BufferedRWPair         # 缓冲读写对
└── TextIOBase                  # 文本 I/O 基类
    ├── TextIOWrapper           # 将缓冲二进制流解码为文本
    └── StringIO                # 内存中的文本缓冲
```

**核心协议方法：**

```python
# 完整的文件对象协议接口
from typing import Protocol, Any, Iterator

class FileProtocol(Protocol):
    """文件对象的核心协议定义"""

    # --- 上下文管理 ---
    def __enter__(self) -> 'FileProtocol': ...
    def __exit__(self, exc_type, exc_val, exc_tb) -> None: ...

    # --- 读操作 ---
    def read(self, size: int = -1) -> Any: ...        # 读取全部或指定大小
    def readline(self, size: int = -1) -> Any: ...     # 读取一行
    def readlines(self, hint: int = -1) -> list: ...   # 读取所有行
    def __iter__(self) -> Iterator: ...                # 迭代行

    # --- 写操作 ---
    def write(self, s: Any) -> int: ...                # 写入数据
    def writelines(self, lines: list) -> None: ...     # 写入多行

    # --- 定位操作 ---
    def seek(self, offset: int, whence: int = 0) -> int: ...  # 移动指针
    def tell(self) -> int: ...                                  # 获取指针位置
    def truncate(self, size: int = None) -> int: ...           # 截断文件

    # --- 刷新与关闭 ---
    def flush(self) -> None: ...    # 刷新缓冲区
    def close(self) -> None: ...    # 关闭文件
    @property
    def closed(self) -> bool: ...   # 是否已关闭

    # --- 属性 ---
    @property
    def name(self) -> Any: ...      # 文件名
    @property
    def mode(self) -> str: ...      # 打开模式
```

**open() 函数的完整签名：**


**mode 参数详解：**

| 模式 | 描述 | 文件存在时 | 文件不存在时 |
|------|------|-----------|-------------|
| 'r' | 只读 (默认) | 从头读 | 抛出 FileNotFoundError |
| 'w' | 只写 | 清空 | 创建 |
| 'x' | 排他创建 | 抛出 FileExistsError | 创建 |
| 'a' | 追加写 | 从末尾写 | 创建 |
| 'r+' | 读写 | 从头读写 | 抛出 FileNotFoundError |
| 'w+' | 写读 | 清空 | 创建 |
| 'a+' | 追加读写 | 从末尾读/写 | 创建 |

加 `'b'` 后缀切换为二进制模式，加 `'t'` 后缀显式指定文本模式（默认）。

### 1.2 文本 vs 二进制 I/O

```python
# 文本模式 — 返回 str，自动编解码和换行符转换
with open('text.txt', 'r', encoding='utf-8') as f:
    content = f.read()  # 返回 str

# 二进制模式 — 返回 bytes，无编解码，无换行符转换
with open('binary.bin', 'rb') as f:
    content = f.read()  # 返回 bytes
```

文本模式和二进制模式的根本区别在于是否经过 `TextIOWrapper` 的编解码层。在 CPython 源码中，`TextIOWrapper` 的 `read()` 方法会调用 `IncrementalDecoder` 逐块解码字节。

**newline 参数的行为：**

| newline 值 | 读取时 | 写入时 |
|------------|--------|--------|
| None (默认) | 通用换行: `\r\n`, `\r`, `\n` → `\n` | `\n` → `os.linesep` |
| '' | 通用换行但不转换 | `\n` 保持不变 |
| '\n' | 只按 `\n` 分割 | `\n` 保持不变 |
| '\r\n' | 只按 `\r\n` 分割 | `\n` → `\r\n` |

### 2.1 文件对象的 CPython 内部实现

在 CPython 源码中，文件对象的创建链路如下：

```c

// open() 调用链: builtins_open -> io_open -> 

static PyObject *
io_open(PyModuleDef *module, PyObject *args, PyObject *kwds)
{
    // 1. 解析参数
    // 2. 根据 mode 构建文件对象层次:
    //    文本模式: TextIOWrapper(BufferedReader(FileIO(fd)))
    //    二进制模式: BufferedReader(FileIO(fd)) 或 BufferedWriter(FileIO(fd))
    // 3. 返回最外层对象
```

**FileIO — 最底层的原始 I/O：**

```c
// CPython: Modules/_io/fileio.c

typedef struct {
    PyObject_HEAD
    int fd;                    // 操作系统文件描述符
    unsigned int readable:1;   // 可读标志
    unsigned int writable:1;   // 可写标志
    unsigned int appending:1;  // 追加标志
    unsigned int closefd:1;   // close时是否关闭fd
    PyObject *weakreflist;    // 弱引用链表
} fileio;

static Py_ssize_t
fileio_readinto(fileio *self, Py_buffer *buf)
{
    Py_ssize_t n;
    Py_BEGIN_ALLOW_THREADS
    // 直接调用操作系统的 read() 系统调用
    n = read(self->fd, buf->buf, buf->len);
    Py_END_ALLOW_THREADS
    // ...
    return n;
}
```

**BufferedReader — 缓冲层的核心：**

```c
// CPython: Modules/_io/bufferedio.c

typedef struct {
    PyObject_HEAD
    PyObject *raw;             // 底层 RawIOBase 对象
    PyObject *raw_lock;       // 线程锁
    PyObject *buffer;         // 缓冲区 (PyBytes对象)
```

### 2.2 缓冲策略的实现原理

```
buffering=-1 (默认):
  二进制模式 → io.DEFAULT_BUFFER_SIZE (通常 8192)
  文本模式   → io.DEFAULT_BUFFER_SIZE

buffering=0:
  仅二进制模式可用 → 无缓冲, 直接 I/O
  文本模式禁用     → ValueError

buffering=1:
  二进制模式 → 行缓冲 (遇到 \n 刷新)
  文本模式   → 默认行为 (总是行缓冲)

buffering>1:
  指定缓冲区大小 (字节)
```

**缓冲区大小对性能的影响：**

```python
import time

def benchmark_buffer_size(path, sizes):
    """测量不同缓冲区大小对读取性能的影响"""
    results = {}
    for size in sizes:
        start = time.perf_counter()
        with open(path, 'rb', buffering=size) as f:
            while f.read(size):
                pass
        elapsed = time.perf_counter() - start
        results[size] = elapsed
    return results

# 典型结果 (SSD):
# buffering=512:   0.45s  — 缓冲区太小, 系统调用频繁
# buffering=4096:  0.12s  — 接近页面大小, 较优
# buffering=8192:  0.08s  — 默认大小, 平衡
# buffering=65536: 0.05s  — 大缓冲区, 更少系统调用
# buffering=1048576: 0.04s — 边际收益递减
```

### 2.4 编解码器的增量解析原理

CPython 的 `codecs` 模块实现了增量编解码器，用于流式处理：

```python
# CPython: Lib/codecs.py (简化)

class IncrementalDecoder:
    """增量解码器 — 处理不完整的字节序列"""

    def __init__(self, errors='strict'):
        self.errors = errors
        self.buffer = b''  # 保存未完成的字节序列

    def decode(self, input, final=False):
        # 将新数据和缓冲区中的数据合并
        data = self.buffer + input
        result = ''
        # 尝试解码, 将不完整的尾部存入缓冲区
        try:
            result = data.decode(self.encoding, errors=self.errors)
            self.buffer = b''
        except UnicodeDecodeError:
            if final:
                # 最后一次调用, 不完整的序列是真正的错误
                raise
            # 将最后一个可能不完整的字节存入缓冲区
            # 例如 UTF-8 中, 0xE0 开头的3字节序列只收到了前2字节
            self.buffer = self._extract_incomplete(data)
        return result
```

**UTF-8 编码的字节结构：**

```
U+0000 ~ U+007F:  0xxxxxxx                        (1字节)
U+0080 ~ U+07FF:  110xxxxx 10xxxxxx               (2字节)
U+0800 ~ U+FFFF:  1110xxxx 10xxxxxx 10xxxxxx       (3字节)
U+10000~U+10FFFF: 11110xxx 10xxxxxx 10xxxxxx 10xxxxxx (4字节)
```

增量解码器在流式场景中至关重要——当网络数据到达时，一个 UTF-8 多字节序列可能被分割在两次 `recv()` 调用中，增量解码器会将不完整的尾部暂存，等待下次数据到达。

## 3. 使用场景

### 3.1 大文件流式处理

```python
def process_large_file(path, processor, chunk_size=65536):
    """流式处理大文件, 避免一次性加载到内存"""
    with open(path, 'r', encoding='utf-8') as f:
        while True:
            chunk = f.read(chunk_size)
            if not chunk:
                break
            # 注意: chunk 可能截断多字节字符
            # 更安全的方式是按行读取
            processor(chunk)

def process_large_file_by_line(path, processor):
    """按行流式处理 — 推荐方式"""
    with open(path, 'r', encoding='utf-8') as f:
        for line in f:  # 利用缓冲区的迭代器协议
            processor(line)
```

### 3.2 配置文件的原子写入

```python
import os
import tempfile
from pathlib import Path

def atomic_write(path, content, encoding='utf-8'):
    """原子写入文件 — 要么完全成功, 要么完全失败"""
    path = Path(path)
    # 1. 写入临时文件 (同一文件系统, 确保可以原子重命名)
    fd, tmp_path = tempfile.mkstemp(
        dir=path.parent,   # 关键: 同一目录确保同一文件系统
        prefix='.tmp_',
        suffix=path.suffix
    )
    try:
        with os.fdopen(fd, 'w', encoding=encoding) as f:
            f.write(content)
            f.flush()
            # 2. fsync 确保数据落盘
            os.fsync(f.fileno())
        # 3. 原子重命名 (POSIX 上是原子操作)
        os.replace(tmp_path, path)
    except:
        # 清理临时文件
        os.unlink(tmp_path)
        raise
```

### 3.3 内存映射处理超大文件

```python
import mmap
import os

def search_in_large_file(path, pattern):
    """在大文件中搜索模式 — 使用 mmap 避免加载整个文件"""
    pattern_bytes = pattern.encode('utf-8') if isinstance(pattern, str) else pattern
    results = []

    with open(path, 'rb') as f:
        # 获取文件大小
        size = os.fstat(f.fileno()).st_size
        if size == 0:
            return results

        mm = mmap.mmap(f.fileno(), 0, access=mmap.ACCESS_READ)
        try:
            pos = 0
            while True:
                found = mm.find(pattern_bytes, pos)
                if found == -1:
                    break
                results.append(found)
                pos = found + 1
        finally:
            mm.close()

    return results
```

### 3.4 进程间共享内存

```python
import mmap
import os
import struct
import time

def producer():
    """写进程 — 向共享内存写入数据"""
    with open('/tmp/shared.dat', 'wb') as f:
        f.write(b'\x00' * 4096)  # 预分配空间

    with open('/tmp/shared.dat', 'r+b') as f:
        mm = mmap.mmap(f.fileno(), 0)
        try:
            # 写入标志位 (偏移0) 和数据 (偏移4)
            for i in range(100):
                struct.pack_into('!I', mm, 0, 1)  # 标志: 数据就绪
                struct.pack_into('!I', mm, 4, i)   # 数据: 计数值
                time.sleep(0.1)
            struct.pack_into('!I', mm, 0, 0)  # 标志: 结束
        finally:
            mm.close()

def consumer():
    """读进程 — 从共享内存读取数据"""
    with open('/tmp/shared.dat', 'r+b') as f:
        mm = mmap.mmap(f.fileno(), 0)
        try:
            while True:
                flag = struct.unpack_from('!I', mm, 0)[0]
                if flag == 0:
                    break
                if flag == 1:
                    value = struct.unpack_from('!I', mm, 4)[0]
                    print(f"Read: {value}")
                    struct.pack_into('!I', mm, 0, 2)  # 标志: 已读取
                time.sleep(0.05)
        finally:
            mm.close()
```

```python
        self.path = Path(path)
        self._last_stat = self._get_stat()

    def _get_stat(self):
        try:
            st = self.path.stat()
            return (st.st_mtime_ns, st.st_size)
        except FileNotFoundError:
            return None

    def check(self):
        """检查文件是否变更, 返回变更类型"""
        current = self._get_stat()
        if current == self._last_stat:
            return None
        old = self._last_stat
        self._last_stat = current
        if old is None and current is not None:
            return 'created'
        if old is not None and current is None:
            return 'deleted'
        return 'modified'
```

```python
"""
实现一个符合文件对象协议的内存文件系统
支持读/写/seek/tell/迭代
"""
import io
from typing import Optional

class InMemoryFile:
    """基于 bytearray 的内存文件, 完整实现文件对象协议"""

    def __init__(self, initial_data: bytes = b'', mode: str = 'r'):
        self._buffer = bytearray(initial_data)
        self._pos = 0
        self._mode = mode
        self._closed = False

        if 'w' in mode:
            self._buffer.clear()

    def __enter__(self):
        return self

    def __exit__(self, *args):
        self.close()

    def _check_closed(self):
        if self._closed:
            raise ValueError("I/O operation on closed file")

    def _check_readable(self):
        if 'r' not in self._mode and '+' not in self._mode:
            raise IOError("File not open for reading")

    def _check_writable(self):
        if 'w' not in self._mode and 'a' not in self._mode and '+' not in self._mode:
            raise IOError("File not open for writing")

    def read(self, size: int = -1) -> bytes:
        self._check_closed()
        self._check_readable()
        if size < 0:
            result = bytes(self._buffer[self._pos:])
            self._pos = len(self._buffer)
        else:
            result = bytes(self._buffer[self._pos:self._pos + size])
            self._pos += len(result)
        return result

    def write(self, data: bytes) -> int:
        self._check_closed()
        self._check_writable()
        if isinstance(data, str):
            data = data.encode('utf-8')
        if 'a' in self._mode:
            self._pos = len(self._buffer)
        # 如果在中间写入, 需要扩展缓冲区
        end = self._pos + len(data)
        if end > len(self._buffer):
            self._buffer.extend(b'\x00' * (end - len(self._buffer)))
        self._buffer[self._pos:end] = data
        self._pos = end
        return len(data)

    def seek(self, offset: int, whence: int = 0) -> int:
        self._check_closed()
        if whence == 0:  # SEEK_SET
            self._pos = offset
        elif whence == 1:  # SEEK_CUR
            self._pos += offset
        elif whence == 2:  # SEEK_END
            self._pos = len(self._buffer) + offset
        else:
            raise ValueError(f"Invalid whence: {whence}")
        return self._pos

    def tell(self) -> int:
        self._check_closed()
        return self._pos

    def readline(self, size: int = -1) -> bytes:
        self._check_closed()
        self._check_readable()
        newline_pos = self._buffer.find(b'\n', self._pos)
        if newline_pos == -1:
            # 没有换行符, 读取到末尾
            result = bytes(self._buffer[self._pos:])
            self._pos = len(self._buffer)
        else:
            end = newline_pos + 1  # 包含换行符
            if size > 0:
                end = min(end, self._pos + size)
            result = bytes(self._buffer[self._pos:end])
            self._pos = end
        return result

    def readlines(self, hint: int = -1) -> list:
        self._check_closed()
        lines = []
        total = 0
        while True:
            line = self.readline()
            if not line:
                break
            lines.append(line)
            total += len(line)
            if 0 < hint <= total:
                break
        return lines

    def __iter__(self):
        return self

    def __next__(self):
        line = self.readline()
        if not line:
            raise StopIteration
        return line

    def truncate(self, size: Optional[int] = None) -> int:
        self._check_closed()
        if size is None:
            size = self._pos
        del self._buffer[size:]
        return size

    def flush(self):
        self._check_closed()
        # 内存文件, flush 是空操作

    def close(self):
        self._closed = True

    @property
    def closed(self):
        return self._closed

    @property
    def name(self):
        return '<memory>'

    @property
    def mode(self):
        return self._mode


# 测试
```

```python
def scan_directory(root, pattern='**/*.py', max_depth=None):
    """递归扫描目录, 支持深度限制"""
    root = Path(root).resolve()
    results = []

    for path in root.glob(pattern):
        # 计算相对深度
        rel = path.relative_to(root)
        depth = len(rel.parts)

        if max_depth is not None and depth > max_depth:

            st = path.stat()
            results.append({
                'path': str(path),
                'relative': str(rel),
                'size': st.st_size,
                'modified': time.strftime('%Y-%m-%d %H:%M:%S',
                                          time.localtime(st.st_mtime)),
                'type': 'dir' if path.is_dir() else 'file',
            })
        except (PermissionError, FileNotFoundError):
            continue

    return results

    """安全地拼接路径, 防止路径遍历攻击"""
    base = Path(base_dir).resolve()
    target = (base / user_path).resolve()

    # 确保目标路径在基础目录内
    try:
        target.relative_to(base)
    except ValueError:
        raise ValueError(f"Path traversal detected: {user_path}")

    return target


def find_duplicates(directory):
    """基于文件大小和内容哈希查找重复文件"""
    import hashlib

    directory = Path(directory)
    size_map = {}

    # 第一轮: 按文件大小分组
    for path in directory.rglob('*'):
        if path.is_file():
            st = path.stat()
            size_map.setdefault(st.st_size, []).append(path)

    # 第二轮: 对相同大小的文件计算哈希
    hash_map = {}
    duplicates = []

    for size, paths in size_map.items():
        if len(paths) < 2:
            continue
        for path in paths:
            try:
                h = hashlib.sha256(path.read_bytes()).hexdigest()
                key = (size, h)
                hash_map.setdefault(key, []).append(path)
            except (PermissionError, OSError):
                continue

    for key, paths in hash_map.items():
        if len(paths) > 1:
            duplicates.append(paths)

    return duplicates
```

```python
    """安全的增量解码器 — 处理未知编码和错误"""

    def __init__(self, encoding='utf-8', errors='replace'):
        self.encoding = encoding
        self.errors = errors
        self.decoder = codecs.getincrementaldecoder(encoding)(errors=errors)

    def decode(self, data, final=False):
        try:
            return self.decoder.decode(data, final)
        except UnicodeDecodeError:
            # 回退策略: 逐字节解码
            result = []
            for byte in data:
                try:

                except:
                    result.append('�')  # Unicode 替换字符
            return ''.join(result)

            errors=self.errors

            if 'a' <= c <= 'z':

            elif 'A' <= c <= 'Z':
```

```python
"""
使用 mmap 创建和操作结构化二进制数据文件
支持索引查找和原子更新
"""
import mmap
import os
import struct
from pathlib import Path


class MmapIndexFile:
    """
    基于 mmap 的索引文件
    布局:
      [0:4]   魔数 (0x49445800 = "IDX\0")
      [4:8]   版本号
      [8:12]  记录数
      [12:16] 记录大小
      [16:]   记录数组 (每条: key(8字节) + offset(8字节) + length(4字节))
    """

    MAGIC = 0x49445800
    VERSION = 1
    HEADER_SIZE = 16
    RECORD_SIZE = 20  # 8 + 8 + 4

    def __init__(self, path, max_records=10000):
        self.path = Path(path)
        self.max_records = max_records
        total_size = self.HEADER_SIZE + max_records * self.RECORD_SIZE

        if not self.path.exists():
            self._create(total_size)

        self._file = open(self.path, 'r+b')
        self._mmap = mmap.mmap(self._file.fileno(), 0)

    def _create(self, size):
        """创建并初始化索引文件"""
        with open(self.path, 'wb') as f:
            f.write(b'\x00' * size)
        with open(self.path, 'r+b') as f:
            mm = mmap.mmap(f.fileno(), 0)
            struct.pack_into('!III', mm, 0,
                             self.MAGIC, self.VERSION, 0)
            mm.flush()
            mm.close()

    def _record_offset(self, index):
        """计算第 index 条记录的偏移量"""
        return self.HEADER_SIZE + index * self.RECORD_SIZE

    def put(self, key, offset, length):
        """插入或更新一条记录"""
        count = struct.unpack_from('!I', self._mmap, 8)[0]

        # 查找空位或已有 key
        for i in range(count):
            off = self._record_offset(i)
            existing_key = struct.unpack_from('!Q', self._mmap, off)[0]
            if existing_key == key:
                # 更新已有记录
                struct.pack_into('!QI', self._mmap, off + 8, offset, length)
                return

        # 追加新记录
        if count >= self.max_records:
            raise OverflowError("Index file is full")
        off = self._record_offset(count)
        struct.pack_into('!QQI', self._mmap, off, key, offset, length)
        struct.pack_into('!I', self._mmap, 8, count + 1)

    def get(self, key):
        """查找记录"""
        count = struct.unpack_from('!I', self._mmap, 8)[0]
        for i in range(count):
            off = self._record_offset(i)
            existing_key = struct.unpack_from('!Q', self._mmap, off)[0]
            if existing_key == key:
                offset = struct.unpack_from('!Q', self._mmap, off + 8)[0]
                length = struct.unpack_from('!I', self._mmap, off + 16)[0]
                return offset, length
        return None

    def delete(self, key):
        """删除记录 — 标记为已删除 (key=0)"""
        count = struct.unpack_from('!I', self._mmap, 8)[0]
        for i in range(count):
            off = self._record_offset(i)
            existing_key = struct.unpack_from('!Q', self._mmap, off)[0]
            if existing_key == key:
                struct.pack_into('!Q', self._mmap, off, 0)
                return True
        return False

    def items(self):
        """遍历所有有效记录"""
        count = struct.unpack_from('!I', self._mmap, 8)[0]
        for i in range(count):
            off = self._record_offset(i)
            key = struct.unpack_from('!Q', self._mmap, off)[0]
            if key == 0:  # 跳过已删除的
                continue
            offset = struct.unpack_from('!Q', self._mmap, off + 8)[0]
            length = struct.unpack_from('!I', self._mmap, off + 16)[0]
            yield key, offset, length

    def close(self):
        if self._mmap:
            self._mmap.flush()
            self._mmap.close()
        if self._file:
            self._file.close()

    def __enter__(self):
        return self

    def __exit__(self, *args):
        self.close()


if __name__ == '__main__':
    import tempfile

    with tempfile.TemporaryDirectory() as td:
        idx_path = os.path.join(td, 'test.idx')

            # 插入记录

            idx.put(key=1002, offset=256, length=512)
            idx.put(key=1003, offset=768, length=128)

            # 查找
            result = idx.get(1002)
            print(f"Key 1002: offset={result[0]}, length={result[1]}")

            # 更新
            idx.put(key=1002, offset=1024, length=2048)

            # 删除
            idx.delete(1001)
            print(f"Key 1001 after delete: {idx.get(1001)}")
```

```python
        else:

                dir=self._path, suffix=suffix

            path.write_bytes(content)
```

### 5.1 常见陷阱

#### 陷阱 1: 未正确关闭文件导致资源泄漏

```python

# 更好 — 使用 os.scandir (一次系统调用获取所有信息)
py_files = []
with os.scandir('src') as entries:
    for entry in entries:
        if entry.is_file() and entry.name.endswith('.py'):
            py_files.append(entry.path)

# pathlib 在 Python 3.12+ 的 walk() 性能已改善
# 但对于大量文件的场景, os.scandir 仍然更快
```

#### 陷阱 5: mmap 的常见错误

```python
import mmap

# 陷阱 1: 空文件无法 mmap
# mmap 要求文件大小 > 0
with open('empty.dat', 'wb') as f:
    pass  # 创建空文件

try:
    with open('empty.dat', 'rb') as f:
        mm = mmap.mmap(f.fileno(), 0)  # ValueError: cannot mmap an empty file
except ValueError as e:
    print(f"错误: {e}")

# 解决: 先给文件分配大小
with open('empty.dat', 'r+b') as f:
    f.truncate(4096)  # 预分配空间
    mm = mmap.mmap(f.fileno(), 0)

# 陷阱 2: mmap 大小必须匹配文件
with open('data.dat', 'wb') as f:
    f.write(b'x' * 100)

with open('data.dat', 'rb') as f:
    # length=0 表示映射整个文件
    mm = mmap.mmap(f.fileno(), 0, access=mmap.ACCESS_READ)
    print(len(mm))  # 100

    # length=200 会扩展映射, 但文件只有 100 字节
    # 在 Linux 上: 扩展到 200, 多出的部分填充 \x00
    # 在 Windows 上: 可能抛出异常
```

#### 陷阱 6: 临时文件的安全问题

```python
import tempfile

# 危险 — mktemp() 存在 TOCTOU 竞态条件
path = tempfile.mktemp()  # 仅生成路径名, 不创建文件
# 在这个时间窗口内, 攻击者可能创建同名的符号链接!

    f.write('sensitive data')

# 安全 — 使用 mkstemp() 或 NamedTemporaryFile
fd, path = tempfile.mkstemp()  # 原子地创建文件
with os.fdopen(fd, 'w') as f:
    f.write('sensitive data')

# 更安全 — 使用 with 语句自动清理
with tempfile.NamedTemporaryFile(mode='w', delete=True) as tf:
    tf.write('sensitive data')
    # 文件在 with 块结束后自动删除
```

```python
def count_lines(path):
    """高效计算大文件的行数"""
    count = 0
    with open(path, 'rb') as f:
        # 分块读取, 避免一次性加载
        buf_size = 1024 * 1024  # 1MB
        while True:
            buf = f.read(buf_size)
            if not buf:
                break
            count += buf.count(b'\n')
    return count


def stream_process(path, process_func, chunk_size=65536):
    """流式处理大文件"""
    from hashlib import sha256
    hash_obj = sha256()
    total = 0

    with open(path, 'rb') as f:
        while True:
            chunk = f.read(chunk_size)
            if not chunk:
                break
            result = process_func(chunk)
            hash_obj.update(chunk)
            total += len(chunk)

    return total, hash_obj.hexdigest()
```

#### 实践 6: 原子写入确保数据完整性

```python
import os
import tempfile
from pathlib import Path


def atomic_write_text(path, text, encoding='utf-8'):
    """原子写入文本文件"""
    path = Path(path)
    path.parent.mkdir(parents=True, exist_ok=True)

    # 写入临时文件
    fd, tmp = tempfile.mkstemp(
        dir=path.parent,  # 同一文件系统确保 rename 原子性
        prefix=f'.{path.name}.',
        suffix='.tmp'
    )
    try:
        with os.fdopen(fd, 'w', encoding=encoding) as f:
            f.write(text)
            f.flush()
            os.fsync(f.fileno())  # 确保数据落盘

        # 原子替换
        os.replace(tmp, path)
    except:
        os.unlink(tmp)
        raise


def atomic_write_binary(path, data):
    """原子写入二进制文件"""
    path = Path(path)
    path.parent.mkdir(parents=True, exist_ok=True)

    fd, tmp = tempfile.mkstemp(dir=path.parent, prefix=f'.{path.name}.')
    try:
        with os.fdopen(fd, 'wb') as f:
            f.write(data)
            f.flush()
            os.fsync(f.fileno())

        os.replace(tmp, path)
    except:
        os.unlink(tmp)
        raise
```

#### 实践 7: 编码问题的防御性编程

```python
def safe_read_file(path, fallback_encodings=('utf-8', 'gbk', 'latin-1')):
    """安全读取文件, 自动检测编码"""
    from pathlib import Path

    path = Path(path)

    # 首先尝试读取 BOM
    raw = path.read_bytes()[:4]
    if raw.startswith(b'\xef\xbb\xbf'):
        return path.read_text(encoding='utf-8-sig')

    # 尝试各种编码
    for encoding in fallback_encodings:
        try:
            return path.read_text(encoding=encoding)
        except UnicodeDecodeError:
            continue

    # 最后手段: 忽略错误
    return path.read_text(encoding='utf-8', errors='replace')
```
