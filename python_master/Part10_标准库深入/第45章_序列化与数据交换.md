# 第45章 序列化与数据交换

序列化是将内存中的对象转换为可存储或可传输的字节流的过程。本章从 pickle 的虚拟机原理、JSON 的 C 加速实现到 struct 的字节序处理，深入剖析 Python 序列化机制的底层原理。

## 45.1 概念与语法

### 45.1.1 pickle 模块

```python
import pickle

# 序列化（dump/dumps）
data = {'name': 'Alice', 'scores': [90, 85, 92], 'active': True}

# dumps — 序列化为 bytes 对象
serialized = pickle.dumps(data)
print(type(serialized))  # <class 'bytes'>
print(len(serialized))   # 约 50+ bytes

# loads — 从 bytes 反序列化
restored = pickle.loads(serialized)
print(restored)  # {'name': 'Alice', 'scores': [90, 85, 92], 'active': True}

# dump/load — 序列化到文件 / 从文件反序列化
with open('data.pkl', 'wb') as f:
    pickle.dump(data, f)

with open('data.pkl', 'rb') as f:
    restored = pickle.load(f)
```

**pickle 协议版本**：

| 版本 | Python | 特性 |
|------|--------|------|
| 0 | 2.x | ASCII 文本协议，可读 |
| 1 | 2.x | 旧二进制格式 |
| 2 | 2.3+ | 高效二进制格式，支持新式类 |
| 3 | 3.0+ | Python 3 专用，支持 bytes |
| 4 | 3.4+ | 支持大对象（4GB+）、带关键字的参数、扩展的 __reduce__ |
| 5 | 3.8+ | 带外数据（out-of-band data），支持序列化大于 4GB 的对象 |

```python

print(f"协议5: {len(serialized_v5)} bytes (带外数据)")

print(pickle.HIGHEST_PROTOCOL)  # 5
```

```python
            return obj.isoformat()
        return super().default(obj)

    for key, value in d.items():

            except ValueError:
                pass
    return d
```

## 45.2 原理与机制

### 45.2.1 PickleMachine 虚拟机

pickle 的反序列化本质上是一个**栈式虚拟机**的执行过程。pickle 字节流由一系列操作码（opcode）组成：

```python
import pickle
import pickletools

# 查看 pickle 操作码
data = [1, 'hello', {'key': 'value'}]
serialized = pickle.dumps(data, protocol=2)

# 反汇编 pickle 字节流
pickletools.dis(serialized)
# 输出：
# 0: \x80 PROTO      2           # 协议版本
# 2: \x95 FRAME      ...         # 帧长度
#    q    BINPUT     0           # 存入 memo[0]
#    }    EMPTY_DICT              # 压入空字典
#    q    BINPUT     1
#    (    MARK                   # 压入 mark
#    X    BINUNICODE 'key'       # 压入 Unicode 字符串
#    q    BINPUT     2
#    X    BINUNICODE 'value'
#    q    BINPUT     3
#    u    SETITEMS               # 从 mark 到栈顶的键值对设置到字典
#    h    BINGET     0
#    X    BINUNICODE 'hello'
#    q    BINPUT     4
#    K    SHORT_BININT 1
#    \x86 TUPLE2                 # 创建二元组
#    \x86 TUPLE2
#    .    STOP                   # 结束
```

**核心操作码**：

| 操作码 | 功能 | 栈效果 |
|--------|------|--------|
| `PROTO` | 声明协议版本 | - |
| `NONE` | 压入 None | +1 |
| `NEWTRUE` | 压入 True | +1 |
| `NEWFALSE` | 压入 False | +1 |
| `INT` | 压入整数 | +1 |
| `BINUNICODE` | 压入 Unicode 字符串 | +1 |
| `EMPTY_LIST` | 压入空列表 | +1 |
| `EMPTY_DICT` | 压入空字典 | +1 |
| `APPENDS` | 将 mark 以上的元素追加到列表 | -n |
| `SETITEMS` | 将 mark 以上的键值对设置到字典 | -2n |
| `TUPLE` | 从 mark 到栈顶创建元组 | -n+1 |
| `REDUCE` | 调用 `__reduce__` 返回的可调用对象 | -2+1 |
| `BUILD` | 调用 `__setstate__` 或更新 `__dict__` | -1 |
| `STOP` | 结束反序列化 | - |

### 45.2.2 `__reduce__` 协议

`__reduce__` 是 pickle 序列化自定义对象的核心协议：

```python
import pickle

class Point:
    def __init__(self, x, y):
        self.x = x
        self.y = y

    def __reduce__(self):
        # 返回 (callable, args) — 反序列化时调用 callable(*args)
        return (self.__class__, (self.x, self.y))

    def __eq__(self, other):
        return self.x == other.x and self.y == other.y

p = Point(3, 4)
serialized = pickle.dumps(p)
restored = pickle.loads(serialized)
print(restored.x, restored.y)  # 3 4

# __reduce__ 返回值的完整形式：
# (callable, args, state, list_items, dict_items, state_setter)
# - callable(*args) 创建空对象
# - state 传给 __setstate__(state) 或更新 __dict__
# - list_items 追加到列表
# - dict_items 更新到字典
# - state_setter(state) 是 Python 3.8+ 的协议5新特性

class AdvancedPoint:
    def __init__(self, x, y, color='red'):
        self.x = x
        self.y = y
        self.color = color
        self._cache = {}  # 不参与序列化

    def __reduce__(self):
        return (
            self.__class__,           # callable
            (self.x, self.y),         # args
            {'color': self.color},    # state（传给 __setstate__）
        )

    def __setstate__(self, state):
        self.color = state['color']
        self._cache = {}  # 重建非序列化属性
```

### 45.2.3 pickle 安全漏洞

```python
import pickle

# 危险！pickle 可以执行任意代码
# 以下操作码会执行 os.system('rm -rf /')
# pickle.loads(b"cos\nsystem\n(S'echo DANGER'\ntR.")  # 绝对不要执行！

# 攻击原理：REDUCE 操作码调用任意可调用对象
# pickle 字节流中的 REDUCE + 对 os.system 的引用 = 任意代码执行

# 安全替代方案：使用 pickletools.optimize 手动审查，或使用更安全的格式

# 受限反序列化
import io

class SafeUnpickler(pickle.Unpickler):
    """只允许反序列化基本类型的 Unpickler。"""

    ALLOWED_CLASSES = {
        'builtins': {'list', 'dict', 'tuple', 'set', 'frozenset', 'str', 'int', 'float', 'bool'},
    }

    def find_class(self, module, name):
        if module in self.ALLOWED_CLASSES:
            if name in self.ALLOWED_CLASSES[module]:
                return getattr(__import__(module), name)
        raise pickle.UnpicklingError(
            f"Forbidden: {module}.{name}"
        )

def safe_loads(data):
    return SafeUnpickler(io.BytesIO(data)).load()

# 使用
safe_data = pickle.dumps({'key': 'value'})  # 安全数据
result = safe_loads(safe_data)  # OK

# 如果数据包含危险引用，safe_loads 会抛出异常
```

```python
"""多格式配置文件加载器"""
import json
import configparser
from pathlib import Path


def load_config(path):
    """根据文件扩展名自动选择解析器。"""
    path = Path(path)
    suffix = path.suffix.lower()

    if suffix == '.json':
        with open(path, encoding='utf-8') as f:
            return json.load(f)
    elif suffix == '.ini':
        config = configparser.ConfigParser()
        config.read(path, encoding='utf-8')
        return {s: dict(config[s]) for s in config.sections()}
    elif suffix == '.py':
        # Python 配置文件（安全风险，仅用于受信环境）
        ns = {}
        exec(path.read_text(encoding='utf-8'), ns)
        return {k: v for k, v in ns.items() if not k.startswith('_')}
    else:
        raise ValueError(f"Unsupported config format: {suffix}")
```

```python
from decimal import Decimal


class APIEncoder(json.JSONEncoder):
    """支持常见 Python 类型的 JSON 编码器。"""

            return float(obj)
```

### 45.3.3 缓存序列化

```python
"""磁盘缓存：使用 pickle 存储计算结果"""
import pickle
import hashlib
from pathlib import Path
from functools import wraps


def disk_cache(cache_dir='.cache'):
    """将函数结果缓存到磁盘。"""
    cache_path = Path(cache_dir)
    cache_path.mkdir(exist_ok=True)

    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            # 生成缓存键
            key_data = pickle.dumps((args, sorted(kwargs.items())))
            key = hashlib.md5(key_data).hexdigest()
            cache_file = cache_path / f"{func.__name__}_{key}.pkl"

            # 检查缓存
            if cache_file.exists():
                with open(cache_file, 'rb') as f:
                    return pickle.load(f)

            # 计算并缓存
            result = func(*args, **kwargs)
            with open(cache_file, 'wb') as f:
                pickle.dump(result, f, protocol=pickle.HIGHEST_PROTOCOL)

            return result

        return wrapper
    return decorator


# 使用
@disk_cache()
def expensive_computation(n):
    return sum(i ** 2 for i in range(n))
```

### 45.4.1 安全序列化方案

```python
"""安全的序列化替代方案：避免 pickle 的任意代码执行风险"""

import json
import hashlib
import hmac
from datetime import datetime


class SafeSerializer:
    """基于 JSON + HMAC 签名的安全序列化器。"""

    def __init__(self, secret_key: str):
        self.secret_key = secret_key.encode('utf-8')

    def dumps(self, data: dict) -> str:
        """序列化并签名。"""
        payload = json.dumps(data, sort_keys=True, separators=(',', ':'))
        signature = hmac.new(
            self.secret_key,
            payload.encode('utf-8'),
            hashlib.sha256
        ).hexdigest()
        return f"{signature}.{payload}"

    def loads(self, token: str) -> dict:
        """验证签名并反序列化。"""
        try:
            signature, payload = token.split('.', 1)
        except ValueError:
            raise ValueError("Invalid token format")

        expected = hmac.new(
            self.secret_key,
            payload.encode('utf-8'),
            hashlib.sha256
        ).hexdigest()

        if not hmac.compare_digest(signature, expected):
            raise ValueError("Invalid signature")

        return json.loads(payload)


# 使用
serializer = SafeSerializer('my-secret-key')
token = serializer.dumps({'user_id': 42, 'role': 'admin'})
data = serializer.loads(token)
print(data)  # {'user_id': 42, 'role': 'admin'}

# 篡改检测
try:
    tampered = token.replace('42', '99')
    serializer.loads(tampered)  # ValueError: Invalid signature
except ValueError as e:
    print(f"检测到篡改: {e}")
```

### 45.4.2 二进制协议解析器

```python
"""基于 struct 的二进制协议解析器"""

import struct
from typing import Dict, Any


class BinaryProtocolParser:
    """解析自定义二进制协议的消息。"""

    # 消息头格式：魔数(2) + 版本(1) + 类型(1) + 长度(4)
    HEADER_FORMAT = '!HBBH'  # 网络序：magic(2) + version(1) + type(1) + length(2)
    HEADER_SIZE = struct.calcsize('!HBBH')

    MAGIC = 0xCAFE
    VERSION = 1

    MSG_TYPES = {
        0x01: 'ping',
        0x02: 'pong',
        0x10: 'data',
        0xFF: 'error',
    }

    def parse_header(self, data: bytes) -> Dict[str, Any]:
        """解析消息头。"""
        if len(data) < self.HEADER_SIZE:
            raise ValueError(f"Header too short: {len(data)}")

        magic, version, msg_type, length = struct.unpack(
            self.HEADER_FORMAT, data[:self.HEADER_SIZE]
        )

        if magic != self.MAGIC:
            raise ValueError(f"Invalid magic: 0x{magic:04X}")
        if version != self.VERSION:
            raise ValueError(f"Unsupported version: {version}")

        return {
            'magic': magic,
            'version': version,
            'type': self.MSG_TYPES.get(msg_type, f'unknown(0x{msg_type:02X})'),
            'type_code': msg_type,
            'length': length,
        }

    def build_message(self, msg_type: int, payload: bytes = b'') -> bytes:
        """构建消息。"""
        header = struct.pack(
            self.HEADER_FORMAT,
            self.MAGIC,
            self.VERSION,
            msg_type,
            len(payload),
        )
        return header + payload


# 使用
parser = BinaryProtocolParser()

# 构建消息
msg = parser.build_message(0x10, b'hello world')
print(f"消息总长: {len(msg)} bytes")

# 解析消息
header = parser.parse_header(msg)
print(header)
# {'magic': 51966, 'version': 1, 'type': 'data', 'type_code': 16, 'length': 11}
```

### 45.5.2 JSON 日期处理

```python
# 陷阱：JSON 不支持日期类型
data = {'date': date(2024, 1, 15)}
# json.dumps(data)  # TypeError: Object of type date is not JSON serializable

# 修复方案1：自定义编码器
class DateEncoder(json.JSONEncoder):
    def default(self, obj):
        if isinstance(obj, (datetime, date)):
            return obj.isoformat()
        return super().default(obj)

json.dumps(data, cls=DateEncoder)  # {"date": "2024-01-15"}
```

### 45.5.4 版本兼容性

```python
import pickle

# 陷阱：高协议版本的 pickle 数据无法在旧版本 Python 中加载
# 协议5的带外数据在 Python 3.7 中无法读取

# 最佳实践：
# 1. 跨版本传输使用较低协议
# 2. 同环境内部使用最高协议
# 3. 长期存储使用 JSON 或其他文本格式

# 跨版本兼容
def compatible_dumps(obj, min_python_version=(3, 6)):
    """根据目标 Python 版本选择协议。"""
    if min_python_version >= (3, 8):
        protocol = 5
    elif min_python_version >= (3, 4):
        protocol = 4
    else:
        protocol = 3
    return pickle.dumps(obj, protocol=protocol)
```

### 45.5.5 循环引用与递归深度

```python
import pickle
import sys

# 陷阱：循环引用可能导致 pickle 栈溢出
a = {}
b = {'ref': a}
a['ref'] = b

# pickle 能处理循环引用（使用 memo 机制）
serialized = pickle.dumps(a)  # OK
restored = pickle.loads(serialized)  # OK

# 但深度嵌套可能超出限制
deep = []
current = deep
for _ in range(1000):
    current.append([])
    current = current[0]

try:
    pickle.dumps(deep)
except RecursionError:
    print("递归深度超限")
    # 修复：增加递归限制
    sys.setrecursionlimit(10000)

# JSON 不支持循环引用
# json.dumps(a)  # ValueError: Circular reference detected
```

### 45.5.6 性能最佳实践

```python
import pickle
import json
import timeit

data = {'items': list(range(10000)), 'name': 'test' * 100}

# pickle vs JSON 性能对比
t_pickle = timeit.timeit(lambda: pickle.dumps(data), number=1000)
t_json = timeit.timeit(lambda: json.dumps(data), number=1000)
print(f"pickle 序列化: {t_pickle:.4f}s")
print(f"JSON 序列化:   {t_json:.4f}s")
# pickle 通常快 2-3 倍

# 高协议版本更紧凑更快速
for proto in range(pickle.HIGHEST_PROTOCOL + 1):
    size = len(pickle.dumps(data, protocol=proto))
    print(f"协议 {proto}: {size} bytes")

# 使用 orjson 获得最佳 JSON 性能（pip install orjson）
# import orjson
```

### 45.5.7 总结

| 陷阱 | 最佳实践 |
|------|---------|
| pickle 任意代码执行 | 不受信任数据使用 JSON，或用 SafeUnpickler |
| JSON 不支持日期 | 自定义 JSONEncoder 或预处理 |
| Windows 编码问题 | 始终指定 encoding='utf-8' |
| 高协议版本不兼容 | 跨版本使用低协议或 JSON |
| 循环引用栈溢出 | pickle 可处理但需注意深度 |
| JSON 序列化性能低 | 考虑 orjson 或 msgpack |
| pickle 二进制模式遗漏 | 必须用 'wb'/'rb' 模式 |
