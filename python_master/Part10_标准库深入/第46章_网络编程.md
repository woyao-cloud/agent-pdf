### 46.1.2 socket 选项

```python
import socket

sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)

# SO_REUSEADDR — 允许重用处于 TIME_WAIT 的地址
sock.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)

# SO_KEEPALIVE — 启用 TCP 保活
sock.setsockopt(socket.SOL_SOCKET, socket.SO_KEEPALIVE, 1)

# TCP_NODELAY — 禁用 Nagle 算法（减少延迟）
sock.setsockopt(socket.IPPROTO_TCP, socket.TCP_NODELAY, 1)

# SO_RCVBUF / SO_SNDBUF — 设置收发缓冲区大小
sock.setsockopt(socket.SOL_SOCKET, socket.SO_RCVBUF, 65536)
sock.setsockopt(socket.SOL_SOCKET, socket.SO_SNDBUF, 65536)

# SO_RCVTIMEO / SO_SNDTIMEO — 设置超时
sock.settimeout(5.0)  # 简便方法

# SO_LINGER — 控制 close() 行为
# l_onoff=1, l_linger=0: close() 立即发送 RST 而非 FIN
linger = struct.pack('ii', 1, 0)
sock.setsockopt(socket.SOL_SOCKET, socket.SO_LINGER, linger)
```

## 46.2 厘理与机制

### 46.2.1 socket 的内核实现

Python 的 socket 模块是对操作系统 socket API 的薄封装：

```
Python socket 模块
    ↓ 调用
Modules/_socket.c (C 扩展)
    ↓ 系统调用
Linux: socket() / bind() / listen() / accept() / recv() / send()
Windows: WSASocket() / bind() / listen() / accept() / recv() / send()
    ↓ 内核
TCP/IP 协议栈 (Linux: net/ipv4/, Windows: tcpip.sys)
    ↓ 硬件
网卡驱动 → 物理网络
```

**关键系统调用**：

```python
# socket() — 创建套接字
# Python: sock = socket.socket(AF_INET, SOCK_STREAM)
# 内核: 分配 socket 结构体，返回文件描述符

# bind() — 绑定地址
# Python: sock.bind(('0.0.0.0', 8080))
# 内核: 将 socket 与 (IP, port) 关联

# listen() — 开始监听
# Python: sock.listen(5)
# 内核: 将 socket 状态改为 LISTEN，创建 SYN 队列和 accept 队列
# backlog=5 表示 accept 队列最大长度（非 SYN 队列）

# accept() — 接受新连接
# Python: conn, addr = sock.accept()
# 内核: 从 accept 队列取出已完成的连接，创建新 socket
# 阻塞模式：队列为空时阻塞
# 非阻塞模式：队列为空时抛出 BlockingIOError

# connect() — 发起连接
# Python: sock.connect(('host', 80))
# 内核: 发送 SYN，等待 SYN-ACK，发送 ACK（三次握手）

# recv() — 接收数据
# Python: data = sock.recv(4096)
# 内核: 从接收缓冲区读取数据

# send() — 发送数据
# Python: sock.sendall(data)
# 内核: 将数据写入发送缓冲区
# send() 只保证写入缓冲区，不保证对方已收到
# sendall() 循环调用 send() 直到所有数据写入
```

### 46.2.2 TCP 三次握手与四次挥手

```python
"""
TCP 三次握手在 Python 中的体现：

客户端                          服务器
  |                               |
  |--- SYN (seq=x) ------------>|  connect() 发出
  |                               |  listen() 后等待
  |<-- SYN-ACK (seq=y,ack=x+1)--|  accept() 准备
  |                               |
  |--- ACK (ack=y+1) ---------->|  三次握手完成
  |                               |  accept() 返回新 conn
  |     连接建立，可收发数据       |

TCP 四次挥手：

主动关闭方                      被动关闭方
  |                               |
  |--- FIN ---------------------->|  close() 发出
  |<-- ACK ----------------------|
  |                               |
  |<-- FIN ----------------------|  close() 发出
  |--- ACK ---------------------->|
  |                               |
  |    TIME_WAIT (2MSL)          |  CLOSED
"""

# TIME_WAIT 状态
# 主动关闭方在发送最后一个 ACK 后进入 TIME_WAIT 状态
# 持续 2*MSL（Maximum Segment Lifetime，通常 60 秒）
# 期间该 (IP, port) 对不可重用（除非 SO_REUSEADDR）
server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
server.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)  # 允许重用
server.bind(('0.0.0.0', 8888))
```

### 46.2.3 I/O 多路复用

```python
import select
import socket

# select — 最基本的 I/O 多路复用（所有平台）
# 缺点：O(n) 复杂度，最大 1024 个 fd（FD_SETSIZE）
readable, writable, exceptional = select.select(
    [sock1, sock2],  # 读监视
    [sock3],         # 写监视
    [],              # 异常监视
    timeout=5.0      # 超时
)

# poll — 更高效的 I/O 多路复用（Unix only）
# 优点：无 fd 数量限制，O(n) 但更轻量
poller = select.poll()
poller.register(sock, select.POLLIN | select.POLLOUT)
events = poller.poll(timeout)  # 返回 [(fd, event), ...]

# epoll — Linux 高性能 I/O 多路复用
# 优点：O(1) 复杂度，无 fd 限制，只返回活跃的 fd
# select.epoll 仅在 Linux 上可用
try:
    ep = select.epoll()
    ep.register(sock.fileno(), select.EPOLLIN)
    events = ep.poll(timeout)  # [(fd, events), ...]
except AttributeError:
    # 非 Linux 平台
    pass

# kqueue — BSD/macOS 高性能 I/O 多路复用
try:
    kq = select.kqueue()
    # ...
except AttributeError:
    # 非 BSD/macOS 平台
    pass

# selectors 模块 — 平台无关的 I/O 多路复用封装
import selectors

sel = selectors.DefaultSelector()  # 自动选择最佳实现

sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
sock.bind(('0.0.0.0', 8888))
sock.listen()
sock.setblocking(False)

sel.register(sock, selectors.EVENT_READ, data=None)

while True:
    events = sel.select(timeout=None)
    for key, mask in events:
        if key.data is None:
            # 服务器 socket 有新连接
            conn, addr = key.fileobj.accept()
            conn.setblocking(False)
            sel.register(conn, selectors.EVENT_READ, data=True)
        else:
            # 客户端 socket 有数据
            data = key.fileobj.recv(1024)
            if data:
                key.fileobj.sendall(data.upper())
            else:
                sel.unregister(key.fileobj)
                key.fileobj.close()
```

### 46.2.5 缓冲区管理

```python
import socket

# TCP 的发送和接收缓冲区
sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)

# 查看缓冲区大小
recv_buf = sock.getsockopt(socket.SOL_SOCKET, socket.SO_RCVBUF)
send_buf = sock.getsockopt(socket.SOL_SOCKET, socket.SO_SNDBUF)
print(f"接收缓冲区: {recv_buf} bytes")
print(f"发送缓冲区: {send_buf} bytes")
# Linux 默认: 接收 87380, 发送 16384
# 实际值可能是设定值的 2 倍（内核额外分配）

# 粘包问题（TCP 特有）
# TCP 是字节流协议，不保留消息边界
# 发送方两次 send("hello") 和 ("world")
# 接收方可能一次 recv() 收到 "helloworld"

# 解决方案1：定长消息
def send_fixed(sock, data, length=1024):
    padded = data.ljust(length).encode('utf-8')
    sock.sendall(padded)

def recv_fixed(sock, length=1024):
    chunks = []
    received = 0
    while received < length:
        chunk = sock.recv(length - received)
        if not chunk:
            raise ConnectionError("Connection closed")
        chunks.append(chunk)
        received += len(chunk)
    return b''.join(chunks).decode('utf-8').strip()

# 解决方案2：长度前缀
import struct

def send_with_length(sock, data):
    encoded = data.encode('utf-8')
    length = len(encoded)
    header = struct.pack('!I', length)  # 4 字节大端序长度
    sock.sendall(header + encoded)

def recv_with_length(sock):
    header = recv_exact(sock, 4)
    length = struct.unpack('!I', header)[0]
    data = recv_exact(sock, length)
    return data.decode('utf-8')

def recv_exact(sock, n):
    """确保恰好接收 n 个字节。"""
    chunks = []
    received = 0
    while received < n:
        chunk = sock.recv(n - received)
        if not chunk:
            raise ConnectionError("Connection closed")
        chunks.append(chunk)
        received += len(chunk)
    return b''.join(chunks)
```

### 46.3.2 异步 HTTP 下载器

```python
"""使用 selectors 的异步 HTTP 下载器"""
import socket
import selectors
from collections import defaultdict


class AsyncDownloader:
    def __init__(self):
        self.sel = selectors.DefaultSelector()
        self.results = {}
        self.buffers = defaultdict(bytearray)

    def download(self, urls):
        for url in urls:
            host, path = self._parse_url(url)
            self._start_download(host, path, url)

        while len(self.results) < len(urls):
            events = self.sel.select(timeout=5.0)
            for key, mask in events:
                self._handle_event(key, mask)

        self.sel.close()
        return self.results

    def _parse_url(self, url):
        url = url.replace('http://', '').replace('https://', '')
        parts = url.split('/', 1)
        host = parts[0]
        path = '/' + parts[1] if len(parts) > 1 else '/'
        return host, path

    def _start_download(self, host, path, original_url):
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.setblocking(False)
        try:
            sock.connect((host, 80))
        except BlockingIOError:
            pass  # 非阻塞连接，等待完成

        request = f'GET {path} HTTP/1.1\r\nHost: {host}\r\nConnection: close\r\n\r\n'
        self.sel.register(sock, selectors.EVENT_WRITE,
                          data=(original_url, request.encode()))

    def _handle_event(self, key, mask):
        sock = key.fileobj
        original_url, request = key.data

        if mask & selectors.EVENT_WRITE:
            sock.sendall(request)
            self.sel.modify(sock, selectors.EVENT_READ,
                            data=original_url)
        elif mask & selectors.EVENT_READ:
            try:
                data = sock.recv(4096)
                if data:
                    self.buffers[original_url].extend(data)
                else:
                    self.results[original_url] = bytes(self.buffers[original_url])
                    self.sel.unregister(sock)
                    sock.close()
            except ConnectionResetError:
                self.sel.unregister(sock)
                sock.close()
```

### 46.3.3 UDP 聊天室

```python
"""UDP 聊天室服务器"""
import socket


class UDPChatServer:
    def __init__(self, host='0.0.0.0', port=9999):
        self.sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
        self.sock.bind((host, port))
        self.clients = {}  # addr -> username

    def run(self):
        print(f"Chat server on {self.sock.getsockname()}")
        while True:
            data, addr = self.sock.recvfrom(1024)
            message = data.decode('utf-8')

            if message.startswith('/join '):
                username = message[6:].strip()
                self.clients[addr] = username
                self.broadcast(f"{username} joined the chat", exclude=addr)
            elif message.startswith('/quit'):
                username = self.clients.pop(addr, 'Unknown')
                self.broadcast(f"{username} left the chat")
            else:
                username = self.clients.get(addr, 'Unknown')
                self.broadcast(f"{username}: {message}")

    def broadcast(self, message, exclude=None):
        data = message.encode('utf-8')
        for addr in self.clients:
            if addr != exclude:
                self.sock.sendto(data, addr)
```

### 46.4.1 多线程 Echo 服务器

```python
"""支持优雅关闭的多线程 TCP Echo 服务器"""
import socket
import threading
import signal
import sys


class ThreadedEchoServer:
    def __init__(self, host='0.0.0.0', port=8888, max_workers=10):
        self.host = host
        self.port = port
        self.max_workers = max_workers
        self.running = False
        self.server_socket = None
        self.active_connections = []
        self.lock = threading.Lock()

    def start(self):
        self.server_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        self.server_socket.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
        self.server_socket.bind((self.host, self.port))
        self.server_socket.listen(self.max_workers)
        self.server_socket.settimeout(1.0)  # 允许定期检查 running 标志
        self.running = True

        print(f"Server listening on {self.host}:{self.port}")

        try:
            while self.running:
                try:
                    conn, addr = self.server_socket.accept()
                    with self.lock:
                        self.active_connections.append(conn)
                    thread = threading.Thread(
                        target=self._handle_client,
                        args=(conn, addr),
                        daemon=True
                    )
                    thread.start()
                except socket.timeout:
                    continue
        except KeyboardInterrupt:
            pass
        finally:
            self.shutdown()

    def _handle_client(self, conn, addr):
        print(f"New connection from {addr}")
        conn.settimeout(30.0)
        try:
            while self.running:
                try:
                    data = conn.recv(1024)
                    if not data:
                        break
                    conn.sendall(data)
                except socket.timeout:
                    conn.sendall(b'ping')
        except (ConnectionResetError, BrokenPipeError):
            pass
        finally:
            conn.close()
            with self.lock:
                if conn in self.active_connections:
                    self.active_connections.remove(conn)
            print(f"Connection closed: {addr}")

    def shutdown(self):
        self.running = False
        with self.lock:
            for conn in self.active_connections:
                try:
                    conn.close()
                except:
                    pass
            self.active_connections.clear()
        if self.server_socket:
            self.server_socket.close()
        print("Server shut down")
```

### 46.4.2 WebSocket 基础实现

```python
"""简易 WebSocket 服务器（RFC 6455 基本帧协议）"""
import socket
import hashlib
import base64
import struct


def compute_websocket_accept(key):
    """计算 WebSocket Sec-WebSocket-Accept 值。"""
    GUID = '258EAFA5-E914-47DA-95CA-C5AB0DC85B11'
    sha1 = hashlib.sha1((key + GUID).encode()).digest()
    return base64.b64encode(sha1).decode()


def parse_websocket_frame(data):
    """解析 WebSocket 帧。"""
    if len(data) < 2:
        return None, None

    fin = (data[0] >> 7) & 1
    opcode = data[0] & 0x0F
    masked = (data[1] >> 7) & 1
    length = data[1] & 0x7F

    offset = 2
    if length == 126:
        length = struct.unpack('!H', data[2:4])[0]
        offset = 4
    elif length == 127:
        length = struct.unpack('!Q', data[2:10])[0]
        offset = 10

    if masked:
        mask_key = data[offset:offset + 4]
        offset += 4
        payload = bytearray(data[offset:offset + length])
        for i in range(length):
            payload[i] ^= mask_key[i % 4]
        payload = bytes(payload)
    else:
        payload = data[offset:offset + length]

    return opcode, payload


def build_websocket_frame(payload, opcode=0x1, mask=False):
    """构建 WebSocket 帧。"""
    frame = bytearray()
    frame.append(0x80 | opcode)  # FIN + opcode

    length = len(payload)
    if length <= 125:
        frame.append(0x80 if mask else 0 | length)
    elif length <= 65535:
        frame.append((0x80 if mask else 0) | 126)
        frame.extend(struct.pack('!H', length))
    else:
        frame.append((0x80 if mask else 0) | 127)
        frame.extend(struct.pack('!Q', length))

    if mask:
        mask_key = b'\x00\x00\x00\x00'  # 简化：使用全零掩码
        frame.extend(mask_key)
        masked = bytearray(payload)
        for i in range(len(masked)):
            masked[i] ^= mask_key[i % 4]
        frame.extend(masked)
    else:
        frame.extend(payload)

    return bytes(frame)


def websocket_handshake(sock):
    """处理 WebSocket 握手。"""
    request = b''
    while b'\r\n\r\n' not in request:
        request += sock.recv(1024)

    headers = {}
    for line in request.decode().split('\r\n')[1:]:
        if ':' in line:
            key, value = line.split(':', 1)
            headers[key.strip().lower()] = value.strip()

    key = headers.get('sec-websocket-key', '')
    accept = compute_websocket_accept(key)

    response = (
        'HTTP/1.1 101 Switching Protocols\r\n'
        'Upgrade: websocket\r\n'
        'Connection: Upgrade\r\n'
        f'Sec-WebSocket-Accept: {accept}\r\n\r\n'
    )
    sock.sendall(response.encode())
```

### 46.5.3 连接超时

```python
import socket

# 陷阱：connect() 可能长时间阻塞
# 在网络不可达时，默认 TCP 超时可达 2 分钟

# 最佳实践：设置超时
sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
sock.settimeout(5.0)
try:
    sock.connect(('unreachable.host', 80))
except socket.timeout:
    print("Connection timed out")
finally:
    sock.close()

# 更精确的超时控制（使用 select）
def connect_with_timeout(host, port, timeout=5.0):
    sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    sock.setblocking(False)
    try:
        sock.connect((host, port))
    except BlockingIOError:
        import select
        _, writable, _ = select.select([], [sock], [], timeout)
        if not writable:
            sock.close()
            raise socket.timeout("Connection timed out")
        err = sock.getsockopt(socket.SOL_SOCKET, socket.SO_ERROR)
        if err != 0:
            sock.close()
            raise ConnectionError(f"Connection failed: error {err}")
    sock.setblocking(True)
    return sock
```

### 46.5.4 缓冲区大小选择

```python
# 陷阱：recv(bufsize) 的 bufsize 不是最大消息长度
# 它是一次系统调用的最大读取量
# 消息可能跨越多次 recv()

# 最佳实践：
# - 对于小消息：1024 或 4096 足够
# - 对于大消息：使用循环读取 + 消息边界协议
# - bufsize 应为 2 的幂（对齐内存页）
# - 通常 8192 是一个好的默认值

BUFSIZE = 8192  # 8KB，适合大多数场景

# 接收完整消息的正确方式
def recv_message(sock, expected_length):
    """确保接收完整的消息。"""
    chunks = []
    received = 0
    while received < expected_length:
        chunk = sock.recv(min(BUFSIZE, expected_length - received))
        if not chunk:
            raise ConnectionError("Connection closed prematurely")
        chunks.append(chunk)
        received += len(chunk)
    return b''.join(chunks)
```

### 46.5.6 总结

| 陷阱 | 最佳实践 |
|------|---------|
| TIME_WAIT 阻止重启 | 设置 SO_REUSEADDR |
| TCP 粘包 | 使用长度前缀或分隔符协议 |
| connect 长时间阻塞 | 设置 timeout 或使用 select |
| recv 返回不完整数据 | 循环读取直到完整 |

| send 部分发送 | 使用 sendall 代替 send |
