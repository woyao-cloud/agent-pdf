# 第49章 日志系统

日志是生产环境可观测性的基石。本章从 logging 模块的 Logger/Handler/Formatter 架构、线程安全机制到日志传播原理，深入剖析 Python 日志系统的底层机制与生产级实践。

## 49.1 概念与语法

### 49.1.1 logging 模块架构

```python
import logging

# 基本使用
logging.basicConfig(level=logging.DEBUG)
logging.debug('debug message')
logging.info('info message')
logging.warning('warning message')
logging.error('error message')
logging.critical('critical message')

# 日志级别（从低到高）
# DEBUG    — 调试信息
# INFO     — 一般信息
# WARNING  — 警告（默认级别）
# ERROR    — 错误
# CRITICAL — 严重错误

print(logging.DEBUG)    # 10
print(logging.INFO)     # 20
print(logging.WARNING)  # 30
print(logging.ERROR)    # 40
print(logging.CRITICAL) # 50
```

**核心组件**：

```
Logger（记录器）
  │
  ├── Handler（处理器）— 决定日志输出到哪里
  │     ├── StreamHandler     — 控制台输出
  │     ├── FileHandler       — 文件输出
  │     ├── RotatingFileHandler — 按大小轮转
  │     ├── TimedRotatingFileHandler — 按时间轮转
  │     ├── SMTPHandler       — 邮件通知
  │     ├── SysLogHandler      — 系统日志
  │     └── HTTPHandler        — HTTP 发送
  │
  ├── Formatter（格式器）— 决定日志的格式
  │
  └── Filter（过滤器）— 决定哪些日志被处理
```

### 49.1.2 Logger 配置

```python
import logging

# 方式1：basicConfig（简单场景）
logging.basicConfig(
    level=logging.DEBUG,
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
    filename='app.log',
    filemode='a',  # 追加模式
)

# 方式2：dictConfig（推荐生产使用）
import logging.config

LOGGING_CONFIG = {
    'version': 1,
    'disable_existing_loggers': False,

    'formatters': {
        'standard': {
            'format': '%(asctime)s [%(levelname)s] %(name)s: %(message)s',
            'datefmt': '%Y-%m-%d %H:%M:%S',
        },
        'detailed': {
            'format': '%(asctime)s [%(levelname)s] %(name)s %(filename)s:%(lineno)d: %(message)s',
        },
    },

    'handlers': {
        'console': {
            'class': 'logging.StreamHandler',
            'level': 'DEBUG',
            'formatter': 'standard',
            'stream': 'ext://sys.stdout',
        },
        'file': {
            'class': 'logging.handlers.RotatingFileHandler',
            'level': 'INFO',
            'formatter': 'detailed',
            'filename': 'app.log',
            'maxBytes': 10485760,  # 10MB
            'backupCount': 5,
        },
    },

    'loggers': {
        'myapp': {
            'level': 'DEBUG',
            'handlers': ['console', 'file'],
            'propagate': True,
        },
    },

    'root': {
        'level': 'WARNING',
        'handlers': ['console'],
    },
}

logging.config.dictConfig(LOGGING_CONFIG)
logger = logging.getLogger('myapp')
logger.info('Application started')
```

### 49.1.4 日志轮转

```python
import logging
from logging.handlers import RotatingFileHandler, TimedRotatingFileHandler

# 按大小轮转
handler = RotatingFileHandler(
    'app.log',
    maxBytes=10 * 1024 * 1024,  # 10MB
    backupCount=5,               # 保留5个备份
    encoding='utf-8',
)
# 生成文件：app.log, app.log.1, app.log.2, ..., app.log.5

# 按时间轮转
handler = TimedRotatingFileHandler(
    'app.log',
    when='midnight',  # 每天午夜轮转
    interval=1,        # 间隔
    backupCount=30,    # 保留30天
    encoding='utf-8',
)
# when 选项：
# 'S' — 秒
# 'M' — 分钟
# 'H' — 小时
# 'D' — 天
# 'midnight' — 每天午夜
# 'W0'-'W6' — 每周一到周日
# 生成文件：app.log, app.log.2024-01-15, app.log.2024-01-14, ...
```

## 49.2 原理与机制

### 49.2.1 Logger 的层级与传播

```python
import logging

# Logger 的层级结构用点号分隔
# root
# ├── myapp
# │   ├── myapp.api
# │   │   └── myapp.api.views
# │   └── myapp.db
# └── other

# 获取 Logger
root = logging.getLogger()           # 根 Logger
app = logging.getLogger('myapp')     # myapp
api = logging.getLogger('myapp.api') # myapp.api

# 传播机制（propagate）
# 当 propagate=True（默认），日志消息会向上传播到父 Logger
# myapp.api 的日志 → myapp.api 的 Handler + myapp 的 Handler + root 的 Handler

# 禁止传播
api.propagate = False  # 日志只由 myapp.api 的 Handler 处理

# 日志消息的处理流程：
# 1. 检查 Logger 的 level，低于则忽略
# 2. 创建 LogRecord
# 3. 传递给 Logger 的所有 Handler
# 4. 如果 propagate=True，传递给父 Logger 的 Handler
# 5. 重复步骤 3-4 直到 root Logger

# 注意：子 Logger 的日志级别独立于父 Logger
# 即使父 Logger 的级别更高，子 Logger 仍会处理
root.setLevel(logging.WARNING)
app.setLevel(logging.DEBUG)
app.debug('This IS logged')  # myapp 的 DEBUG 级别允许
# 但 root 的 Handler 会根据 root 的 level 过滤
```

### 49.2.2 LogRecord 的生命周期

```python
import logging

# LogRecord 是日志消息的内部表示
# 包含所有日志相关的元数据

"""
class LogRecord:
    name: str          # Logger 名称
    msg: str           # 消息模板
    args: tuple        # 格式化参数
    created: float     # 创建时间戳
    levelno: int       # 级别编号
    levelname: str     # 级别名称
    pathname: str      # 源文件路径
    filename: str      # 源文件名
    module: str        # 模块名
    funcName: str      # 函数名
    lineno: int        # 行号
    process: int       # 进程 ID
    thread: int        # 线程 ID
    threadName: str    # 线程名
    exc_info: tuple    # 异常信息
    stack_info: str     # 堆栈信息
"""

# 延迟格式化（性能优化）
# 好：logger.debug('value: %s', expensive_call())  # 格式化在需要时才执行
# 差：logger.debug(f'value: {expensive_call()}')   # 总是执行格式化

# 使用 %s 格式化（推荐）而不是 f-string
# 因为如果日志级别过滤了消息，%s 不会执行格式化
```

### 49.2.3 线程安全机制

```python
import logging
import threading

# logging 模块的线程安全由 RLock 保证
# 每个 Handler 有一个 lock 属性（默认为 RLock）

# 关键源码（Lib/logging/__init__.py）：
"""
class Handler(Filterer):
    def __init__(self):
        self.lock = threading.RLock()

    def acquire(self):
        self.lock.acquire()

    def release(self):
        self.lock.release()

    def emit(self, record):
        # 子类必须实现，emit 内部需要获取锁
        with self.lock:
            # 写入日志
            pass

    def handle(self, record):
        rv = self.filter(record)
        if rv:
            self.acquire()
            try:
                self.emit(record)
            finally:
                self.release()
        return rv
"""

# 自定义 Handler 必须确保线程安全
class ThreadSafeHandler(logging.Handler):
    def emit(self, record):
        try:
            msg = self.format(record)
            # 原子写入操作
            with self.lock:
                self._write(msg)
        except Exception:
            self.handleError(record)
```

### 49.2.4 Handler.emit 流程

```python
import logging

# Handler.emit() 是日志写入的核心方法
# 以 FileHandler 为例：

"""
class FileHandler(StreamHandler):
    def emit(self, record):
        # 1. 如果文件未打开，尝试打开
        if self.stream is None:
            self.stream = self._open()

        # 2. 格式化记录
        try:
            msg = self.format(record)

            # 3. 写入流
            self.stream.write(msg + self.terminator)

            # 4. 刷新缓冲区
            self.flush()
        except Exception:
            self.handleError(record)
"""

# handleError 的默认行为：打印到 stderr，然后继续
# 可以自定义错误处理
class SilentHandler(logging.Handler):
    def handleError(self, record):
        pass  # 静默忽略写入错误
```

### 49.3.2 结构化 JSON 日志

```python
"""结构化 JSON 日志输出"""
import logging
import json
from datetime import datetime


class JSONFormatter(logging.Formatter):
    """将日志格式化为 JSON。"""

    def format(self, record):
        log_entry = {
            'timestamp': datetime.utcnow().isoformat() + 'Z',
            'level': record.levelname,
            'logger': record.name,
            'message': record.getMessage(),
            'module': record.module,
            'function': record.funcName,
            'line': record.lineno,
            'process': record.process,
            'thread': record.thread,
        }

        if record.exc_info:
            log_entry['exception'] = self.formatException(record.exc_info)

        if hasattr(record, 'extra_data'):
            log_entry['data'] = record.extra_data

        return json.dumps(log_entry, ensure_ascii=False)


# 使用
handler = logging.StreamHandler()
handler.setFormatter(JSONFormatter())

logger = logging.getLogger('myapp')
logger.addHandler(handler)
logger.setLevel(logging.DEBUG)

logger.info('User logged in', extra={'extra_data': {'user_id': 42}})
# {"timestamp":"2024-01-15T10:23:45.123456Z","level":"INFO","logger":"myapp",
#  "message":"User logged in","module":"...","function":"...","line":...,
#  "process":1234,"thread":5678,"data":{"user_id":42}}
```

### 49.3.3 自定义 Handler

```python
"""自定义日志 Handler 示例"""
import logging
import queue
import threading


class QueuedHandler(logging.Handler):
    """线程安全的队列日志 Handler，用于异步日志写入。"""

    def __init__(self, capacity=1000):
        super().__init__()
        self.queue = queue.Queue(capacity)
        self._stop_event = threading.Event()
        self._worker = threading.Thread(target=self._consume, daemon=True)
        self._worker.start()

    def emit(self, record):
        try:
            msg = self.format(record)
            self.queue.put_nowait(msg)
        except queue.Full:
            # 队列满时丢弃最旧的日志
            try:
                self.queue.get_nowait()
            except queue.Empty:
                pass
            self.queue.put_nowait(msg)

    def _consume(self):
        """后台线程消费日志。"""
        while not self._stop_event.is_set():
            try:
                msg = self.queue.get(timeout=0.1)
                # 这里可以替换为发送到远程日志服务
                print(msg, flush=True)
            except queue.Empty:
                continue

    def close(self):
        self._stop_event.set()
        self._worker.join(timeout=5)
        super().close()
```

### 49.4.1 日志中间件

```python
"""Web 应用日志中间件"""
import logging
import time
import uuid


class RequestLoggingMiddleware:
    """HTTP 请求日志中间件。"""

    def __init__(self, app, logger_name='access'):
        self.app = app
        self.logger = logging.getLogger(logger_name)

    def __call__(self, environ, start_response):
        request_id = str(uuid.uuid4())[:8]
        start_time = time.perf_counter()

        def custom_start_response(status, headers, exc_info=None):
            duration = time.perf_counter() - start_time
            self.logger.info(
                '%s %s %s %.3fms',
                request_id,
                environ.get('REQUEST_METHOD', '?'),
                environ.get('PATH_INFO', '/'),
                duration * 1000,
            )
            return start_response(status, headers, exc_info)

        return self.app(environ, custom_start_response)
```

## 49.5 常见陷阱与最佳实践

### 49.5.1 日志级别选择

```python
# 陷阱：滥用 WARNING 或 ERROR
# 最佳实践：
# DEBUG   — 开发调试信息（变量值、流程追踪）
# INFO    — 业务关键事件（用户登录、订单创建）
# WARNING — 可恢复的异常（配置缺失但有默认值、API 延迟）
# ERROR   — 需要关注的错误（数据库连接失败、第三方服务异常）
# CRITICAL— 系统级严重故障（磁盘满、内存耗尽）

# 反例
logger.warning('User clicked button')  # 太低级别，应为 INFO 或 DEBUG
logger.error('File not found, using default')  # 可恢复，应为 WARNING
```

### 49.5.2 异常日志的 exc_info

```python
import logging

logger = logging.getLogger(__name__)

# 陷阱：只记录异常消息，没有堆栈
try:
    result = dangerous_operation()
except Exception as e:
    logger.error(f'Operation failed: {e}')  # 只有消息，没有堆栈

# 最佳实践：使用 exc_info=True 记录完整堆栈
try:
    result = dangerous_operation()
except Exception as e:
    logger.error('Operation failed', exc_info=True)  # 包含完整堆栈

# 或使用 logger.exception()（自动包含 exc_info）
try:
    result = dangerous_operation()
except Exception as e:
    logger.exception('Operation failed')  # 等价于 error(msg, exc_info=True)
```

### 49.5.3 性能影响

```python
import logging

logger = logging.getLogger(__name__)

# 陷阱：f-string 总是执行格式化
# 即使日志级别过滤了消息，expensive() 仍会被调用
logger.debug(f'Result: {expensive_computation()}')

# 最佳实践：使用 % 格式化（延迟求值）
logger.debug('Result: %s', expensive_computation)
# expensive_computation 只在 DEBUG 级别启用时才调用

# 更好的做法：先检查级别
if logger.isEnabledFor(logging.DEBUG):
    result = expensive_computation()
    logger.debug('Result: %s', result)
```

### 49.5.4 内存泄漏

```python
# 陷阱：Logger 引用泄漏
# logging 模块内部维护所有 Logger 的字典
# _loggerDict = {}  # name -> Logger

# 如果动态创建大量 Logger 且不清理，会导致内存泄漏

# 最佳实践：
# 1. 使用模块级 Logger（通常每个模块一个）
# 2. 避免在循环中创建新 Logger
# 3. 不再需要的 Logger 应移除 Handler 并设为 disabled

def cleanup_logger(name):
    logger = logging.getLogger(name)
    for handler in logger.handlers[:]:
        handler.close()
        logger.removeHandler(handler)
    logger.disabled = True
```

### 49.5.5 总结

| 陷阱 | 最佳实践 |
|------|---------|
| 级别滥用 | 按语义选择级别，不要全用 WARNING |
| 异常无堆栈 | 使用 exc_info=True 或 logger.exception() |
| f-string 性能浪费 | 使用 % 格式化延迟求值 |
| basicConfig 被忽略 | basicConfig 只在首次调用时生效，确保先于其他 logging 调用 |
| Logger 内存泄漏 | 避免动态创建大量 Logger |
| 日志传播重复输出 | 设置 propagate=False 或精简 Handler |
| RotatingFileHandler 编码 | 指定 encoding='utf-8' |
