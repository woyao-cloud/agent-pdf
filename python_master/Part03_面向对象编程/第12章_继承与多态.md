# 第12章 继承与多态

继承（inheritance）与多态（polymorphism）是面向对象编程的两大支柱。Python 的继承机制远不止"子类复用父类代码"这么简单——它的 MRO 解析采用 C3 线性化算法，`super()` 的行为与 Java/C++ 中的 `super` 截然不同，而多态甚至不依赖继承关系。本章将从 CPython 的类型对象 C 源码出发，逐层剖析继承链的构建、MRO 的计算过程、`super()` 的代理机制、抽象基类的注册体系，以及 Python 独特的鸭子类型多态模型。

---

## 12.1 概念与语法

### 12.1.1 单继承

单继承是最基本的继承形式，子类只有一个直接父类：

```python
class Animal:
    """基类：动物"""
    def __init__(self, name: str):
        self.name = name

    def speak(self) -> str:
        return f"{self.name} makes a sound"

    def move(self) -> str:
        return f"{self.name} moves"

class Dog(Animal):
    """子类：狗，继承自 Animal"""
    def speak(self) -> str:
        # 方法覆盖（override）
        return f"{self.name} says Woof!"

    def fetch(self) -> str:
        # 子类新增方法
        return f"{self.name} fetches the ball"

# 使用示例
dog = Dog("Buddy")
print(dog.speak())   # Buddy says Woof! — 调用子类覆盖的方法
print(dog.move())    # Buddy moves — 调用继承自父类的方法
print(dog.fetch())   # Buddy fetches the ball — 调用子类新增的方法
print(isinstance(dog, Animal))  # True — dog 是 Animal 的实例
print(issubclass(Dog, Animal))  # True — Dog 是 Animal 的子类
```

子类可以通过 `super()` 调用父类的方法：

```python
class Cat(Animal):
    def __init__(self, name: str, indoor: bool):
        super().__init__(name)  # 调用父类的 __init__
        self.indoor = indoor

    def speak(self) -> str:
        return f"{self.name} says Meow!"

cat = Cat("Whiskers", indoor=True)
print(cat.name)    # Whiskers
print(cat.indoor)  # True
print(cat.speak()) # Whiskers says Meow!
```

### 12.1.2 多继承

Python 支持多继承——一个子类可以同时继承多个父类：

```python
class Loggable:
    """可记录日志的混入类"""
    def log(self, message: str) -> None:
        print(f"[LOG] {self.__class__.__name__}: {message}")

class Serializable:
    """可序列化的混入类"""
    def to_dict(self) -> dict:
        return {k: v for k, v in self.__dict__.items()
                if not k.startswith('_')}

class User(Loggable, Serializable):
    """用户类，同时继承日志和序列化能力"""
    def __init__(self, name: str, email: str):
        self.name = name
        self.email = email

    def greet(self) -> str:
        self.log(f"Greeting from {self.name}")
        return f"Hello, I'm {self.name}"

user = User("Alice", "alice@example.com")
print(user.greet())      # [LOG] User: Greeting from Alice\nHello, I'm Alice
print(user.to_dict())    # {'name': 'Alice', 'email': 'alice@example.com'}
```

多继承的语法形式与类型检查：

```python
# 多继承语法
class Child(Parent1, Parent2, Parent3):
    pass

# __bases__ 属性：查看直接父类
print(Child.__bases__)  # (<class 'Parent1'>, <class 'Parent2'>, <class 'Parent3'>)

# isinstance 和 issubclass 支持多继承检查
print(isinstance(user, Loggable))      # True
print(isinstance(user, Serializable))   # True
print(issubclass(User, (Loggable, Serializable)))  # True — 检查是否为元组中任一类的子类
```

### 12.1.3 基本类型检查

```python
class A: pass
class B(A): pass
class C(B): pass

# isinstance 检查实例关系
c = C()
print(isinstance(c, A))  # True — 沿继承链向上查找
print(isinstance(c, B))  # True
print(isinstance(c, C))  # True

# issubclass 检查类之间的关系
print(issubclass(C, A))  # True
print(issubclass(C, B))  # True
print(issubclass(A, C))  # False — 方向不可逆

# object 是所有类的根
print(issubclass(C, object))  # True
print(isinstance(c, object))   # True
```

## 12.2 原理与机制

### 12.2.1 继承链的 C 源码实现

在 CPython 中，每个类型对象由 `PyTypeObject` 结构体表示，继承关系通过 `tp_base` 字段维护：

```c
// CPython Include/cpython/object.h (简化)
typedef struct _typeobject {
    PyObject_VAR_HEAD
    const char *tp_name;             /* 类型名称 */
    Py_ssize_t tp_basicsize;         /* 实例基础大小 */
    Py_ssize_t tp_itemsize;          /* 可变部分元素大小 */
    // ... 大量函数指针 ...

    struct _typeobject *tp_base;     /* 直接基类指针 — 单继承的核心 */
    PyObject *tp_bases;              /* 基类元组 — 多继承的核心 */
    PyObject *tp_mro;                /* MRO 元组 — 方法解析顺序 */
    PyObject *tp_dict;               /* 类型字典 */
    // ... 更多字段 ...
} PyTypeObject;
```

当 Python 执行 `class Dog(Animal):` 时，CPython 的类型创建流程：

1. **收集基类信息**：`type.__new__` 从 `Animal` 获取 `tp_base`，构建 `tp_bases` 元组
2. **计算 MRO**：调用 `PyType_Ready()` 中嵌入的 C3 线性化算法，计算 `tp_mro`
3. **建立类型字典链**：子类的 `tp_dict` 通过 MRO 链接到各祖先的 `tp_dict`，形成属性查找路径

```c
// CPython Objects/typeobject.c — type_new_impl (简化)
static PyObject *
type_new_impl(PyTypeObject *metatype, PyObject *name,
              PyObject *bases, PyObject *orig_dict)
{
    // ... 参数校验 ...

    // 1. 收集基类，设置 tp_bases
    if (bases == NULL || PyTuple_GET_SIZE(bases) == 0) {
        // 无显式基类 → 默认继承 object
        bases = PyTuple_Pack(1, (PyObject *)&PyBaseObject_Type);
    }

    // 2. 计算方法解析顺序（MRO）
    //    调用 mro_internal() → C3 线性化
    PyObject *mro = type_compute_mro(type);

    // 3. 调用 PyType_Ready() 完成类型初始化
    PyType_Ready(type);

    return (PyObject *)type;
}
```

属性查找在 C 层的实现——沿 MRO 搜索字典：

```c
// CPython Objects/typeobject.c — _PyType_Lookup (简化)
PyObject *
_PyType_Lookup(PyTypeObject *type, PyObject *name)
{
    // 沿 MRO 搜索属性
    for (int i = 0; i < PyTuple_GET_SIZE(type->tp_mro); i++) {
        PyTypeObject *base = (PyTypeObject *)PyTuple_GET_ITEM(type->tp_mro, i);
        PyObject *dict = base->tp_dict;

        // 在当前基类的类型字典中查找
        PyObject *res = PyDict_GetNamedItem(dict, name);
        if (res != NULL) {
            return res;  // 找到即返回
        }
    }
    return NULL;  // 所有祖先类中均未找到
}
```

### 12.2.2 MRO 与 C3 线性化算法

MRO（Method Resolution Order，方法解析顺序）决定了属性和方法查找时的类搜索顺序。Python 3 采用 C3 线性化算法计算 MRO。

#### 菱形继承问题

```
       A
      / \
     B   C
      \ /
       D
```

```python
class A:
    def who(self):
        return "A"

class B(A):
    def who(self):
        return "B"

class C(A):
    def who(self):
        return "C"

class D(B, C):
    pass

d = D()
print(d.who())  # B — MRO 决定了查找顺序
print(D.__mro__)
# (<class '__main__.D'>, <class '__main__.B'>,
#  <class '__main__.C'>, <class '__main__.A'>, <class 'object'>)
```

#### C3 线性化的完整算法

C3 纗性化是 Keith Knight 提出的算法，其核心原则：

1. **单调性**：如果 C3 在某继承链中 X 排在 Y 之前，那么在所有包含 X 和 Y 的子类 MRO 中，X 始终在 Y 之前
2. **局部优先**：子类声明中父类的顺序决定了优先级（`class D(B, C)` 意味着 B 优先于 C）
3. **一致性**：不存在矛盾的前后关系

**形式化定义**：

设类 `C` 的直接父类为 `C1, C2, ..., Cn`，则 `C` 的 MRO 为：

```
L(C) = C + merge(L(C1), L(C2), ..., L(Cn), [C1, C2, ..., Cn])
```

`merge` 操作的规则：

1. 取所有列表的第一个元素组成候选集
2. 从中选取一个"好头"（good head）：该元素不出现在任何其他列表的尾部（非首位位置）
3. 将"好头"加入结果序列，并从所有列表中移除该元素
4. 重复步骤 1-3，直到所有列表为空；若无"好头"可选，则报 TypeError

#### 手动计算 C3 示例

对于菱形继承 `class D(B, C)`，其中 `B(A)`, `C(A)`：

```
L(A) = A + merge(L(object), [object])
     = A + merge([object], [object])
     = [A, object]

L(B) = B + merge(L(A), [A])
     = B + merge([A, object], [A])
     = [B] + [A, object]      — A 是好头，移除 A 后 object 是好头
     = [B, A, object]

L(C) = C + merge(L(A), [A])
     = [C, A, object]

L(D) = D + merge(L(B), L(C), [B, C])
     = D + merge([B, A, object], [C, A, object], [B, C])

步骤1: 候选头 = B, C, B。检查 B：B 不在 [C, A, object] 的尾部 → B 是好头
     = [D, B] + merge([A, object], [C, A, object], [C])

步骤2: 候选头 = A, C, C。检查 A：A 在 [C, A, object] 的尾部 → A 不是好头
                              检查 C：C 不在 [A, object] 的尾部 → C 是好头
     = [D, B, C] + merge([A, object], [A, object])

步骤3: 候选头 = A, A。检查 A：A 不在任何列表的尾部 → A 是好头
     = [D, B, C, A] + merge([object], [object])

步骤4: object 是好头
     = [D, B, C, A, object]
```

这与 Python 实际输出完全一致：

```python
print(D.__mro__)
# (<class 'D'>, <class 'B'>, <class 'C'>, <class 'A'>, <class 'object'>)
```

#### 不可线性化的继承报错

当继承关系存在矛盾时，C3 线性化会抛出 `TypeError`：

```python
class X:
    def method(self):
        return "X"

class Y:
    def method(self):
        return "Y"

# 尝试创建矛盾的继承顺序
try:
    class Bad(X, Y):
        pass

    class Worse(Y, X):
        pass

    class Impossible(Bad, Worse):
        pass
        # TypeError: Cannot create a consistent method resolution
        # order (MRO) for bases X, Y
except TypeError as e:
    print(f"MRO 冲突: {e}")
```

CPython 中 MRO 计算失败的检测代码：

```c
// CPython Objects/typeobject.c — mro_internal (简化)
static int
mro_internal(PyTypeObject *type)
{
    PyObject *mro = type_compute_mro(type);
    if (mro == NULL) {
        // mro_thermal() 返回 NULL → 不可线性化
        PyErr_Format(PyExc_TypeError,
                     "Cannot create a consistent method resolution "
                     "order (MRO) for bases %.100s",
                     /* ... */);
        return -1;
    }
    type->tp_mro = mro;  // 缓存到类型对象
    return 0;
}
```

### 12.2.3 super() 的真实机制

#### super() 不是调用父类

初学者最常见的误解：`super()` 调用的是"父类"。实际上，`super()` 是一个代理对象（proxy object），它根据 MRO 找到当前类在 MRO 序列中的**下一个**类，而不是直接父类。

```python
class A:
    def method(self):
        print(f"A.method called on {self.__class__.__name__}")

class B(A):
    def method(self):
        print(f"B.method called on {self.__class__.__name__}")
        super().method()  # 调用 MRO 中 B 的下一个类

class C(A):
    def method(self):
        print(f"C.method called on {self.__class__.__name__}")
        super().method()  # 调用 MRO 中 C 的下一个类

class D(B, C):
    def method(self):
        print(f"D.method called on {self.__class__.__name__}")
        super().method()  # 调用 MRO 中 D 的下一个类

d = D()
d.method()
# 输出：
# D.method called on D
# B.method called on D  ← super() 在 B 中调用的不是 A，而是 C！
# C.method called on D  ← super() 在 C 中调用的不是 A，而是 A
# A.method called on D

# 理解：D 的 MRO 是 [D, B, C, A, object]
# D.method() 中的 super() → 调用 B.method()
# B.method() 中的 super() → 调用 C.method()（不是 A！）
# C.method() 中的 super() → 调用 A.method()
```

#### super(B, self) 的工作机制

`super()` 的完整签名是 `super(type, obj)`：

- `type`：MRO 中起始搜索位置的类
- `obj`：提供 MRO 的实例（或类，当第二个参数是类时）

```python
# super() 的两种用法
class Example(Base):
    def method(self):
        # 以下两种写法等价（Python 3 中）
        super().method()            # 自动推断 type=Example, obj=self
        super(Example, self).method()  # 显式指定

# super(type, obj) 的查找逻辑：
# 1. 获取 obj.__class__.__mro__（或 obj.__mro__，如果 obj 是类）
# 2. 在 MRO 中找到 type 的位置
# 3. 从该位置的下一个类开始搜索属性
```

CPython 中 `super()` 的实现源码：

```c
// CPython Objects/typeobject.c — super_getattro (简化)
static PyObject *
super_getattro(PySuperObject *self, PyObject *name)
{
    // 如果没有设置 self->obj，走普通属性查找
    if (self->obj == NULL) {
        return PyObject_GenericGetAttr((PyObject *)self, name);
    }

    // 获取 self->obj 的 MRO
    PyObject *mro = PyType_Type.tp_mro;  // 实际从 self->type 获取

    // 在 MRO 中找到 self->type 的位置，从下一个开始搜索
    int start = -1;
    for (int i = 0; i < PyTuple_GET_SIZE(mro); i++) {
        if (PyTuple_GET_ITEM(mro, i) == (PyObject *)self->type) {
            start = i + 1;  // 从下一个类开始！
            break;
        }
    }

    // 从 start 位置沿 MRO 搜索属性
    for (int i = start; i < PyTuple_GET_SIZE(mro); i++) {
        PyTypeObject *cls = (PyTypeObject *)PyTuple_GET_ITEM(mro, i);
        PyObject *res = PyDict_GetItem(cls->tp_dict, name);
        if (res != NULL) {
            // 找到后，如果是一个描述符，需要绑定
            return res;
        }
    }
    // 未找到 → AttributeError
}
```

`super` 对象的结构体定义：

```c
// CPython Include/cpython/object.h
typedef struct {
    PyObject_HEAD
    PyTypeObject *type;    // 搜索的起始类（super 的第一个参数）
    PyObject *obj;         // 提供 MRO 的对象（super 的第二个参数）
    PyTypeObject *obj_type;  // 缓存的 obj 的类型，避免重复查找
} PySuperObject;
```

#### super() 的边界情况

```python
# 边界1：super() 与 __class__ 闭包变量
class Parent:
    def show_class(self):
        # Python 3 中，方法内部有一个隐式的 __class__ 闭包变量
        print(f"__class__ is {__class__.__name__}")

class Child(Parent):
    pass

Child().show_class()  # __class__ is Parent（不是 Child！）
# 注意：__class__ 指向定义方法的类，不是调用者的类

# 边界2：super() 的第二个参数是类时
class Base:
    @classmethod
    def create(cls):
        return cls()

class Derived(Base):
    @classmethod
    def create(cls):
        # super(Derived, cls) — 第二个参数是类，不是实例
        # 此时查找使用 cls.__mro__（Derived 的 MRO），
        # 而非 Derived.__mro__（等价，但语义不同）
        return super(Derived, cls).create()

# 边界3：super() 在 __init__ 中必须链式调用
class A:
    def __init__(self):
        print("A.__init__")

class B(A):
    def __init__(self):
        print("B.__init__")
        super().__init__()  # 必须调用，否则 A.__init__ 不会执行

class C(A):
    def __init__(self):
        print("C.__init__")
        super().__init__()  # 必须调用

class D(B, C):
    def __init__(self):
        print("D.__init__")
        super().__init__()  # 触发完整的协作式初始化链

D()
# D.__init__
# B.__init__
# C.__init__  ← 注意：不是 A.__init__！B 的 super() 调用的是 C
# A.__init__
```

### 12.2.4 抽象基类（ABC）的实现原理

Python 的 `abc` 模块通过元类 `ABCMeta` 和描述符协议实现抽象基类。其 C 层实现涉及多个协作组件：

```c
// CPython Modules/_abc.c — 核心数据结构
typedef struct {
    PyObject_HEAD
    PyObject *abc_data;       /* 抽象方法集合 */
    uint64_t abc_version;     /* 版本号，用于缓存失效 */
} _abc_data;

// 抽象基类检查的缓存机制
// PyTypeObject 中有 tp_flags: Py_TPFLAGS_HAVE_ABC
// 以及 tp_abc 指针，缓存 issubclass/issubclass 的结果
```

#### @abstractmethod 的实现

`@abstractmethod` 是一个描述符，它标记方法为"必须被覆盖"：

```python
from abc import ABC, abstractmethod

class Shape(ABC):
    @abstractmethod
    def area(self) -> float:
        """计算面积 — 子类必须实现"""
        ...

    @abstractmethod
    def perimeter(self) -> float:
        """计算周长 — 子类必须实现"""
        ...

    def describe(self) -> str:
        """非抽象方法 — 子类可直接继承"""
        return f"{self.__class__.__name__}: area={self.area()}, perimeter={self.perimeter()}"
```

`@abstractmethod` 的工作机制：

1. **标记阶段**：`@abstractmethod` 装饰器将函数对象的 `__isabstractmethod__` 属性设为 `True`
2. **收集阶段**：`ABCMeta.__new__` 在创建类时扫描命名空间，收集所有 `__isabstractmethod__ == True` 的方法名到 `__abstractmethods__` 集合
3. **检查阶段**：当尝试实例化类时，`ABCMeta.__call__` 检查 `__abstractmethods__` 是否为空；不为空则抛出 `TypeError`

```python
# 模拟 @abstractmethod 的核心实现
class AbstractMethodDescriptor:
    """简化版的 @abstractmethod 实现"""
    def __init__(self, func):
        self.func = func
        self.__isabstractmethod__ = True  # 关键标记

    def __set_name__(self, owner, name):
        # 在类创建时被调用
        pass

    def __get__(self, obj, objtype=None):
        if obj is None:
            return self
        return self.func.__get__(obj, objtype)

# 验证 __abstractmethods__ 的收集
class MyABC(ABC):
    @abstractmethod
    def must_implement(self):
        ...

    def optional_method(self):
        ...

print(MyABC.__abstractmethods__)
# frozenset({'must_implement'})

# 尝试实例化抽象类
try:
    shape = Shape()
except TypeError as e:
    print(f"不能实例化抽象类: {e}")
    # Can't instantiate abstract class Shape with abstract methods area, perimeter
```

#### 虚拟子类与注册机制

ABC 最强大的特性之一是"虚拟子类"——不通过继承，而是通过注册（register）建立 `isinstance`/`issubclass` 关系：

```python
from abc import ABCMeta

class MyIterable(ABC):
    @abstractmethod
    def __iter__(self):
        ...

# 注册虚拟子类
class CustomContainer:
    """没有继承 MyIterable，但注册为虚拟子类"""
    def __iter__(self):
        return iter([1, 2, 3])

MyIterable.register(CustomContainer)

print(issubclass(CustomContainer, MyIterable))  # True！
print(isinstance(CustomContainer(), MyIterable))  # True！
print(CustomContainer.__mro__)
# (<class 'CustomContainer'>, <class 'object'>)
# 注意：MRO 中没有 MyIterable，注册不影响方法解析！
```

注册机制的 C 层实现原理：

```c
// CPython Modules/_abc.c — _abc_register (简化)
static PyObject *
_abc_register(PyABCObject *self, PyObject *subclass)
{
    // 1. 将 subclass 添加到 self->_abc_registry 集合
    // 2. 递增 self->abc_version，使所有缓存失效
    // 3. 返回 subclass（允许作为装饰器使用）

    if (PySet_Add(self->_abc_registry, subclass) < 0)
        return NULL;

    self->abc_version++;  // 版本号递增，缓存失效
    Py_RETURN_NONE;
}
```

`isinstance` 对虚拟子类的检查逻辑：

```c
// CPython Objects/abstract.c — abstract_isinstance (简化)
static int
abstract_isinstance(PyObject *inst, PyObject *cls)
{
    // 1. 首先尝试正常的 isinstance 检查（沿 MRO）
    int result = PyObject_IsInstance(inst, cls);
    if (result != 0)
        return result;  // 正常继承关系找到，或出错

    // 2. 检查 cls 是否有 __subclasshook__
    PyObject *subclass_hook = PyObject_GetAttrString(cls, "__subclasshook__");
    if (subclass_hook != NULL) {
        // 调用 __subclasshook__(inst, cls)
        result = _abc_subclasscheck(cls, inst);
    }

    // 3. 检查 _abc_registry（虚拟子类注册表）
    // 4. 检查 _abc_cache（缓存加速）
    return result;
}
```

#### __subclasshook__ 协议

`__subclasshook__` 允许自定义 `issubclass` 的行为，这是 ABC 实现"鸭子类型检查"的核心：

```python
from abc import ABCMeta

class Sized(ABC):
    """自定义子类检查协议"""
    @abstractmethod
    def __len__(self):
        ...

    @classmethod
    def __subclasshook__(cls, subclass):
        """只要实现了 __len__，就视为 Sized 的虚拟子类"""
        if cls is Sized:
            if hasattr(subclass, '__len__'):
                return True   # 确认是子类
            return False      # 确认不是子类
        return NotImplemented  # 交给其他钩子处理

# 任何实现了 __len__ 的类都被视为 Sized
class MyContainer:
    def __len__(self):
        return 42

print(isinstance(MyContainer(), Sized))   # True！没有继承关系
print(issubclass(MyContainer, Sized))      # True！
```

### 12.2.5 协议与鸭子类型的运行时机制

Python 的多态不依赖继承声明，而是基于"协议"（protocol）。协议是一个类只要实现了特定方法就被视为满足某种接口。

属性查找在 C 层的完整流程：

```c
// CPython Objects/object.c — PyObject_GetAttr (简化)
PyObject *
PyObject_GetAttr(PyObject *v, PyObject *name)
{
    PyTypeObject *tp = Py_TYPE(v);

    // 步骤1：在类型对象中查找（沿 MRO）
    // 优先查找数据描述符（定义了 __get__ 和 __set__ 的描述符）
    PyObject *descr = _PyType_Lookup(tp, name);
    if (descr != NULL) {
        descrgetfunc f = Py_TYPE(descr)->tp_descr_get;
        if (f != NULL && PyDescr_IsData(descr)) {
            // 数据描述符优先级最高
            return f(descr, v, (PyObject *)tp);
        }
    }

    // 步骤2：在实例字典中查找
    PyObject *dict = _PyObject_GetDictPtr(v);
    if (dict != NULL) {
        PyObject *res = PyDict_GetItem(dict, name);
        if (res != NULL) {
            Py_INCREF(res);
            return res;
        }
    }

    // 步骤3：非数据描述符或普通方法（从 MRO 找到的非数据描述符）
    if (descr != NULL) {
        descrgetfunc f = Py_TYPE(descr)->tp_descr_get;
        if (f != NULL) {
            return f(descr, v, (PyObject *)tp);
        }
        return descr;
    }

    // 步骤4：__getattr__ 钩子
    if (tp->tp_getattr != NULL) {
        return tp->tp_getattr(v, name_str);
    }

    // 步骤5：AttributeError
    PyErr_Format(PyExc_AttributeError, ...);
}
```

这意味着 Python 的"鸭子类型"在 C 层就是属性查找——只要对象拥有所需的方法（即属性查找成功），就可以调用，无需关心继承关系。

### 12.3.3 抽象基类：接口契约与类型检查

```python
from abc import ABC, abstractmethod
from typing import Sequence

class DataSource(ABC):
    """数据源抽象基类 — 定义所有数据源必须实现的接口"""

    @abstractmethod
    def connect(self) -> None:
        """建立连接"""
        ...

    @abstractmethod
    def read(self, query: str) -> Sequence[dict]:
        """读取数据"""
        ...

    @abstractmethod
    def write(self, data: Sequence[dict]) -> int:
        """写入数据，返回写入条数"""
        ...

    @abstractmethod
    def disconnect(self) -> None:
        """断开连接"""
        ...

    def execute(self, query: str) -> Sequence[dict]:
        """模板方法 — 使用抽象方法定义的骨架"""
        self.connect()
        try:
            result = self.read(query)
            return result
        finally:
            self.disconnect()

class CSVDataSource(DataSource):
    """CSV 文件数据源"""
    def __init__(self, filepath: str):
        self.filepath = filepath
        self._file = None

    def connect(self) -> None:
        self._file = open(self.filepath, 'r', encoding='utf-8')

    def read(self, query: str) -> Sequence[dict]:
        import csv
        if self._file is None:
            raise RuntimeError("Not connected")
        reader = csv.DictReader(self._file)
        return list(reader)

    def write(self, data: Sequence[dict]) -> int:
        import csv
        with open(self.filepath, 'w', encoding='utf-8', newline='') as f:
            if data:
                writer = csv.DictWriter(f, fieldnames=data[0].keys())
                writer.writeheader()
                writer.writerows(data)
                return len(data)
            return 0

    def disconnect(self) -> None:
        if self._file:
            self._file.close()
            self._file = None

# 使用
csv_ds = CSVDataSource("data.csv")
# 如果缺少任何抽象方法的实现，实例化时会报 TypeError
```

```python
from collections.abc import Sequence, Mapping
import math

    def area(self) -> float:
        """计算面积"""
        ...

    def perimeter(self) -> float:

        ...

    @property
    def name(self) -> str:
        return "Circle"

class Rectangle(Shape):
    def __init__(self, width: float, height: float):
        self.width = width
        self.height = height

    def area(self) -> float:
        return self.width * self.height

    def perimeter(self) -> float:
        return 2 * (self.width + self.height)

    @property
    def name(self) -> str:
        return "Rectangle"

# 尝试实例化抽象类

    Shape()  # TypeError
except TypeError as e:

# 尝试部分实现

try:

except TypeError as e:

print(circle.describe())  # Circle: area=78.54, perimeter=31.42

# ========== 2. 虚拟子类注册 ==========

class MyABC(ABC):
    @abstractmethod
    def required_method(self):
        ...

# 注册虚拟子类方式1：调用 register()
class ExternalClass:
    def required_method(self):
        return "external implementation"

MyABC.register(ExternalClass)

print(f"ExternalClass 是 MyABC 的子类? {issubclass(ExternalClass, MyABC)}")  # True
print(f"ExternalClass 实例是 MyABC 的实例? {isinstance(ExternalClass(), MyABC)}")  # True
print(f"ExternalClass.__mro__ 包含 MyABC? {MyABC in ExternalClass.__mro__}")  # False!

@MyABC.register
class AnotherExternal:
    def required_method(self):
        return "another implementation"

print(f"AnotherExternal 是 MyABC 的子类? {issubclass(AnotherExternal, MyABC)}")  # True

# ========== 3. __subclasshook__ 自定义子类检查 ==========

class Drawable(ABC):
    """任何实现了 draw 方法的类都被视为 Drawable"""
    @abstractmethod
    def draw(self):
        ...

    @classmethod
    def __subclasshook__(cls, C):
        if cls is Drawable:
            if hasattr(C, 'draw'):
                return True
            return False
        return NotImplemented

class Canvas:
    """没有继承 Drawable，但实现了 draw 方法"""
    def draw(self):

print(f"Canvas 实例是 Drawable 的实例? {isinstance(Canvas(), Drawable)}")  # True

# 注意：虚拟子类不能调用 Drawable 上的具体方法（因为没有在 MRO 中）
# 但 isinstance/issubclass 检查通过

        return 0
```

### 12.4.3 继承 vs 组合 vs 委托完整对比

```python
"""
三种代码复用方式的对比：继承、组合、委托
"""
from abc import ABC, abstractmethod

# ========== 方式1：继承 ==========
class VehicleBase:
    def __init__(self, brand: str):
        self.brand = brand

    def start_engine(self) -> str:
        return f"{self.brand} engine started"

    def stop_engine(self) -> str:
        return f"{self.brand} engine stopped"

class CarByInheritance(VehicleBase):
    def __init__(self, brand: str, seats: int):
        super().__init__(brand)
        self.seats = seats

    def drive(self) -> str:
        return f"Driving {self.brand} with {self.seats} seats"

# ========== 方式2：组合 ==========
class Engine:
    def start(self) -> str:
        return "Engine started"

    def stop(self) -> str:
        return "Engine stopped"

class GPS:
    def navigate(self, destination: str) -> str:
        return f"Navigating to {destination}"

class CarByComposition:
    """通过组合获得功能，而非继承"""
    def __init__(self, brand: str, seats: int):
        self.brand = brand
        self.seats = seats
        self.engine = Engine()  # 组合：has-a 关系
        self.gps = GPS()        # 组合

    def start_engine(self) -> str:
        return f"{self.brand}: {self.engine.start()}"

    def stop_engine(self) -> str:
        return f"{self.brand}: {self.engine.stop()}"

    def navigate(self, destination: str) -> str:
        return f"{self.brand}: {self.gps.navigate(destination)}"

    def drive(self) -> str:
        return f"Driving {self.brand} with {self.seats} seats"

# ========== 方式3：委托（使用 __getattr__）==========
class CarDelegate:
    """通过 __getattr__ 自动委托方法调用"""
    def __init__(self, brand: str, seats: int):
        self.brand = brand
        self.seats = seats
        self._engine = Engine()
        self._gps = GPS()

    def drive(self) -> str:
        return f"Driving {self.brand} with {self.seats} seats"

    def __getattr__(self, name):
        """自动委托到组合对象"""
        # 先尝试引擎
        if hasattr(self._engine, name):
            return getattr(self._engine, name)
        # 再尝试 GPS
        if hasattr(self._gps, name):
            return getattr(self._gps, name)
        raise AttributeError(
            f"'{self.__class__.__name__}' object has no attribute '{name}'"
        )

# ========== 对比测试 ==========
print("=== 继承 ===")
car1 = CarByInheritance("Toyota", 5)
print(car1.start_engine())
print(car1.drive())

print("\n=== 组合 ===")
car2 = CarByComposition("Honda", 7)
print(car2.start_engine())
print(car2.navigate("Beijing"))
print(car2.drive())

print("\n=== 委托 ===")
car3 = CarDelegate("Ford", 4)
print(car3.start())       # 委托到 Engine.start()
print(car3.navigate("Shanghai"))  # 委托到 GPS.navigate()
print(car3.drive())

# 优缺点总结
print("\n=== 对比总结 ===")
print("继承: 紧耦合，子类依赖父类实现，修改父类影响所有子类")
print("组合: 松耦合，可运行时替换组件，但需要更多委托代码")
print("委托: 最灵活，__getattr__ 自动转发，但调试困难，IDE补全失效")
```

### 12.4.4 协议与鸭子类型实战

```python
"""
演示 Python 的协议多态：不依赖继承的接口兼容
"""
from typing import Protocol, runtime_checkable

# ========== 1. 基础鸭子类型 ==========

def process_iterable(items):
    """纯鸭子类型 — 不做任何类型检查"""
    result = []
    for item in items:
        result.append(str(item).upper())
    return result

# 任何可迭代对象都能用
print(process_iterable([1, 2, 3]))        # ['1', '2', '3']
print(process_iterable("abc"))              # ['A', 'B', 'C']
print(process_iterable(range(3)))           # ['0', '1', '2']

class CustomIterable:
    def __iter__(self):
        return iter(["x", "y", "z"])

print(process_iterable(CustomIterable()))  # ['X', 'Y', 'Z']

# ========== 2. 使用 Protocol 进行静态类型检查 ==========

@runtime_checkable
class Drawable(Protocol):
    """绘制协议 — 定义 draw 和 resize 方法"""
    def draw(self) -> str: ...
    def resize(self, factor: float) -> None: ...

class SVGElement:
    """不需要继承 Drawable，只要方法签名匹配"""
    def draw(self) -> str:
        return "<svg>...</svg>"

    def resize(self, factor: float) -> None:
        print(f"Resizing SVG by {factor}")

class CanvasElement:
    def draw(self) -> str:
        return "Canvas rendering"

    def resize(self, factor: float) -> None:
```

```python
"""
Mixin 设计模式：正确与错误的用法

# ========== 正确的 Mixin 设计 ==========

class LoggingMixin:
    """日志混入 — 提供日志能力，不定义 __init__"""
    _log_level = "INFO"

    def log(self, message: str, level: str = None):
        level = level or self._log_level
        print(f"[{level}] {self.__class__.__name__}: {message}")

    def log_error(self, message: str):
        self.log(message, level="ERROR")

    def log_debug(self, message: str):
        self.log(message, level="DEBUG")


class SerializationMixin:
    """序列化混入 — 提供 to_dict/from_dict 能力"""
    def to_dict(self) -> dict:
        """将实例属性转为字典（排除私有属性）"""
        return {
            k: v for k, v in self.__dict__.items()
            if not k.startswith('_')
        }

    @classmethod
    def from_dict(cls, data: dict):
        """从字典创建实例"""
        return cls(**data)


class ValidationMixin:
    """校验混入 — 提供字段校验能力"""
    _validators: dict = {}  # 子类覆盖此字段

    def validate(self) -> list[str]:
        """执行所有校验，返回错误列表"""
        errors = []
        for field, validator in self._validators.items():
            value = getattr(self, field, None)
            try:
                if not validator(value):
                    errors.append(f"Invalid {field}: {value}")
            except Exception as e:
                errors.append(f"Validation error for {field}: {e}")
        return errors


class User(LoggingMixin, SerializationMixin, ValidationMixin):
    """用户类 — 组合多个 Mixin"""
    _validators = {
        'name': lambda v: isinstance(v, str) and len(v) >= 2,
        'email': lambda v: isinstance(v, str) and '@' in v,
        'age': lambda v: isinstance(v, int) and 0 <= v <= 150,
    }

    def __init__(self, name: str, email: str, age: int):
        self.name = name
        self.email = email
        self.age = age

    def greet(self):
        self.log(f"User {self.name} is greeting")
        return f"Hello, I'm {self.name}"


# 使用
user = User("Alice", "alice@example.com", 30)
print(user.greet())              # [INFO] User: User Alice is greeting\nHello, I'm Alice
print(user.validate())           # [] — 无错误
print(user.to_dict())            # {'name': 'Alice', 'email': 'alice@example.com', 'age': 30}

bad_user = User("", "invalid", -1)
print(bad_user.validate())       # 有校验错误

# ========== 错误的 Mixin 设计 ==========

class BadMixin:
    """反面教材：Mixin 定义了 __init__"""
    def __init__(self, **kwargs):
        self.mixin_data = kwargs.pop('mixin_data', None)
        super().__init__(**kwargs)  # 依赖 super() 的 MRO 顺序

class AnotherBadMixin:
    """反面教材：Mixin 与特定类耦合"""
    def save(self):

        self.db.save(self)  # 如果没有 db 属性就会报错

# 1: LoggingMixin
# 2: SerializationMixin
# 3: ValidationMixin
# 4: object
```

### 12.5.2 陷阱：__init__ 未链式调用导致初始化缺失

```python
    def __init__(self):
        self.base_data = "initialized"
        print("Base.__init__")

class Middle(Base):
    def __init__(self):
        # 忘记调用 super().__init__()！

        print("Middle.__init__")

    def __init__(self):
        super().__init__()
        self.derived_data = "initialized"
        print("Derived.__init__")

        super().__init__(**kwargs)  # 传递剩余参数

class GoodMiddle(GoodBase):
```

### 12.5.3 陷阱：深度继承链导致的脆弱基类问题

```python
# 陷阱：深度继承链（Fragile Base Class Problem）
class Animal:
    def move(self):
        return "moving"

    def travel(self):
        return self.move()  # 依赖 self.move()

class Mammal(Animal):
    def move(self):
        return "walking"

class Dog(Mammal):
    def move(self):
        return "running"

class GuideDog(Dog):
    def travel(self):
        # 覆盖了 travel，但可能忘记调用 super().travel()
        return "guided walking"

# 问题1：GuideDog.travel() 完全覆盖了父类的 move 逻辑
# 问题2：修改 Animal.travel() 可能破坏所有子类
# 问题3：继承链越长，维护成本越高

    """行为策略 — 可替换"""
    def move(self) -> str:
        return "moving"

class WalkingBehavior(MovementBehavior):
    def move(self) -> str:
        return "walking"

    def move(self) -> str:
        return "running"

class GuidedBehavior(MovementBehavior):
    def move(self) -> str:
        return "guided walking"

class BetterAnimal:
    """组合而非继承"""
    def __init__(self, movement: MovementBehavior):
        self.movement = movement

    def travel(self) -> str:
        return self.movement.move()

# 运行时可以灵活替换行为
```

### 12.5.4 陷阱：Mixin 的 __init__ 冲突

```python
# 陷阱：多个 Mixin 都定义了 __init__，参数传递混乱
class MixinA:
    def __init__(self, a_param=None, **kwargs):
        self.a_data = a_param
        super().__init__(**kwargs)

class MixinB:
    def __init__(self, b_param=None, **kwargs):
        self.b_data = b_param
        super().__init__(**kwargs)

class MixinC:
    def __init__(self, c_param=None, **kwargs):
        self.c_data = c_param
        super().__init__(**kwargs)

class MyClass(MixinA, MixinB, MixinC):
    def __init__(self, name, a_param=None, b_param=None, c_param=None):
        # 关键：使用 **kwargs 让 super().__init__() 链式传递
        super().__init__(name=name, a_param=a_param,
                         b_param=b_param, c_param=c_param)
        self.name = name

# 注意 object.__init__ 不接受关键字参数
# 如果 MRO 链末端的类收到未知关键字参数，会报错
# 解决方案：确保每个 Mixin 都用 **kwargs 接收并传递多余参数

obj = MyClass("test", a_param="A", b_param="B", c_param="C")
print(f"name={obj.name}, a={obj.a_data}, b={obj.b_data}, c={obj.c_data}")
# name=test, a=A, b=B, c=C

# 但是 object.__init__() 不接受参数！
# MyClass 的 MRO: [MyClass, MixinA, MixinB, MixinC, object]
# 链末端的 super().__init__(**kwargs) 中如果 kwargs 不为空就会报错
# 因为 object.__init__() 不接受任何参数

# 解决方案：Mixin 中的 super().__init__() 必须传递 **kwargs
# 且每个 Mixin 从 kwargs 中 pop 自己的参数
class SafeMixinA:
    def __init__(self, **kwargs):
        self.a_data = kwargs.pop('a_param', None)  # pop 而非 get
        super().__init__(**kwargs)  # 传递剩余参数

class SafeMixinB:
    def __init__(self, **kwargs):
        self.b_data = kwargs.pop('b_param', None)
        super().__init__(**kwargs)

class SafeMixinC:
    def __init__(self, **kwargs):
        self.c_data = kwargs.pop('c_param', None)
        super().__init__(**kwargs)

class SafeClass(SafeMixinA, SafeMixinB, SafeMixinC):
    def __init__(self, name, **kwargs):
        super().__init__(**kwargs)  # name 不通过 kwargs 传递
        self.name = name

# 现在 object.__init__() 收到的是空的 kwargs，不会报错
obj2 = SafeClass("test", a_param="A", b_param="B", c_param="C")
print(f"name={obj2.name}, a={obj2.a_data}, b={obj2.b_data}, c={obj2.c_data}")
```

### 12.5.5 陷阱：isinstance 的误用与协议检查的正确姿势

```python
# 陷阱1：过度使用 isinstance 做类型检查
def process_data_bad(data):
    """反面教材：用 isinstance 做类型分派"""
    if isinstance(data, list):
        return [x * 2 for x in data]
    elif isinstance(data, dict):
        return {k: v * 2 for k, v in data.items()}
    elif isinstance(data, str):
        return data * 2
    else:
        raise TypeError(f"Unsupported type: {type(data)}")

# 问题：扩展性差，每增加一种类型就要修改此函数

# 正确方式1：鸭子类型 — 直接尝试操作
def process_data_duck(data):
    """鸭子类型方式 — 直接操作，处理异常"""
    try:
        return [x * 2 for x in data]  # 假设是可迭代的
    except TypeError:
        try:
            return data * 2  # 假设支持乘法
        except TypeError:
            raise TypeError(f"Cannot process {type(data)}")

# 正确方式2：协议检查 — 使用 hasattr 或 Protocol
def process_data_protocol(data):
    """协议检查方式 — 检查是否有需要的方法"""
    if hasattr(data, '__iter__') and not isinstance(data, str):
        return [x * 2 for x in data]
    elif hasattr(data, '__mul__'):
        return data * 2
    raise TypeError(f"Cannot process {type(data)}")

# 陷阱2：isinstance 检查虚拟子类的误解
from abc import ABC, abstractmethod

class MyABC(ABC):
    @abstractmethod
    def required(self):
        ...

@MyABC.register
class VirtualSub:
    """虚拟子类 — 注册但不继承"""
    def required(self):
        return "implemented"

vs = VirtualSub()
print(isinstance(vs, MyABC))  # True — isinstance 通过了！
print(MyABC in VirtualSub.__mro__)  # False — 但 MRO 中没有 MyABC！

# 关键区别：
# isinstance/issubclass 对虚拟子类返回 True
# 但虚拟子类不能调用 MyABC 中定义的具体方法（因为不在 MRO 中）
# 虚拟子类的 super() 也不会调用 MyABC 的方法

# 陷阱3：滥用 register 绕过抽象方法检查
class IncompleteClass:
    """没有实现 required 方法"""
    pass

MyABC.register(IncompleteClass)
print(issubclass(IncompleteClass, MyABC))  # True — 但实例化后调用 required 会崩溃！

ic = IncompleteClass()
try:
    ic.required()  # AttributeError!
except AttributeError as e:
    print(f"虚拟子类缺少方法: {e}")

# 最佳实践：不要用 register 让不完整的类通过 isinstance 检查
# 如果需要协议检查，使用 __subclasshook__ 或 typing.Protocol
```

### 12.5.7 最佳实践总结

```python
"""
继承与多态的最佳实践清单
"""

# ===== 1. 优先使用组合而非继承 =====
# 继承表达 "is-a" 关系（狗是动物）
# 组合表达 "has-a" 关系（汽车有引擎）
# 如果不确定，选组合

# ===== 2. Mixin 设计原则 =====
# - Mixin 不定义 __init__（或使用 **kwargs 透传）
# - Mixin 只提供方法，不存储状态（或只存储混入相关状态）
# - Mixin 通过 super() 调用下一个类（支持协作式调用）
# - Mixin 类名以 Mixin 结尾，表明意图

# ===== 3. 抽象基类使用原则 =====
# - 用 ABC 定义接口契约，不提供具体实现
# - 用 @abstractmethod 标记必须覆盖的方法
# - 用 @abstractmethod 标记必须覆盖的属性
# - 不要过度使用 register()——虚拟子类没有方法解析支持

# ===== 4. super() 使用原则 =====
# - 永远在 __init__ 中调用 super().__init__(**kwargs)
# - 理解 super() 是 MRO 代理，不是"父类调用"
# - 在多继承中，super() 确保每个类只被调用一次
# - 使用 **kwargs 透传参数，避免参数丢失

# ===== 5. 协议优于继承 =====
# - 用鸭子类型实现多态，不依赖继承关系
# - 用 typing.Protocol 进行静态类型检查
# - 用 @runtime_checkable 支持运行时 isinstance 检查
# - 用 hasattr 做轻量级协议检查

# ===== 6. 继承链深度控制 =====
# - 继承链不超过3层（基类 → 中间类 → 具体类）
# - 超过3层考虑用组合或 Mixin 重构
# - 避免在继承链中间层修改方法签名

# ===== 7. 避免的陷阱 =====
# - 不要直接调用 ParentClass.method(self)——用 super()
# - 不要在 Mixin 中假设特定兄弟类的存在
# - 不要用 isinstance 做类型分派——用多态或 Protocol
# - 不要让 Mixin 的 __init__ 消耗掉其他类需要的参数

print("继承与多态的最佳实践清单已列出")
```

### 12.5.8 性能对比

```python
"""
各种继承与多态模式的性能对比
"""
import timeit

# ===== 测试1：方法调用的开销 =====

class BaseMethod:
    def compute(self, x):
        return x * 2

class InheritedMethod(BaseMethod):
    pass

class OverriddenMethod(BaseMethod):
    def compute(self, x):
        return x * 2  # 相同逻辑，但覆盖了父类方法

class DuckTypeMethod:
    def compute(self, x):
        return x * 2  # 独立类，没有继承关系

def direct_call():
    """直接调用"""
    obj = DuckTypeMethod()
    return obj.compute(42)

def inherited_call():
    """继承调用"""
    obj = InheritedMethod()
    return obj.compute(42)

def overridden_call():
    """覆盖调用"""
    obj = OverriddenMethod()
    return obj.compute(42)

# 性能测试
iterations = 1_000_000
print("=== 方法调用性能对比 ===")
print(f"直接调用:      {timeit.timeit(direct_call, number=iterations):.4f}s")
print(f"继承调用:      {timeit.timeit(inherited_call, number=iterations):.4f}s")
print(f"覆盖调用:      {timeit.timeit(overridden_call, number=iterations):.4f}s")
# 结论：三者性能几乎无差异，CPython 对方法调用的优化使得继承链长度不影响性能

# ===== 测试2：super() vs 直接调用 =====

class SuperBase:
    def compute(self, x):
        return x * 2

class SuperMiddle(SuperBase):
    def compute(self, x):
        return super().compute(x)

class SuperDerived(SuperMiddle):
    def compute(self, x):
        return super().compute(x)

class DirectBase:
    def compute(self, x):
        return x * 2

class DirectMiddle(DirectBase):
    def compute(self, x):
        return DirectBase.compute(self, x)

class DirectDerived(DirectMiddle):

print(f"super() 调用:  {timeit.timeit(super_call, number=iterations):.4f}s")
print(f"直接父类调用:  {timeit.timeit(direct_parent_call, number=iterations):.4f}s")
# 结论：super() 有极微小的额外开销（MRO 查找），但通常可忽略不计

# ===== 测试3：isinstance vs hasattr vs Protocol 检查 =====

from abc import ABC, abstractmethod
from typing import Protocol, runtime_checkable

class CheckABC(ABC):
    @abstractmethod
    def method(self): ...

class CheckImpl(CheckABC):
    def method(self): return 42

@runtime_checkable
class CheckProtocol(Protocol):
    def method(self) -> int: ...

class CheckDuck:
    def method(self): return 42

obj_abc = CheckImpl()
obj_duck = CheckDuck()

def isinstance_check():
    return isinstance(obj_abc, CheckABC)

def hasattr_check():
    return hasattr(obj_duck, 'method')

def protocol_check():
    return isinstance(obj_duck, CheckProtocol)

print(f"\n=== 类型检查性能对比 ===")
check_iterations = 10_000_000
print(f"isinstance:    {timeit.timeit(isinstance_check, number=check_iterations):.4f}s")
print(f"hasattr:       {timeit.timeit(hasattr_check, number=check_iterations):.4f}s")
print(f"Protocol检查:  {timeit.timeit(protocol_check, number=check_iterations):.4f}s")
# 结论：isinstance 最快（有缓存），hasattr 中等，Protocol 检查最慢（需要遍历方法列表）
```
