## 1. 概念与语法

### 1.1 multiprocessing 模块概览

`multiprocessing` 模块是 Python 绕过 GIL 实现真正并行计算的核心工具。每个进程拥有独立的 Python 解释器实例和 GIL，因此多进程可以真正利用多核 CPU。

```
multiprocessing 模块架构:
┌────────────────────────────────────────────────────────────┐
│                    高层API                                   │
│  Pool ──── 进程池（最常用）                                   │
│  concurrent.futures.ProcessPoolExecutor ── 标准接口          │
├────────────────────────────────────────────────────────────┤
│                    中层API                                   │
│  Process ── 进程对象                                         │
│  Queue  ─── 进程安全队列                                     │
│  Pipe  ─── 进程间管道                                        │
│  Lock/RLock/Semaphore/Event/Condition ── 同步原语            │
├────────────────────────────────────────────────────────────┤
│                    共享状态                                   │
│  Value ──── 共享单值                                         │
│  Array ──── 共享数组                                         │
│  Manager ── 管理器（代理更丰富的数据类型）                      │
├────────────────────────────────────────────────────────────┤
│                    底层机制                                   │
│  fork() ─── Linux 进程复制                                   │
│  spawn() ── 进程创建（序列化参数）                             │
│  forkserver() ── 混合模式                                    │
└────────────────────────────────────────────────────────────┘
```

### 1.2 Process 对象

```python
from multiprocessing import Process, current_process
import os

def worker(name):
    """子进程执行的函数"""
    print(f"进程 {name}: PID={os.getpid()}, "
          f"父PID={os.getppid()}, "
          f"进程名={current_process().name}")

# 方式1：传入目标函数
p = Process(
    target=worker,           # 目标函数
    args=("子进程",),         # 位置参数（元组，注意逗号）
    kwargs={},                # 关键字参数
    name="Worker-1",          # 进程名
    daemon=False,             # 是否为守护进程
)

p.start()      # 启动子进程
p.join()       # 等待子进程完成
p.is_alive()   # 检查是否存活
p.pid          # 子进程PID
p.exitcode     # 退出码（None=运行中, 0=正常, 负数=信号终止）

# 方式2：继承Process类
class MyProcess(Process):
    def __init__(self, value):
        super().__init__(name=f"Custom-{value}")
        self.value = value
        self.result = None
    
    def run(self):
        """进程启动时执行的方法"""
        self.result = self.value ** 2
        print(f"{self.name}: 计算 {self.value}^2 = {self.result}")

p = MyProcess(42)
p.start()
p.join()
```

### 1.3 进程池 (Pool)

```python
from multiprocessing import Pool
import time

def square(x):
    """简单计算函数"""
    return x * x

# 方式1：上下文管理器（推荐）
with Pool(processes=4) as pool:
    # map: 顺序映射，保持结果顺序
    results = pool.map(square, range(10))
    print(f"map: {results}")
    
    # map_async: 异步映射，不阻塞主进程
    result = pool.map_async(square, range(10))
    results = result.get(timeout=10)
    
    # imap: 惰性迭代，内存友好
    for result in pool.imap(square, range(10)):
        print(result, end=" ")
    
    # imap_unordered: 惰性迭代，不保证顺序，最快完成先返回
    for result in pool.imap_unordered(square, range(10)):
        print(result, end=" ")
    
    # apply: 单任务执行
    result = pool.apply(square, args=(5,))
    
    # apply_async: 异步单任务执行
    result = pool.apply_async(square, args=(5,))
    result = result.get(timeout=5)  # 获取结果

# 方式2：手动管理
pool = Pool(processes=4)
try:
    results = pool.map(square, range(10))
finally:
    pool.close()    # 不再接受新任务
    pool.join()     # 等待所有任务完成
```

### 1.4 ProcessPoolExecutor

```python
from concurrent.futures import ProcessPoolExecutor, as_completed

def heavy_computation(n):
    """CPU密集型计算"""
    return sum(i * i for i in range(n))

with ProcessPoolExecutor(max_workers=4) as executor:
    # 方式1：map（保持顺序）
    results = list(executor.map(heavy_computation, [10**6] * 8))
    
    # 方式2：submit + as_completed（按完成顺序）
    futures = {
        executor.submit(heavy_computation, 10**6): i
        for i in range(8)
    }
    for future in as_completed(futures):
        idx = futures[future]
        result = future.result()
        print(f"任务{idx}完成: {result}")
```

### 1.5 进程间通信 (IPC)

```python
from multiprocessing import Process, Queue, Pipe, Value, Array
from multiprocessing import Manager

# ============ Queue（推荐方式） ============
def producer(q):
    for i in range(5):
        q.put(f"消息{i}")
    q.put(None)  # 哨兵值

def consumer(q):
    while True:
        msg = q.get()
        if msg is None:
            break
        print(f"消费者: {msg}")

q = Queue(maxsize=10)
p1 = Process(target=producer, args=(q,))
p2 = Process(target=consumer, args=(q,))
p1.start(); p2.start()
p1.join(); p2.join()

# ============ Pipe（双向管道） ============
def sender(conn):
    conn.send("Hello from sender")
    msg = conn.recv()
    print(f"Sender收到: {msg}")
    conn.close()

def receiver(conn):
    msg = conn.recv()
    print(f"Receiver收到: {msg}")
    conn.send("Hello from receiver")
    conn.close()

parent_conn, child_conn = Pipe()
p1 = Process(target=sender, args=(child_conn,))
p2 = Process(target=receiver, args=(parent_conn,))
p1.start(); p2.start()
p1.join(); p2.join()

# ============ Value / Array（共享内存） ============
def worker(shared_val, shared_arr):
    with shared_val.get_lock():
        shared_val.value += 1
    for i in range(len(shared_arr)):
        with shared_arr.get_lock():
            shared_arr[i] += 1

shared_val = Value('i', 0)      # 有符号整数
shared_arr = Array('d', [0.0] * 5)  # 双精度浮点数组

processes = [Process(target=worker, args=(shared_val, shared_arr))
             for _ in range(5)]
for p in processes:
    p.start()
for p in processes:
    p.join()

print(f"Value: {shared_val.value}")  # 5
print(f"Array: {shared_arr[:]}")     # [5.0, 5.0, 5.0, 5.0, 5.0]

# ============ Manager（代理复杂对象） ============
def worker(shared_dict, shared_list):
    shared_dict['key'] = 'value'
    shared_list.append(42)

with Manager() as manager:
    shared_dict = manager.dict()
    shared_list = manager.list()
    
    p = Process(target=worker, args=(shared_dict, shared_list))
    p.start()
    p.join()
    print(f"Dict: {shared_dict}")  # {'key': 'value'}
    print(f"List: {shared_list}")  # [42]
```

## 2. 原理与机制

### 2.1 进程启动方法的底层实现

Python 提供三种进程启动方法，各有不同的底层机制：

#### 2.1.1 fork（仅 Linux/macOS）

```c
// Lib/multiprocessing/popen_fork.py
class Popen:
    def __init__(self, process_obj):
        # 直接调用os.fork()
        self.pid = os.fork()
        if self.pid == 0:
            # 子进程：执行目标函数
            self._launch(process_obj)
            os._exit(0)
        else:
            # 父进程：继续
            pass
```

`fork()` 的内核实现：

```c
// Linux kernel: kernel/fork.c (简化)
SYSCALL_DEFINE0(fork)
{
    return do_fork(SIGCHLD, 0, 0, NULL);
}

long do_fork(unsigned long clone_flags, ...)
{
    // 1. 分配新的task_struct
    struct task_struct *p = copy_process(clone_flags, ...);
    
    // 2. 子进程获得父进程的：
    //    - 内存空间（写时复制）
    //    - 文件描述符表（共享）
    //    - 信号处理器
    //    - 但有独立的：
    //    - PID
    //    - 进程空间（COW，初始共享物理页）
    //    - 线程（只有调用fork的线程被复制）
    
    // 3. 唤醒子进程
    wake_up_new_task(p);
    return pid_vnr(p->pid);
}
```

fork 的内存模型——写时复制 (Copy-On-Write)：

```
fork()后内存状态:
┌─────────────────────────────────────────┐
│  物理内存页                              │
│  ┌──────┐ ┌──────┐ ┌──────┐ ┌──────┐ │
│  │ P1   │ │ P2   │ │ P3   │ │ P4   │ │
│  │只读  │ │只读  │ │只读  │ │只读  │ │
│  └──┬───┘ └──┬───┘ └──┬───┘ └──┬───┘ │
│     │         │         │         │     │
│  ┌──┴────────┴─────────┴────────┴──┐ │
│  │        共享页表(COW)              │ │
│  └──┬────────┬─────────┬────────┬──┘ │
│     │        │         │        │     │
│  ┌──┴──┐ ┌──┴──┐  ┌──┴──┐ ┌──┴──┐  │
│  │父进程│ │父进程│  │子进程│ │子进程│  │
│  │虚拟  │ │虚拟  │  │虚拟  │ │虚拟  │  │
│  │地址  │ │地址  │  │地址  │ │地址  │  │
│  └─────┘ └─────┘  └─────┘ └─────┘  │
└─────────────────────────────────────────┘

子进程修改P3页时:
┌─────────────────────────────────────────┐
│  P3被复制 → 父子进程各有独立副本        │
│  父进程P3(只读) → 子进程P3'(可写)        │
│  其他页仍然共享                          │
└─────────────────────────────────────────┘
```

fork 的问题：

1. **不安全**：如果父进程有线程，fork 只复制调用 fork 的线程，其他线程持有的锁永远无法释放，导致子进程死锁。
2. **内存膨胀**：虽然使用 COW，但 Python 的引用计数机制会导致几乎所有页被修改，COW 优势消失。

#### 2.1.2 spawn（所有平台，Windows 默认）

```python
# Lib/multiprocessing/popen_spawn_posix.py (Linux/Mac)
# Lib/multiprocessing/popen_spawn_win32.py (Windows)

class Popen:
    def __init__(self, process_obj):
        # 1. 序列化进程对象到管道/文件
        preparation_data = spawn.prepare(process_obj._config)
        
        # 2. 创建管道用于传递序列化数据
        parent_r, child_w = os.pipe()
        
        # 3. 启动全新的Python解释器进程
        # 执行: python -c 'from multiprocessing.spawn import spawn_main; spawn_main(...)'
        self.pid = spawnv_with_exec(...)
        
        # 4. 通过管道发送序列化数据
        os.write(child_w, pickle.dumps(preparation_data))
```

spawn 的启动流程：

```
spawn启动过程:
父进程                              子进程（全新进程）
───────                              ────────
序列化参数                              
序列化函数引用                            
序列化模块信息                            
创建管道                                 
│                                     
├─ fork + exec ───────────────────→ 启动新Python解释器
│                                     
通过管道发送序列化数据 ───────────→ 接收序列化数据
│                                     
│                                      反序列化参数
│                                      反序列化函数
│                                      导入必要模块
│                                      执行目标函数
│                                      将结果放入队列
│                                      退出
├─ 等待子进程 ←─────────────────── 完成
```

#### 2.1.3 forkserver（仅 Linux/macOS）

```python
# Lib/multiprocessing/forkserver.py

# forkserver的启动流程:
# 1. 主进程启动时创建一个forkserver进程
# 2. forkserver进程预导入所有必要模块
# 3. 每次需要新进程时，forkserver进程fork自己
# 4. 由于forkserver已经导入了模块，fork是安全的

class ForkServer:
    def __init__(self):
        # 启动forkserver进程
        self._forkserver_pid = os.fork()
        if self._forkserver_pid == 0:
            # forkserver子进程
            # 预导入模块
            for module in self._inherited_modules:
                importlib.import_module(module)
            # 等待创建进程的请求
            self._serve_forever()
    
    def _serve_forever(self):
        while True:
            # 接收创建进程请求
            request = self._recv_request()
            pid = os.fork()
            if pid == 0:
                # 新工作进程
                self._launch(request)
                os._exit(0)
```

三种启动方法对比：

| 特性 | fork | spawn | forkserver |
|------|------|-------|------------|
| 平台 | Linux/Mac | 全平台 | Linux/Mac |
| 启动速度 | 最快(~1ms) | 最慢(~50ms) | 中等(~10ms) |
| 内存使用 | COW（但Python引用计数抵消优势） | 全新分配 | COW（forkserver已导入模块） |
| 线程安全 | 不安全 | 安全 | 安全 |
| 文件描述符 | 继承全部 | 只继承管道 | 继承forkserver的 |
| 推荐场景 | 无线程的简单程序 | 通用场景（Windows必须） | 需要快速启动+线程安全 |

#### 2.2.1 pickle 的限制

#### 2.3.2 Manager 的实现

Manager 通过 Manager 进程代理对象访问：

```
Manager进程间通信:
┌──────────────┐                    ┌──────────────┐
│  主进程       │    IPC (socket)    │  Manager进程  │
│              │◄──────────────────►│              │

```

```python
# Manager的底层实现 (Lib/multiprocessing/managers.py)
class ListProxy:
    """代理列表——每次操作都通过IPC转发"""
    def append(self, value):
        # 通过IPC调用Manager进程中的实际list.append
        return self._callmethod('append', (value,))
    
    def __getitem__(self, index):
        # 通过IPC调用Manager进程中的实际list.__getitem__
        return self._callmethod('__getitem__', (index,))

# 因此：Manager代理对象比Value/Array慢得多
# 每次操作都涉及：序列化 → IPC传输 → 反序列化
```

### 2.4 进程池的内部调度机制

```python
# Lib/multiprocessing/pool.py (简化)
class Pool:
    def __init__(self, processes=None, ...):
        # 1. 创建任务队列（所有工作进程共享）
        self._taskqueue = Queue()
        
        # 2. 创建结果队列（所有工作进程向此写入结果）
        self._resultqueue = Queue()
        
        # 3. 创建工作进程
        self._processes = []
        for i in range(processes):
            p = Process(target=worker, args=(self._taskqueue, ...))
            p.start()
            self._processes.append(p)
        
        # 4. 启动结果处理线程
        self._result_handler = Thread(target=self._handle_results)
        self._result_handler.start()

def worker(inqueue, outqueue, ...):
    """工作进程的主循环"""
    while True:
        # 从任务队列获取任务
        task = inqueue.get()
        if task is None:  # 终止信号
            break
        
        # 执行任务
        job, func, args, kwargs = task
        try:
            result = func(*args, **kwargs)
            # 将结果放入结果队列
            outqueue.put((job, result))
        except Exception as e:
            outqueue.put((job, e))
```

## 3. 使用场景

### 3.1 场景：大规模数据并行处理

```python
from multiprocessing import Pool
import time

def process_chunk(data):
    """处理数据块"""
    return [x * x for x in data]

def parallel_data_processing():
    """并行处理大数据集"""
    data = list(range(1_000_000))
    chunk_size = 100_000
    chunks = [data[i:i+chunk_size] 
              for i in range(0, len(data), chunk_size)]
    
    # 串行处理
    start = time.perf_counter()
    results_serial = [process_chunk(chunk) for chunk in chunks]
    serial_time = time.perf_counter() - start
    
    # 并行处理
    start = time.perf_counter()
    with Pool() as pool:
        results_parallel = pool.map(process_chunk, chunks)
    parallel_time = time.perf_counter() - start
    
    print(f"串行: {serial_time:.3f}s")
    print(f"并行: {parallel_time:.3f}s")
    print(f"加速比: {serial_time/parallel_time:.2f}x")

parallel_data_processing()
```

### 3.2 场景：MapReduce 模式

```python
from multiprocessing import Pool
from collections import Counter
import re

def map_function(text_chunk):
    """Map阶段：统计每个单词的出现次数"""
    words = re.findall(r'\w+', text_chunk.lower())
    return Counter(words)

def reduce_function(counters):
    """Reduce阶段：合并所有计数器"""
    total = Counter()
    for counter in counters:
        total.update(counter)
    return total

def map_reduce_word_count(text, num_workers=4):
    """并行MapReduce单词计数"""
    # 分割文本为块
    chunk_size = len(text) // num_workers
    chunks = [text[i:i+chunk_size] for i in range(0, len(text), chunk_size)]
    
    # Map阶段（并行）
    with Pool(num_workers) as pool:
        partial_counts = pool.map(map_function, chunks)
    
    # Reduce阶段
    total_counts = reduce_function(partial_counts)
    
    return total_counts.most_common(10)

# 使用示例
if __name__ == "__main__":
    sample_text = "Hello world hello Python python world " * 100000
    result = map_reduce_word_count(sample_text)
```

### 3.4 场景：生产者-消费者多进程模式

```python
from multiprocessing import Process, Queue, JoinableQueue
import time
import random

def producer(queue, producer_id, count):
    """生产者进程"""
    for i in range(count):
        item = f"P{producer_id}-item{i}"
        queue.put(item)
        print(f"  生产者{producer_id}: 产生 {item}")
        time.sleep(random.uniform(0.01, 0.05))
    # 发送终止信号
    queue.put(None)

def consumer(queue, consumer_id):
    """消费者进程"""
    while True:
        item = queue.get()
        if item is None:
            queue.task_done()
            break
        print(f"    消费者{consumer_id}: 处理 {item}")
        time.sleep(random.uniform(0.02, 0.1))
        queue.task_done()

def run_producer_consumer():
    """运行生产者-消费者"""
    queue = JoinableQueue()
    
    # 启动消费者
    consumers = [
        Process(target=consumer, args=(queue, i), daemon=True)
        for i in range(3)
    ]
    for c in consumers:
        c.start()
    
    # 启动生产者
    producers = [
        Process(target=producer, args=(queue, i, 10))
        for i in range(2)
    ]
    for p in producers:
        p.start()
    
    # 等待生产者完成
    for p in producers:
        p.join()
    
    # 发送终止信号给消费者
    for _ in consumers:
        queue.put(None)
    
    # 等待消费者完成
    for c in consumers:
        c.join(timeout=5)

if __name__ == "__main__":
    run_producer_consumer()
```

### 4.2 进程间通信方式对比

```python
"""
进程间通信方式对比
Queue / Pipe / Value+Array / Manager
"""
import multiprocessing as mp
import time

def benchmark_queue(size=10000):
    """Queue通信基准"""
    def producer(q, count):
        for i in range(count):
            q.put(i)
        q.put(None)  # 终止信号
    
    def consumer(q, result_list):
        total = 0
        while True:
            item = q.get()
            if item is None:
                break
            total += item
        result_list.append(total)
    
    q = mp.Queue()
    with mp.Manager() as manager:
        result_list = manager.list()
        p1 = mp.Process(target=producer, args=(q, size))
        p2 = mp.Process(target=consumer, args=(q, result_list))
        start = time.perf_counter()
        p1.start(); p2.start()
        p1.join(); p2.join()
        elapsed = time.perf_counter() - start
    
    return elapsed, result_list[0]

def benchmark_pipe(size=10000):
    """Pipe通信基准"""
    def sender(conn, count):
        for i in range(count):
            conn.send(i)
        conn.send(None)
        conn.close()
    
    def receiver(conn, result_list):
        total = 0
        while True:
            item = conn.recv()
            if item is None:
                break
            total += item
        result_list.append(total)
    
    parent_conn, child_conn = mp.Pipe()
    with mp.Manager() as manager:
        result_list = manager.list()
        p1 = mp.Process(target=sender, args=(child_conn, size))
        p2 = mp.Process(target=receiver, args=(parent_conn, result_list))
        start = time.perf_counter()
        p1.start(); p2.start()
        p1.join(); p2.join()
        elapsed = time.perf_counter() - start
    
    return elapsed, result_list[0]

def benchmark_shared_memory(size=10000):
    """共享内存通信基准"""
    def worker(arr, count, result_val):
        total = 0
        for i in range(count):
            total += arr[i]
        result_val.value = total
    
    arr = mp.Array('i', list(range(size)))
    result_val = mp.Value('i', 0)
    
    p = mp.Process(target=worker, args=(arr, size, result_val))
    start = time.perf_counter()
    p.start()
    p.join()
    elapsed = time.perf_counter() - start
    
    return elapsed, result_val.value

def run_ipc_benchmark():
    """运行IPC基准测试"""
    size = 100000
    
    print("进程间通信方式基准测试")
    print(f"数据量: {size} 个整数")
    print("=" * 50)
    
    q_time, q_result = benchmark_queue(size)
    print(f"Queue:      {q_time:.3f}s, 结果={q_result}")
    
    p_time, p_result = benchmark_pipe(size)
    print(f"Pipe:       {p_time:.3f}s, 结果={p_result}")
    
    s_time, s_result = benchmark_shared_memory(size)
    print(f"共享内存:   {s_time:.3f}s, 结果={s_result}")

if __name__ == "__main__":
    run_ipc_benchmark()
```

```python
"""
共享状态的正确使用方式
Value/Array vs Manager 对比
"""
import multiprocessing as mp
import time

# ❌ 错误：进程间不能直接共享Python对象
def bad_shared_state():
    """进程间无法共享普通Python对象"""
    shared_list = []  # 每个进程有自己的副本
    
    def worker(item):
        shared_list.append(item)  # 修改的是子进程的副本！
        print(f"  子进程内: {shared_list}")
    
    processes = []
    for i in range(3):
        p = mp.Process(target=worker, args=(i,))
        processes.append(p)
        p.start()
    
    for p in processes:
        p.join()
    
    print(f"主进程: {shared_list}")  # 空列表！

# ✅ 正确方式1：使用Value/Array（共享内存，最快）
def shared_memory_example():
    """使用Value和Array共享原始数据"""
    counter = mp.Value('i', 0)   # 有符号整数
    numbers = mp.Array('d', [0.0] * 5)  # 双精度浮点数组
    
    def worker(counter, numbers, worker_id):
        for i in range(1000):
            with counter.get_lock():  # 必须加锁！
                counter.value += 1
        with numbers.get_lock():
            numbers[worker_id] = counter.value
    
    processes = [
        mp.Process(target=worker, args=(counter, numbers, i))
        for i in range(5)
    ]
    for p in processes:
        p.start()
    for p in processes:
        p.join()
    
    print(f"计数器: {counter.value}")  # 5000
    print(f"数组: {numbers[:]}")

# ✅ 正确方式2：使用Manager（更灵活但更慢）
def manager_example():
    """使用Manager共享复杂对象"""
    with mp.Manager() as manager:
        shared_dict = manager.dict()
        shared_list = manager.list()
        shared_lock = manager.Lock()
        
        def worker(worker_id):
            with shared_lock:
                shared_dict[f"worker_{worker_id}"] = f"result_{worker_id}"
                shared_list.append(worker_id)
        
        processes = [
            mp.Process(target=worker, args=(i,))
            for i in range(5)
        ]
        for p in processes:
            p.start()
        for p in processes:
            p.join()
        
        print(f"字典: {dict(shared_dict)}")
        print(f"列表: {list(shared_list)}")

if __name__ == "__main__":
    print("=== 错误方式：普通Python对象 ===")
    bad_shared_state()
    
    print("\n=== 正确方式1：共享内存 ===")
    shared_memory_example()
    
    print("\n=== 正确方式2：Manager ===")
    manager_example()
```

### 5.3 陷阱：fork 模式下的死锁

```python
import threading
from multiprocessing import Process

# ❌ 错误：fork模式下线程持有的锁导致子进程死锁
lock = threading.Lock()

def init():
    """主进程中的初始化"""
    lock.acquire()  # 获取锁
    # 如果在此处fork，子进程继承锁的锁定状态
    # 但只有获取锁的线程被复制到子进程
    # 其他等待锁的线程不会被复制
    # 子进程中的锁永远不会被释放 → 死锁！

# ✅ 正确：使用spawn模式避免fork死锁
import multiprocessing as mp
mp.set_start_method('spawn')  # 全局设置

# ✅ 正确：确保fork前没有活跃的锁
def safe_init():
    """安全的初始化"""
    # 确保所有锁都已释放
    lock.acquire()
    try:
        pass  # 初始化代码
    finally:
        lock.release()  # 释放锁后再fork
```

### 5.5 陷阱：大对象序列化开销

```python
import multiprocessing as mp
import time

# ❌ 错误：传递大对象给子进程
def process_large_data(data):
    """每次调用都序列化/反序列化大对象"""
    return sum(data)

def bad_approach():
    large_data = list(range(10_000_000))  # ~80MB
    
    with mp.Pool(4) as pool:
        # 每个任务都序列化80MB数据！
        start = time.perf_counter()
        results = pool.map(process_large_data, [large_data] * 4)
        elapsed = time.perf_counter() - start
    
    print(f"传递大对象: {elapsed:.3f}s")
    return results

# ✅ 正确：使用initializer共享数据
def init_worker(data):
    """进程初始化时加载共享数据（只序列化一次）"""
    global shared_data
    shared_data = data

def process_shared_data(chunk_start, chunk_end):
    """使用进程内的共享数据"""
    return sum(shared_data[chunk_start:chunk_end])

def good_approach():
    large_data = list(range(10_000_000))
    chunk_size = len(large_data) // 4
    chunks = [(i * chunk_size, (i + 1) * chunk_size) for i in range(4)]
    
    with mp.Pool(4, initializer=init_worker, initargs=(large_data,)) as pool:
        start = time.perf_counter()
        results = pool.starmap(process_shared_data, chunks)
        elapsed = time.perf_counter() - start
    
    print(f"共享数据: {elapsed:.3f}s")
    return results

# ✅ 正确：使用共享内存避免序列化
def shared_memory_approach():
    import multiprocessing.shared_memory
    import numpy as np
    
    # 创建共享内存
    data = np.arange(10_000_000, dtype=np.float64)
    shm = multiprocessing.shared_memory.SharedMemory(
        create=True, size=data.nbytes, name="my_shm"
    )
    
    # 将数据写入共享内存
    shared_array = np.ndarray(data.shape, dtype=data.dtype, buffer=shm.buf)
    np.copyto(shared_array, data)
    
    def worker(start, end):
        # 直接从共享内存读取，无需序列化
        existing_shm = multiprocessing.shared_memory.SharedMemory(name="my_shm")
        arr = np.ndarray((10_000_000,), dtype=np.float64, buffer=existing_shm.buf)
        result = np.sum(arr[start:end])
        existing_shm.close()
        return result
    
    chunk_size = len(data) // 4
    chunks = [(i * chunk_size, (i + 1) * chunk_size) for i in range(4)]
    
    with mp.Pool(4) as pool:
        start = time.perf_counter()
        results = pool.starmap(worker, chunks)
        elapsed = time.perf_counter() - start
    
    print(f"共享内存: {elapsed:.3f}s")
    shm.close()
    shm.unlink()
    return results
```

### 5.6 最佳实践总结

| 实践 | 说明 |
|------|------|
| **始终使用 `if __name__ == "__main__"`** | Windows spawn 模式必需 |
| **优先使用 `concurrent.futures.ProcessPoolExecutor`** | 更现代的 API，自动资源管理 |
| **优先使用 `Queue` 进行进程间通信** | 线程安全、pickle 友好 |
| **大对象使用共享内存或 initializer** | 避免重复序列化 |
| **设置合理的 `max_workers`** | 通常等于 `os.cpu_count()` |
| **子进程函数定义在模块顶层** | 确保可 pickle |
| **使用 `daemon=True` 防止孤儿进程** | 主进程退出时自动终止子进程 |
| **处理子进程异常** | 使用 `try/except` 或 `error_callback` |
| **避免 fork 模式** | 使用 `spawn` 或 `forkserver` |
| **资源清理** | 使用上下文管理器或 `try/finally` |

### 5.7 进程间通信方式选择指南

| 方式 | 速度 | 灵活性 | 适用场景 |
|------|------|--------|----------|
| `Queue` | 中 | 高 | 通用进程间数据传递 |
| `Pipe` | 高 | 低 | 双进程双向通信 |
| `Value/Array` | 最高 | 低 | 共享原始数值/数组 |
| `shared_memory` | 最高 | 中 | 共享大数据（NumPy数组等） |
| `Manager` | 低 | 最高 | 共享dict/list等复杂对象 |
| `Manager.Proxy` | 低 | 最高 | 自定义代理对象 |
