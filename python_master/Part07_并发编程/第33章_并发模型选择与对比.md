## 1. 概念与语法

### 1.1 三大并发模型回顾

Python 提供三大并发模型，各自适用于不同场景：

```
┌──────────────────────────────────────────────────────────────┐
│                    Python 三大并发模型                         │
├──────────────────────────────────────────────────────────────┤
│                                                              │
│  多线程 (threading)         多进程 (multiprocessing)         │
│  ┌─────────────────┐       ┌─────────────────────┐          │
│  │ 共享地址空间      │       │ 独立地址空间          │          │
│  │ 受GIL限制        │       │ 绕过GIL              │          │
│  │ 线程级并发       │       │ 进程级并行            │          │
│  │ 轻量级(~8MB/线程) │       │ 重量级(~30MB/进程)   │          │
│  └─────────────────┘       └─────────────────────┘          │
│                                                              │
│              协程 (asyncio)                                  │
│              ┌─────────────────┐                             │
│              │ 单线程事件循环    │                             │
│              │ 协作式调度        │                             │
│              │ 协程级并发        │                             │
│              │ 极轻量(~1KB/协程) │                             │
│              └─────────────────┘                             │
│                                                              │
└──────────────────────────────────────────────────────────────┘
```

### 1.2 并发模型 API 对照表

| 功能 | threading | multiprocessing | asyncio |
|------|-----------|-----------------|---------|
| 创建执行单元 | `Thread(target=fn)` | `Process(target=fn)` | `asyncio.create_task(coro())` |
| 等待完成 | `t.join()` | `p.join()` | `await task` |
| 获取结果 | 共享变量 | `Queue/Value` | `task.result()` |
| 池化 | `ThreadPoolExecutor` | `ProcessPoolExecutor` | `asyncio.gather()` |
| 互斥锁 | `threading.Lock()` | `multiprocessing.Lock()` | `asyncio.Lock()` |
| 事件 | `threading.Event()` | `multiprocessing.Event()` | `asyncio.Event()` |
| 信号量 | `threading.Semaphore()` | `multiprocessing.Semaphore()` | `asyncio.Semaphore()` |
| 条件变量 | `threading.Condition()` | `multiprocessing.Condition()` | `asyncio.Condition()` |
| 队列 | `queue.Queue()` | `multiprocessing.Queue()` | `asyncio.Queue()` |
| 管道 | N/A | `multiprocessing.Pipe()` | N/A |
| 共享内存 | N/A | `Value/Array` | N/A |
| 本地存储 | `threading.local()` | N/A | `contextvars` |
| 超时 | `t.join(timeout)` | `p.join(timeout)` | `asyncio.wait_for(coro, timeout)` |
| 取消 | 不支持 | `p.terminate()` | `task.cancel()` |

### 1.3 混合并发模型

```python
import asyncio
from concurrent.futures import ThreadPoolExecutor, ProcessPoolExecutor

# ─── asyncio + ThreadPoolExecutor ───
async def async_with_blocking_io():
    """异步IO + 阻塞IO"""
    loop = asyncio.get_event_loop()
    
    # 在线程池中运行阻塞IO
    with ThreadPoolExecutor() as pool:
        result = await loop.run_in_executor(
            pool, blocking_io_function, arg1, arg2
        )
    return result

# ─── asyncio + ProcessPoolExecutor ───
async def async_with_cpu_heavy():
    """异步IO + CPU密集型计算"""
    loop = asyncio.get_event_loop()
    
    # 在进程池中运行CPU密集型任务
    with ProcessPoolExecutor() as pool:
        result = await loop.run_in_executor(
            pool, cpu_heavy_function, data
        )
    return result

# ─── threading + multiprocessing ───
import threading
from multiprocessing import Pool

def thread_with_process():
    """线程中启动进程池"""
    results = []
    
    def callback(result):
        results.extend(result)
    
    with Pool(4) as pool:
        pool.map_async(cpu_heavy_function, data_chunks, callback=callback)
    
    return results

# ─── asyncio + threading + multiprocessing ───
async def full_hybrid():
    """三模型混合"""
    loop = asyncio.get_event_loop()
    
    # 异步IO
    async_task = asyncio.create_task(async_io_operation())
    
    # 在线程池中运行阻塞IO
    with ThreadPoolExecutor(4) as thread_pool:
        blocking_result = await loop.run_in_executor(
            thread_pool, blocking_io_operation
        )
    
    # 在进程池中运行CPU密集型
    with ProcessPoolExecutor(4) as process_pool:
        cpu_result = await loop.run_in_executor(
            process_pool, cpu_heavy_operation
        )
    
    async_result = await async_task
    
    return async_result, blocking_result, cpu_result
```

```python
"""
创建开销对比测试
"""
import time
import threading
import multiprocessing as mp
import asyncio

def measure_creation():
    """测量不同并发单元的创建开销"""
    
    # 线程创建
    def empty_thread():
        pass
    
    start = time.perf_counter()
    threads = []
    for _ in range(1000):
        t = threading.Thread(target=empty_thread)
        threads.append(t)
    thread_create_time = time.perf_counter() - start
    
    for t in threads:
        t.start()
    start = time.perf_counter()
    for t in threads:
        t.join()
    thread_join_time = time.perf_counter() - start
    
    # 进程创建
    def empty_process():
        pass
    
    start = time.perf_counter()
    processes = []
    for _ in range(10):  # 进程开销大，只测10个
        p = mp.Process(target=empty_process)
        processes.append(p)
    process_create_time = time.perf_counter() - start
    
    for p in processes:
        p.start()
    start = time.perf_counter()
    for p in processes:
        p.join()
    process_join_time = time.perf_counter() - start
    
    # 协程创建
    async def empty_coro():
        return 1
    
    start = time.perf_counter()
    tasks = []
    for _ in range(10000):
        task = asyncio.create_task(empty_coro())
        tasks.append(task)
    coro_create_time = time.perf_counter() - start
    
    print(f"线程创建:  {thread_create_time/1000*1e6:.1f}μs/个")
    print(f"线程等待:  {thread_join_time/1000*1e6:.1f}μs/个")
    print(f"进程创建:  {process_create_time/10*1e3:.1f}ms/个")
    print(f"进程等待:  {process_join_time/10*1e3:.1f}ms/个")
    print(f"协程创建:  {coro_create_time/10000*1e6:.1f}μs/个")
```

```python
"""
上下文切换开销对比测试
"""
import time
import threading
import multiprocessing as mp
import asyncio

def benchmark_context_switch():
    """测量上下文切换开销"""
    
    SWITCH_COUNT = 10000
    
    # 线程切换
    def thread_ping_pong(count, event1, event2):
        for _ in range(count):
            event1.set()
            event2.wait()
            event2.clear()
    
    event1 = threading.Event()
    event2 = threading.Event()
    t = threading.Thread(target=thread_ping_pong, 
                        args=(SWITCH_COUNT, event1, event2))
    start = time.perf_counter()
    t.start()
    for _ in range(SWITCH_COUNT):
        event1.wait()
        event1.clear()
        event2.set()
    t.join()
    thread_switch_time = (time.perf_counter() - start) / SWITCH_COUNT
    
    # 进程切换
    def process_ping_pong(count, conn):
        for _ in range(count):
            conn.send("ping")
            conn.recv()
    
    parent_conn, child_conn = mp.Pipe()
    p = mp.Process(target=process_ping_pong, 
                   args=(SWITCH_COUNT//10, child_conn))
    start = time.perf_counter()
    p.start()
    for _ in range(SWITCH_COUNT//10):
        parent_conn.recv()
        parent_conn.send("pong")
    p.join()
    process_switch_time = (time.perf_counter() - start) / (SWITCH_COUNT//10)
    
    # 协程切换
    async def coro_ping_pong(count):
        for _ in range(count):
            await asyncio.sleep(0)  # 让出控制权
    
    async def measure_coro_switch():
        start = time.perf_counter()
        await coro_ping_pong(SWITCH_COUNT)
        return (time.perf_counter() - start) / SWITCH_COUNT
    
    coro_switch_time = asyncio.run(measure_coro_switch())
    
    print(f"线程切换: {thread_switch_time*1e6:.2f}μs")
    print(f"进程切换: {process_switch_time*1e6:.2f}μs")
    print(f"协程切换: {coro_switch_time*1e6:.2f}μs")
```

#### 2.2.3 通信开销

```python
"""
通信开销对比测试
"""
import time
import threading
import multiprocessing as mp
import asyncio
import queue

def benchmark_communication():
    """测量进程间/线程间/协程间通信开销"""
    MSG_COUNT = 10000
    
    # 线程Queue
    def thread_queue_test():
        q = queue.Queue()
        def producer():
            for i in range(MSG_COUNT):
                q.put(i)
            q.put(None)
        def consumer():
            total = 0
            while True:
                item = q.get()
                if item is None:
                    break
                total += item
        
        start = time.perf_counter()
        t1 = threading.Thread(target=producer)
        t2 = threading.Thread(target=consumer)
        t1.start(); t2.start()
        t1.join(); t2.join()
        return time.perf_counter() - start
    
    # 进程Queue
    def process_queue_test():
        q = mp.Queue()
        def producer():
            for i in range(MSG_COUNT):
                q.put(i)
            q.put(None)
        def consumer():
            while True:
                item = q.get()
                if item is None:
                    break
        
        start = time.perf_counter()
        p1 = mp.Process(target=producer)
        p2 = mp.Process(target=consumer)
        p1.start(); p2.start()
        p1.join(); p2.join()
        return time.perf_counter() - start
    
    # 协程Queue
    async def coro_queue_test():
        q = asyncio.Queue()
        
        async def producer():
            for i in range(MSG_COUNT):
                await q.put(i)
            await q.put(None)
        
        async def consumer():
            while True:
                item = await q.get()
                if item is None:
                    break
        
        start = time.perf_counter()
        await asyncio.gather(producer(), consumer())
        return time.perf_counter() - start
    
    thread_time = thread_queue_test()
    process_time = process_queue_test()
    coro_time = asyncio.run(coro_queue_test())
    
    print(f"线程Queue: {thread_time:.3f}s ({MSG_COUNT/thread_time:.0f} msg/s)")
    print(f"进程Queue: {process_time:.3f}s ({MSG_COUNT/process_time:.0f} msg/s)")
    print(f"协程Queue: {coro_time:.3f}s ({MSG_COUNT/coro_time:.0f} msg/s)")
```

### 3.1 线程/进程/协程性能基准测试

```python
"""
并发模型性能基准测试
全面对比 threading/multiprocessing/asyncio 在不同场景下的表现
"""
import time
import threading
import multiprocessing as mp
import asyncio
from concurrent.futures import ThreadPoolExecutor, ProcessPoolExecutor

# ─── CPU密集型任务 ───
def cpu_task(n):
    """质数计数"""
    count = 0
    for i in range(2, n):
        for j in range(2, int(i**0.5)+1):
            if i % j == 0:
                break
        else:
            count += 1
    return count

# ─── IO密集型任务 ───
def io_task_sync(duration):
    """模拟阻塞IO"""
    time.sleep(duration)
    return duration

async def io_task_async(duration):
    """模拟异步IO"""
    await asyncio.sleep(duration)
    return duration

# ─── 混合型任务 ───
def mixed_task_sync(compute, duration):
    """CPU计算 + 阻塞IO"""
    cpu_task(compute)
    time.sleep(duration)
    return compute

async def mixed_task_async(compute, duration):
    """CPU计算 + 异步IO"""
    loop = asyncio.get_event_loop()
    await loop.run_in_executor(None, cpu_task, compute)
    await asyncio.sleep(duration)
    return compute

class ConcurrencyBenchmark:
    """并发模型基准测试"""
    
    def __init__(self, task_count=4, cpu_size=50000, io_duration=0.5):
        self.task_count = task_count
        self.cpu_size = cpu_size
        self.io_duration = io_duration
    
    # ─── CPU密集型基准 ───
    def benchmark_cpu_serial(self):
        start = time.perf_counter()
        results = [cpu_task(self.cpu_size) for _ in range(self.task_count)]
        return time.perf_counter() - start
    
    def benchmark_cpu_threads(self):
        start = time.perf_counter()
        with ThreadPoolExecutor(max_workers=self.task_count) as pool:
            results = list(pool.map(cpu_task, [self.cpu_size] * self.task_count))
        return time.perf_counter() - start
    
    def benchmark_cpu_processes(self):
        start = time.perf_counter()
        with ProcessPoolExecutor(max_workers=self.task_count) as pool:
            results = list(pool.map(cpu_task, [self.cpu_size] * self.task_count))
        return time.perf_counter() - start
    
    # ─── IO密集型基准 ───
    def benchmark_io_serial(self):
        start = time.perf_counter()
```

### 3.2 适用场景决策树

```
开始: 你需要并发吗？
│
├── 否 → 使用同步代码（最简单、最可预测）
│
└── 是 → 任务类型是什么？
    │
    ├── 纯CPU密集型
    │   ├── 任务数 ≤ CPU核心数？
    │   │   ├── 是 → multiprocessing.Pool
    │   │   │        ProcessPoolExecutor
    │   │   └── 否 → multiprocessing.Pool + 合理chunk_size
    │   │            避免过多进程
    │   │
    │   ├── 需要共享大量数据？
    │   │   ├── 是 → shared_memory + multiprocessing
    │   │   │        或 NumPy + multiprocessing
    │   │   └── 否 → multiprocessing.Queue/Pipe
    │   │
    │   └── 任务很短（< 1ms）？
    │       ├── 是 → 批量化处理（减少进程创建开销）
    │       │        Pool.map with chunksize
    │       └── 否 → 正常使用进程池
    │
    ├── 纯IO密集型
    │   ├── 并发数 ≤ 100？
    │   │   ├── 是 + 有异步库 → asyncio
    │   │   ├── 是 + 无异步库 → threading
    │   │   └── 是 + 混合库 → ThreadPoolExecutor
    │   │
    │   ├── 并发数 100-1000？
    │   │   ├── 是 + 有异步库 → asyncio
    │   │   └── 是 + 无异步库 → ThreadPoolExecutor
    │   │                (注意内存)
    │   │
    │   └── 并发数 > 1000？
    │       ├── 是 → asyncio（唯一选择）
    │       └── 注意：需要异步库支持
    │
    └── 混合型（CPU + IO）
        ├── IO为主 + 少量CPU
        │   └── asyncio + run_in_executor(ThreadPoolExecutor)
        │
        ├── CPU为主 + 少量IO
        │   └── multiprocessing + Queue
        │
        └── CPU和IO都重
            └── asyncio + ProcessPoolExecutor
                事件循环处理IO
                进程池处理CPU
```

```python
"""
混合并发模型：asyncio + ThreadPool + ProcessPool
处理IO密集型 + CPU密集型混合任务
"""
import asyncio
import time
from concurrent.futures import ThreadPoolExecutor, ProcessPoolExecutor

class HybridConcurrency:
    """混合并发处理器"""
    
    def __init__(self, max_threads=10, max_processes=4):
        self.max_threads = max_threads
        self.max_processes = max_processes
    
    async def process_io_bound(self, urls):
        """处理IO密集型任务"""
        import aiohttp
        
        semaphore = asyncio.Semaphore(self.max_threads)
        
        async def fetch(session, url):
            async with semaphore:
                async with session.get(url, timeout=aiohttp.ClientTimeout(total=10)) as resp:
                    return await resp.text()
        
        async with aiohttp.ClientSession() as session:
            tasks = [fetch(session, url) for url in urls]
            results = await asyncio.gather(*tasks, return_exceptions=True)
        
        return results
    
    async def process_cpu_bound(self, data_chunks):
        """处理CPU密集型任务"""
        loop = asyncio.get_event_loop()
        
        with ProcessPoolExecutor(max_workers=self.max_processes) as pool:
            futures = [
                loop.run_in_executor(pool, self._cpu_heavy, chunk)
                for chunk in data_chunks
            ]
            results = await asyncio.gather(*futures)
        
        return results
    
    def _cpu_heavy(self, data):
        """CPU密集型计算"""
        return sum(x * x for x in data)
    
    async def process_mixed(self, tasks):
        """处理混合型任务"""
        loop = asyncio.get_event_loop()
        results = []
        
        with ThreadPoolExecutor(max_workers=self.max_threads) as thread_pool, \
             ProcessPoolExecutor(max_workers=self.max_processes) as process_pool:
            
            async def process_task(task):
                task_type, data = task
                
                if task_type == "io":
                    # IO任务：异步处理
                    result = await self._async_io(data)
                elif task_type == "cpu":
                    # CPU任务：进程池处理
                    result = await loop.run_in_executor(
                        process_pool, self._cpu_heavy, data
                    )
                elif task_type == "blocking_io":
                    # 阻塞IO：线程池处理
                    result = await loop.run_in_executor(
                        thread_pool, self._blocking_io, data
                    )
                return result
            
            results = await asyncio.gather(
                *[process_task(task) for task in tasks],
                return_exceptions=True
            )
        
        return results
    
    async def _async_io(self, data):
        await asyncio.sleep(0.1)
        return f"async_result_{data}"
    
    def _blocking_io(self, data):
        time.sleep(0.1)
        return f"blocking_result_{data}"

async def main():
    hybrid = HybridConcurrency(max_threads=10, max_processes=4)
    
    # 混合任务
    tasks = [
        ("io", "url1"),
        ("cpu", list(range(1000000))),
        ("blocking_io", "url2"),
        ("io", "url3"),
        ("cpu", list(range(1000000))),
    ]
    
    results = await hybrid.process_mixed(tasks)
    for result in results:
        print(f"结果: {result}")

if __name__ == "__main__":
    asyncio.run(main())
```

```python
"""
```

```python
import asyncio
import time
from concurrent.futures import ProcessPoolExecutor
import json

class HybridServer:
    """混合并发服务器"""
    
    def __init__(self, host='127.0.0.1', port=8888, max_workers=4):
        self.host = host
        self.port = port
        self.max_workers = max_workers
        self.process_pool = None
        self.request_count = 0
        self.lock = asyncio.Lock()

    @staticmethod
    def heavy_computation(data):
        """CPU密集型计算——在进程池中执行"""
        import hashlib
        result = hashlib.sha256(str(data).encode()).hexdigest()
        # 模拟额外计算
        for i in range(1000000):
            result = hashlib.sha256(result.encode()).hexdigest()
        return {"hash": result, "input_size": len(str(data))}
    
    # ─── 异步IO处理 ───
    async def handle_client(self, reader, writer):
        """处理客户端连接——异步IO"""
        addr = writer.get_extra_info('peername')
        
        try:
            # 读取请求
            data = await reader.read(65536)
            if not data:
                return
            
            request = json.loads(data.decode())
            request_type = request.get("type", "unknown")
            
            async with self.lock:
                self.request_count += 1
            
            if request_type == "compute":
                # CPU密集型：在进程池中执行
                loop = asyncio.get_event_loop()
                result = await loop.run_in_executor(
                    self.process_pool,
                    self.heavy_computation,
                    request.get("data", "")
                )
                response = {"status": "ok", "result": result}

                # IO密集型：直接在事件循环中处理

                response = {"status": "error", "message": "unknown type"}

            writer.write(json.dumps(response).encode())
            await writer.drain()
        
        except Exception as e:
            error_response = {"status": "error", "message": str(e)}
            writer.write(json.dumps(error_response).encode())
            await writer.drain()
        finally:
            writer.close()
            await writer.wait_closed()
```

```python
import threading

def cpu_heavy():
```

```python
    print(f"  多线程: {parallel:.3f}s")
    print(f"  加速比: {serial/parallel:.2f}x")
    
    # 注意：不是所有NumPy操作都释放GIL
    # 释放GIL的操作需要内部使用Py_BEGIN_ALLOW_THREADS

# 教训：C扩展可以释放GIL实现真正并行，但需要确认
```

### 5.3 陷阱：asyncio 中混用阻塞代码

```python
import asyncio
import time

# ❌ 错误：在协程中使用阻塞IO
async def bad_mixed():
    # 阻塞整个事件循环5秒！
    time.sleep(5)
    # 其他所有协程都被阻塞
    await asyncio.sleep(1)
    return "done"

# ✅ 正确方式1：使用异步替代
async def good_async():
    await asyncio.sleep(5)  # 非阻塞
    await asyncio.sleep(1)
    return "done"

# ✅ 正确方式2：使用run_in_executor
async def good_executor():
    loop = asyncio.get_event_loop()
    
    # 阻塞IO在线程池中执行
    result = await loop.run_in_executor(None, time.sleep, 5)
    
    # CPU密集型在进程池中执行
    with ProcessPoolExecutor() as pool:
        result = await loop.run_in_executor(pool, cpu_heavy, data)
    
    return "done"

# ✅ 正确方式3：分片让出控制权
async def good_chunked():
    total = 0
    for i in range(10_000_000):
        total += i
        if i % 100_000 == 0:
            await asyncio.sleep(0)  # 定期让出控制权
    return total
```

### 5.4 陷阱：进程间通信选择错误

```python
import multiprocessing as mp
import time

# ❌ 错误：用Manager传递大数组（序列化开销）
def bad_large_data():
    with mp.Manager() as manager:
        shared_list = manager.list(range(1_000_000))  # 极慢！
        # Manager代理每次操作都涉及IPC
        # manager.list()[0] → 序列化请求 → IPC → 反序列化响应

# ✅ 正确：用共享内存传递大数组
def good_large_data():
    import multiprocessing.shared_memory
    import numpy as np
    
    # 创建共享内存
    data = np.arange(1_000_000, dtype=np.float64)
    shm = mp.shared_memory.SharedMemory(
        create=True, size=data.nbytes
    )
    
    # 写入共享内存
    shared_array = np.ndarray(
        data.shape, dtype=data.dtype, buffer=shm.buf
    )
    np.copyto(shared_array, data)
    
    # 工作进程直接访问共享内存，零拷贝！
    def worker(shm_name, shape, dtype):
        existing_shm = mp.shared_memory.SharedMemory(name=shm_name)
        arr = np.ndarray(shape, dtype=dtype, buffer=existing_shm.buf)
        result = np.sum(arr)
        existing_shm.close()
        return result
    
    with mp.Pool(4) as pool:
        results = pool.map(
            lambda name: worker(name, data.shape, data.dtype),
            [shm.name] * 4
        )
    
    shm.close()
    shm.unlink()

# 选择指南：
# ┌──────────────┬────────────────┬──────────────┐
# │ 数据类型     │ 推荐方式       │ 原因          │
# ├──────────────┼────────────────┼──────────────┤
# │ 小对象       │ Queue/Pipe     │ 简单，开销可接受│
# │ 原始数组     │ Value/Array    │ 零拷贝共享内存  │
# │ NumPy数组    │ shared_memory  │ 零拷贝共享内存  │
# │ dict/list    │ Manager        │ 唯一选择       │
# │ 大数据       │ 文件/mmap      │ 避免序列化     │
# └──────────────┴────────────────┴──────────────┘
```

### 5.5 陷阱：线程/进程数过多

```python
import os
import time
from concurrent.futures import ThreadPoolExecutor, ProcessPoolExecutor

# ❌ 错误：创建过多线程
def bad_thread_count():
    # 1000个线程 → 8GB内存，大量上下文切换
    with ThreadPoolExecutor(max_workers=1000) as pool:
        results = list(pool.map(io_task, range(1000)))

# ✅ 正确：合理的线程数
def good_thread_count():
    # IO密集型：线程数 = min(CPU*5, 目标并发数)
    cpu_count = os.cpu_count() or 4
    max_threads = min(cpu_count * 5, 100)
    with ThreadPoolExecutor(max_workers=max_threads) as pool:
        results = list(pool.map(io_task, range(1000)))

# ❌ 错误：创建过多进程
def bad_process_count():
    # 64个进程 → 2GB内存，调度开销大
    with ProcessPoolExecutor(max_workers=64) as pool:
        results = list(pool.map(cpu_task, range(1000)))

# ✅ 正确：进程数等于核心数
def good_process_count():
    cpu_count = os.cpu_count() or 4
    with ProcessPoolExecutor(max_workers=cpu_count) as pool:
        results = list(pool.map(cpu_task, range(1000),
                                chunksize=len(range(1000)) // cpu_count))

# 最佳并发数经验公式：
# CPU密集型：max_workers = os.cpu_count()
# IO密集型：max_workers = min(os.cpu_count() * 5, 100)
# 混合型：max_workers = os.cpu_count() + 4 (ThreadPoolExecutor)
```

### 5.6 最佳实践总结

| 场景 | 推荐模型 | 原因 | 注意事项 |
|------|----------|------|----------|
| CPU 密集型计算 | multiprocessing | 绕过 GIL，真正并行 | 进程数=核心数，使用 chunksize |
| 少量 IO (<100) | threading | 简单直接，开销可接受 | 使用 ThreadPoolExecutor |
| 大量 IO (>1000) | asyncio | 单线程极低开销 | 需要异步库支持 |
| 混合型 (IO为主) | asyncio + ThreadPool | IO异步 + 阻塞IO线程池 | 控制线程池大小 |
| 混合型 (CPU为主) | multiprocessing + Queue | CPU并行 + 队列通信 | 使用 Pool.map |
| 大规模混合型 | asyncio + ProcessPool | 异步IO + 进程池计算 | 使用 run_in_executor |
| GUI 应用后台任务 | threading | 避免阻塞主线程 | 使用 Event 信号退出 |
| Web 服务器 | asyncio | 高并发连接处理 | aiohttp/FastAPI |
| 数据处理管道 | multiprocessing | 大数据分片并行处理 | 使用共享内存 |
| 实时系统 | asyncio | 确定性延迟控制 | 避免阻塞调用 |
