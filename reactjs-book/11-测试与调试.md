# 第11章 测试与调试

测试不是可选项——它是保证应用质量、预防回归、让重构变得安全的必要手段。本章将深入讲解 React 19 应用的单元测试、组件测试、端到端测试以及调试技巧。

### 11.1.2 测试纯函数

纯函数是最容易测试的——给定相同的输入，始终返回相同的输出。

```typescript
// utils/todo.ts
export interface Todo {
  id: string;
  text: string;
  completed: boolean;
}

type TodoAction =
  | { type: 'ADD_TODO'; payload: { text: string } }
  | { type: 'TOGGLE_TODO'; payload: { id: string } }
  | { type: 'DELETE_TODO'; payload: { id: string } }
  | { type: 'CLEAR_COMPLETED' };

export function todoReducer(state: Todo[], action: TodoAction): Todo[] {
  switch (action.type) {
    case 'ADD_TODO':
      return [
        ...state,
        {
          id: crypto.randomUUID(),
          text: action.payload.text,
          completed: false,
        },
      ];
    case 'TOGGLE_TODO':
      return state.map((todo) =>
        todo.id === action.payload.id
          ? { ...todo, completed: !todo.completed }
          : todo
      );
    case 'DELETE_TODO':
      return state.filter((todo) => todo.id !== action.payload.id);
    case 'CLEAR_COMPLETED':
      return state.filter((todo) => !todo.completed);
    default:
      return state;
  }
}
```

```typescript
// utils/__tests__/todo.test.ts
import { describe, it, expect } from 'vitest';
import { todoReducer } from '../todo';

describe('todoReducer', () => {
  it('应该添加新的待办事项', () => {
    const initialState: Todo[] = [];
    const state = todoReducer(initialState, {
      type: 'ADD_TODO',
      payload: { text: '学习 React 19' },
    });

    expect(state).toHaveLength(1);
    expect(state[0].text).toBe('学习 React 19');
    expect(state[0].completed).toBe(false);
    expect(state[0].id).toBeDefined();
    // 验证不可变性：初始状态未被修改
    expect(initialState).toHaveLength(0);
  });

  it('应该切换待办事项的完成状态', () => {
    const initialState: Todo[] = [
      { id: '1', text: '学习 React 19', completed: false },
    ];

    const state = todoReducer(initialState, {
      type: 'TOGGLE_TODO',
      payload: { id: '1' },
    });

    expect(state[0].completed).toBe(true);

    // 再次切换
    const nextState = todoReducer(state, {
      type: 'TOGGLE_TODO',
      payload: { id: '1' },
    });
    expect(nextState[0].completed).toBe(false);
  });

  it('应该删除待办事项', () => {
    const initialState: Todo[] = [
      { id: '1', text: '学习 React 19', completed: false },
      { id: '2', text: '编写测试', completed: true },
    ];

    const state = todoReducer(initialState, {
      type: 'DELETE_TODO',
      payload: { id: '1' },
    });

    expect(state).toHaveLength(1);
    expect(state[0].id).toBe('2');
  });

  it('应该清除所有已完成的事项', () => {
    const initialState: Todo[] = [
      { id: '1', text: '学习', completed: true },
      { id: '2', text: '工作', completed: false },
      { id: '3', text: '运动', completed: true },
    ];

    const state = todoReducer(initialState, {
      type: 'CLEAR_COMPLETED',
    });

    expect(state).toHaveLength(1);
    expect(state[0].id).toBe('2');
  });

  it('遇到未知 action 应返回原状态', () => {
    const initialState: Todo[] = [{ id: '1', text: '测试', completed: false }];
    const state = todoReducer(initialState, { type: 'UNKNOWN' } as any);
    expect(state).toBe(initialState);
  });
});
```

### 11.1.3 测试自定义 Hooks

`renderHook` 是 React Testing Library 提供的专门用于测试 hooks 的工具。

```typescript
// hooks/useCounter.ts
import { useState, useCallback } from 'react';

export function useCounter(initialValue = 0) {
  const [count, setCount] = useState(initialValue);

  const increment = useCallback(() => setCount((c) => c + 1), []);
  const decrement = useCallback(() => setCount((c) => c - 1), []);
  const reset = useCallback(() => setCount(initialValue), [initialValue]);

  return { count, increment, decrement, reset };
}
```

```typescript
// hooks/__tests__/useCounter.test.ts
import { renderHook, act } from '@testing-library/react';
import { describe, it, expect } from 'vitest';
import { useCounter } from '../useCounter';

describe('useCounter', () => {
  it('应该使用默认初始值 0', () => {
    const { result } = renderHook(() => useCounter());

    expect(result.current.count).toBe(0);
  });

  it('应该使用自定义初始值', () => {
    const { result } = renderHook(() => useCounter(10));

    expect(result.current.count).toBe(10);
  });

  it('increment 应该增加计数', () => {
    const { result } = renderHook(() => useCounter(0));

    act(() => {
      result.current.increment();
    });

    expect(result.current.count).toBe(1);
  });

  it('decrement 应该减少计数', () => {
    const { result } = renderHook(() => useCounter(5));

    act(() => {
      result.current.decrement();
    });

    expect(result.current.count).toBe(4);
  });

  it('reset 应该重置为初始值', () => {
    const { result } = renderHook(() => useCounter(100));

    act(() => {
      result.current.increment();
      result.current.increment();
    });
    expect(result.current.count).toBe(102);

    act(() => {
      result.current.reset();
    });
    expect(result.current.count).toBe(100);
  });
});
```

**测试异步 hooks：**

```typescript
// hooks/useUser.ts
export function useUser(userId: string) {
  const [state, setState] = useState({
    data: null as User | null,
    isLoading: true,
    error: null as Error | null,
  });

  useEffect(() => {
    let cancelled = false;

    async function fetchUser() {
      setState((prev) => ({ ...prev, isLoading: true }));
      try {
        const response = await fetch(`/api/users/${userId}`);
        const data = await response.json();
        if (!cancelled) {
          setState({ data, isLoading: false, error: null });
        }
      } catch (error) {
        if (!cancelled) {
          setState({ data: null, isLoading: false, error: error as Error });
        }
      }
    }

    fetchUser();
    return () => { cancelled = true; };
  }, [userId]);

  return state;
}
```

```typescript
// hooks/__tests__/useUser.test.ts
import { renderHook, waitFor } from '@testing-library/react';
import { describe, it, expect, vi } from 'vitest';
import { useUser } from '../useUser';

describe('useUser', () => {
  it('应该获取用户数据并更新状态', async () => {
    const mockUser = { id: '1', name: '张三' };
    global.fetch = vi.fn().mockResolvedValue({
      json: () => Promise.resolve(mockUser),
    });

    const { result } = renderHook(() => useUser('1'));

    // 初始状态是 loading
    expect(result.current.isLoading).toBe(true);

    // 等待异步操作完成
    await waitFor(() => {
      expect(result.current.isLoading).toBe(false);
    });

    expect(result.current.data).toEqual(mockUser);
    expect(result.current.error).toBeNull();
  });

  it('应该处理获取失败的情况', async () => {
    global.fetch = vi.fn().mockRejectedValue(new Error('网络错误'));

    const { result } = renderHook(() => useUser('1'));

    await waitFor(() => {
      expect(result.current.isLoading).toBe(false);
    });

    expect(result.current.data).toBeNull();
    expect(result.current.error).toBeInstanceOf(Error);
    expect(result.current.error?.message).toBe('网络错误');
  });
});
```

## 11.2 组件测试

### 11.2.1 测试哲学：按行为测试，而非实现细节

React Testing Library 的核心哲学是：**测试应该模拟用户使用应用的方式，而非测试 React 的内部实现。** 这意味着：

- 按用户可见的文本和角色查询元素，而非 class 或 id
- 测试交互结果，而非 state 变化
- 不要直接测试组件的内部方法或状态

### 11.2.2 查询优先级

```jsx
// 按优先级从高到低排列

// 1. 可访问性查询（推荐）
getByRole('button', { name: /提交/i });    // 按钮
getByRole('heading', { name: /标题/i });   // 标题
getByRole('textbox', { name: /用户名/i }); // 输入框

// 2. 文本查询（次选）
getByText('提交表单');      // 文本匹配
getByLabelText('用户名');    // label 关联

// 3. test-id（备选，仅在无法使用上述方式时）
getByTestId('submit-button');
```

### 11.2.3 userEvent vs fireEvent

始终优先使用 `@testing-library/user-event`，它模拟的是真实的用户操作。

```typescript
// 错误：fireEvent 只触发事件，不模拟真实行为
fireEvent.change(input, { target: { value: 'hello' } });

// 正确：userEvent 模拟真实键盘输入
await user.type(input, 'hello');

// userEvent 的优势：
// - 触发键盘事件（keyDown, keyPress, keyUp）
// - 触发 focus/blur 事件
// - 处理 debounce 等异步行为
```

### 11.2.4 异步测试

```typescript
import { render, screen, waitFor } from '@testing-library/react';
import userEvent from '@testing-library/user-event';
import { http, HttpResponse } from 'msw';
import { setupServer } from 'msw/node';

// 使用 MSW 模拟 API
const server = setupServer(
  http.get('/api/users', () => {
    return HttpResponse.json([
      { id: '1', name: '张三' },
      { id: '2', name: '李四' },
    ]);
  })
);

beforeAll(() => server.listen());
afterEach(() => server.resetHandlers());
afterAll(() => server.close());

describe('UserList', () => {
  it('加载完成后显示用户列表', async () => {
    render(<UserList />);

    // 初始应显示加载状态
    expect(screen.getByText(/加载中/i)).toBeInTheDocument();

    // 等待用户列表出现
    await waitFor(() => {
      expect(screen.getByText('张三')).toBeInTheDocument();
      expect(screen.getByText('李四')).toBeInTheDocument();
    });

    // 加载指示器应消失
    expect(screen.queryByText(/加载中/i)).not.toBeInTheDocument();
  });

  it('API 失败时显示错误信息', async () => {
    server.use(
      http.get('/api/users', () => {
        return new HttpResponse(null, { status: 500 });
      })
    );

    render(<UserList />);

    await waitFor(() => {
      expect(screen.getByText(/加载失败/i)).toBeInTheDocument();
    });
  });
});
```

### 11.3.2 Page Object 模式

```typescript
// e2e/pages/LoginPage.ts
import { Page, Locator } from '@playwright/test';

export class LoginPage {
  readonly page: Page;
  readonly usernameInput: Locator;
  readonly passwordInput: Locator;
  readonly submitButton: Locator;
  readonly errorMessage: Locator;

  constructor(page: Page) {
    this.page = page;
    this.usernameInput = page.getByLabel(/用户名/i);
    this.passwordInput = page.getByLabel(/密码/i);
    this.submitButton = page.getByRole('button', { name: /登录/i });
    this.errorMessage = page.getByRole('alert');
  }

  async goto() {
    await this.page.goto('/login');
  }

  async login(username: string, password: string) {
    await this.usernameInput.fill(username);
    await this.passwordInput.fill(password);
    await this.submitButton.click();
  }

  async getErrorMessage() {
    return this.errorMessage.textContent();
  }
}
```

```typescript
// e2e/specs/login.spec.ts
import { test, expect } from '@playwright/test';
import { LoginPage } from '../pages/LoginPage';

test.describe('登录流程', () => {
  test('成功登录后跳转到仪表盘', async ({ page }) => {
    const loginPage = new LoginPage(page);
    await loginPage.goto();
    await loginPage.login('admin', 'password123');

    await expect(page).toHaveURL(/\/dashboard/);
    await expect(page.getByText('欢迎回来')).toBeVisible();
  });

  test('错误密码时显示错误提示', async ({ page }) => {
    const loginPage = new LoginPage(page);
    await loginPage.goto();
    await loginPage.login('admin', 'wrong');

    const error = await loginPage.getErrorMessage();
    expect(error).toContain('用户名或密码错误');
  });
});
```

## 11.4 调试工具

### 11.4.1 React DevTools

React DevTools 是调试 React 应用的首选工具，提供三个核心面板：

**Components 面板：**
- 查看组件树结构和层级关系
- 检查组件的 props、state、hooks 值
- 搜索和过滤组件
- 查看组件源代码位置

**Profiler 面板：**
- 记录渲染性能
- 识别不必要的重新渲染
- 查看每个组件的渲染耗时
- 火焰图直观展示渲染瀑布

**Source 面板（React 19 新功能）：**
- 直接查看组件的源码
- 在 DevTools 中编辑组件并热更新

### 11.4.2 Chrome DevTools

**Breakpoints 调试：**
```javascript
// 在代码中添加 debugger 语句
function handleSubmit(data) {
  debugger; // 浏览器会在此处暂停
  // 处理提交逻辑
}

// 条件断点（在 Sources 面板中设置）
// 只在 count > 10 时暂停
if (count > 10) { debugger; }
```

**Network 面板：**
- 检查 API 请求和响应
- 模拟慢速网络
- 检查请求头、cookies
- 拦截和修改请求

**Performance 面板：**
- 记录页面加载和交互性能
- 分析 JavaScript 执行时间
- 识别布局抖动（Layout Thrashing）
- 检查长任务（Long Tasks）

## 11.5 常见测试陷阱

### 陷阱1：测试实现细节而非用户行为

```typescript
// 错误：直接测试内部状态
expect(component.state('count')).toBe(1);

// 正确：测试用户可见的结果
expect(screen.getByText('1')).toBeInTheDocument();
```

### 陷阱2：未清理 Mock

```typescript
// 错误：Mock 在测试间泄漏
afterAll(() => {
  vi.restoreAllMocks(); // 只在所有测试结束后恢复
});

// 正确：每个测试后清理
afterEach(() => {
  vi.clearAllMocks();
});
```

### 陷阱3：异步测试时序问题

```typescript
// 错误：没有等待异步操作
user.click(button);
expect(screen.getByText('加载完成')).toBeInTheDocument();

// 正确：使用 waitFor 等待
await user.click(button);
await waitFor(() => {
  expect(screen.getByText('加载完成')).toBeInTheDocument();
});
```

### 陷阱4：过度 Mock

**过度 Mock 的问题：** 当你 mock 了太多依赖，测试就变成了在测试你的 mock，而非真实代码。尽量使用真实实现或集成测试。

### 陷阱5：不稳定的测试

常见原因及解决方案：
- **时序依赖**：使用 `waitFor` 和 `findBy` 而非固定 `setTimeout`
- **网络依赖**：使用 MSW 等 mock 库模拟一致的网络响应
- **随机数据**：在测试中使用固定数据，或使用 `faker.seed()` 确保可重复

## 小结

测试金字塔的每一层都有其独特的价值：单元测试提供快速的反馈，组件测试验证用户交互的正确性，E2E 测试确保关键流程的端到端可用性。结合 React DevTools 和 Chrome DevTools 等调试工具，你可以高效地定位和修复问题。

记住测试的核心原则：**测试行为，而非实现。** 遵循这一原则，你的测试套件将更加稳健、可维护，并在重构时真正发挥价值。
